43template <uint32_t State>
44std::unique_ptr<wrapper::TcpClient> TcpClientBuilder<State>::build() {
45#if __cplusplus >= 202002L
46 if constexpr (!((State & BuilderState::Ready) == BuilderState::Ready)) {
47 throw diagnostics::BuilderException(
48 "TcpClientBuilder: Mandatory handlers (on_data/on_message and on_error) must be registered.");
49 }
50#endif
51
52 std::unique_ptr<wrapper::TcpClient> client;
53 if (independent_context_) {
54 client = std::make_unique<wrapper::TcpClient>(host_, port_, std::make_shared<boost::asio::io_context>());
55 client->manage_external_context(true);
56 } else {
57 client = std::make_unique<wrapper::TcpClient>(host_, port_);
58 }
59
60 if (this->on_data_) client->on_data(this->on_data_);
61 if (this->on_data_batch_) client->on_data_batch(this->on_data_batch_);
62 if (this->on_connect_) client->on_connect(this->on_connect_);
63 if (this->on_disconnect_) client->on_disconnect(this->on_disconnect_);
64 if (this->on_error_) client->on_error(this->on_error_);
65 if (this->on_backpressure_) client->on_backpressure(this->on_backpressure_);
66
67 client->retry_interval(retry_interval_);
68 client->max_retries(max_retries_);
69 client->connection_timeout(connection_timeout_);
70
71 if (this->bp_strategy_set_) client->backpressure_strategy(this->bp_strategy_);
72 client->backpressure_threshold(this->get_effective_backpressure_threshold());
73
74 if (this->framer_factory_) {
75 client->framer(this->framer_factory_());
76 }
77 if (this->on_message_) {
78 client->on_message(std::move(this->on_message_));
79 }
80 if (this->on_message_batch_) {
81 client->on_message_batch(std::move(this->on_message_batch_));
82 }
83
84 if (auto_start_) {
85 client->auto_start(true);
86 }
87
88 return client;
89}
