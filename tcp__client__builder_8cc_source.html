42std::unique_ptr<wrapper::TcpClient> TcpClientBuilder::build() {
43 std::unique_ptr<wrapper::TcpClient> client;
44 if (independent_context_) {
45 client = std::make_unique<wrapper::TcpClient>(host_, port_, std::make_shared<boost::asio::io_context>());
46 client->manage_external_context(true);
47 } else {
48 client = std::make_unique<wrapper::TcpClient>(host_, port_);
49 }
50
51 if (on_data_) client->on_data(on_data_);
52 if (on_connect_) client->on_connect(on_connect_);
53 if (on_disconnect_) client->on_disconnect(on_disconnect_);
54 if (on_error_) client->on_error(on_error_);
55
56 client->retry_interval(retry_interval_);
57 client->max_retries(max_retries_);
58 client->connection_timeout(connection_timeout_);
59
60 if (framer_factory_) {
61 client->framer(framer_factory_());
62 }
63 if (on_message_) {
64 client->on_message(std::move(on_message_));
65 }
66
67 if (auto_start_) {
68 client->auto_start(true);
69 }
70
71 return client;
72}
