46 void IoContextManager::start() {
47  std::shared_ptr<IoContext> context;
48  {
49  std::unique_lock<std::mutex> lock(mutex_);
50 
51  // If we don't own the context, we treat start() as a check/no-op.
52  if (!owns_context_ && ioc_) {
53  if (ioc_->stopped()) {
54  UNILINK_LOG_WARNING("io_context_manager", "start",
55  "External io_context is stopped. The external owner must restart/run it.");
56  } else {
57  UNILINK_LOG_DEBUG("io_context_manager", "start",
58  "IoContextManager using external context. Thread creation skipped.");
59  }
60  // running_ represents "internal thread running", so we leave it false.
61  // Callers checking is_running() will get false, which is correct (we aren't running it).
62  return;
63  }
64 
65  cv_.wait(lock, [this] { return !stopping_; });
66 
67  if (running_) {
68  UNILINK_LOG_DEBUG("io_context_manager", "start", "IoContextManager already running, ignoring start call.");
69  return;
70  }
71 
72  // Prevent self-join if start() is called from within the IO thread
73  if (io_thread_.joinable() && io_thread_.get_id() == std::this_thread::get_id()) {
74  UNILINK_LOG_ERROR("io_context_manager", "start", "Cannot restart IoContextManager from within its own thread.");
75  return;
76  }
77 
78  if (!ioc_) {
79  ioc_ = std::make_shared<IoContext>();
80  owns_context_ = true;
81  }
82 
83  if (ioc_->stopped()) {
84  ioc_->restart();
85  }
86  work_guard_ = std::make_unique<WorkGuard>(ioc_->get_executor());
87  context = ioc_; // Capture shared_ptr to io_context
88 
89  // If there was a previous thread, ensure it's joined before starting a new one
90  if (io_thread_.joinable()) {
91  io_thread_.join();
92  }
93 
94  // Launch the io_context run in a new thread
95  io_thread_ = std::thread([this, context]() {
96  try {
97  UNILINK_LOG_DEBUG("io_context_manager", "start", "IoContext thread started.");
98  context->run();
99  UNILINK_LOG_DEBUG("io_context_manager", "start", "IoContext thread finished running.");
100  } catch (const std::exception& e) {
101  UNILINK_LOG_ERROR("io_context_manager", "run", "Thread error: " + std::string(e.what()));
102  }
103  running_.store(false); // Mark as not running when thread exits
104  });
105  running_.store(true); // Mark as running after thread is launched
106  }
107 }

109 void IoContextManager::stop() {
110  std::thread worker; // Declare outside the lock
111  {
112  std::lock_guard<std::mutex> lock(mutex_);
113 
114  // If we don't own the context, we just return.
115  if (!owns_context_ && ioc_) {
116  // running_ should already be false, but strict no-op is safest.
117  return;
118  }
119 
120  if (!running_.load() && !io_thread_.joinable()) {
121  UNILINK_LOG_DEBUG("io_context_manager", "stop",
122  "IoContextManager not running or thread not joinable, ignoring stop call.");
123  return;
124  }
125 
126  stopping_ = true;
127 
128  // Reset work_guard to allow io_context to stop once all tasks are done
129  if (work_guard_) {
130  work_guard_.reset();
131  }
132 
133  // Stop io_context only if we own it
134  if (ioc_ && owns_context_) {
135  ioc_->stop();
136  }
137 
138  // Move the thread handle out of the protected member to join it outside the lock
139  if (io_thread_.joinable()) {
140  // Prevent self-join if stop() is called from within the IO thread
141  if (io_thread_.get_id() != std::this_thread::get_id()) {
142  worker = std::move(io_thread_);
143  } else {
144  UNILINK_LOG_ERROR("io_context_manager", "stop",
145  "Cannot join IoContext thread from within itself. Skipping join.");
146 
147  // REVERT STATE: Failed to stop.
148  stopping_ = false;
149 
150  // NOTIFY: Wake up anyone waiting in start()
151  cv_.notify_all();
152  return;
153  }
154  }
155  } // Mutex is released here
156 
157  // Join the thread outside the lock to prevent deadlocks
158  if (worker.joinable()) {
159  UNILINK_LOG_DEBUG("io_context_manager", "stop", "Joining IoContext thread.");
160  worker.join();
161  UNILINK_LOG_DEBUG("io_context_manager", "stop", "IoContext thread joined.");
162  }
163 
164  // After the thread has joined, perform final cleanup that might require the mutex again
165  {
166  std::lock_guard<std::mutex> lock(mutex_);
167  stopping_ = false;
168 
169  // Ensure running flag is sync with reality
170  if (!worker.joinable() && !io_thread_.joinable()) {
171  running_.store(false);
172  }
173  }
174  cv_.notify_all();
175 }
