46 void IoContextManager::start() {
47  std::shared_ptr<IoContext> context;
48  {
49  std::lock_guard<std::mutex> lock(mutex_);
50  if (running_) {
51  UNILINK_LOG_DEBUG("io_context_manager", "start", "IoContextManager already running, ignoring start call.");
52  return;
53  }
54 
55  if (!ioc_) {
56  ioc_ = std::make_shared<IoContext>();
57  owns_context_ = true;
58  }
59 
60  if (ioc_->stopped()) {
61  ioc_->restart();
62  }
63  work_guard_ = std::make_unique<WorkGuard>(ioc_->get_executor());
64  context = ioc_; // Capture shared_ptr to io_context
65 
66  // If there was a previous thread, ensure it's joined before starting a new one
67  if (io_thread_.joinable()) {
68  io_thread_.join();
69  }
70 
71  // Launch the io_context run in a new thread
72  io_thread_ = std::thread([this, context]() {
73  try {
74  UNILINK_LOG_DEBUG("io_context_manager", "start", "IoContext thread started.");
75  context->run();
76  UNILINK_LOG_DEBUG("io_context_manager", "start", "IoContext thread finished running.");
77  } catch (const std::exception& e) {
78  UNILINK_LOG_ERROR("io_context_manager", "run", "Thread error: " + std::string(e.what()));
79  }
80  running_.store(false); // Mark as not running when thread exits
81  });
82  running_.store(true); // Mark as running after thread is launched
83  }
84 }

86 void IoContextManager::stop() {
87  std::thread worker; // Declare outside the lock
88  {
89  std::lock_guard<std::mutex> lock(mutex_);
90  if (!running_.load() && !io_thread_.joinable()) {
91  UNILINK_LOG_DEBUG("io_context_manager", "stop",
92  "IoContextManager not running or thread not joinable, ignoring stop call.");
93  return;
94  }
95 
96  // Reset work_guard to allow io_context to stop once all tasks are done
97  if (work_guard_) {
98  work_guard_.reset();
99  }
100 
101  // Stop io_context
102  if (ioc_) {
103  ioc_->stop();
104  }
105 
106  // Move the thread handle out of the protected member to join it outside the lock
107  if (io_thread_.joinable()) {
108  worker = std::move(io_thread_);
109  }
110 
111  running_.store(false);
112  } // Mutex is released here
113 
114  // Join the thread outside the lock to prevent deadlocks
115  if (worker.joinable()) {
116  UNILINK_LOG_DEBUG("io_context_manager", "stop", "Joining IoContext thread.");
117  worker.join();
118  UNILINK_LOG_DEBUG("io_context_manager", "stop", "IoContext thread joined.");
119  }
120 
121  // After the thread has joined, perform final cleanup that might require the mutex again
122  {
123  std::lock_guard<std::mutex> lock(mutex_);
124  if (owns_context_) {
125  ioc_.reset(); // Reset owned io_context
126  }
127  // The work_guard_ should already be reset at the beginning of stop to unblock io_context.run()
128  // If it's a shared context that is not owned, it should not be reset here.
129  // If the context is owned, ioc_.reset() effectively cleans up the underlying Boost.Asio context.
130  }
131 }
