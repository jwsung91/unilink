48 std::unique_ptr<wrapper::TcpServer> TcpServerBuilder::build() {
49  // Client limit validation
50  if (!client_limit_set_) {
51  throw std::runtime_error(
52  "Client limit must be set before building server. Use single_client(), multi_client(n), or "
53  "unlimited_clients()");
54  }
55 
56  std::shared_ptr<boost::asio::io_context> external_ioc;
57  // IoContext management
58  if (use_independent_context_) {
59  // Use independent IoContext (for test isolation)
60  auto independent_context = common::IoContextManager::instance().create_independent_context();
61  external_ioc = std::shared_ptr<boost::asio::io_context>(std::move(independent_context));
62  } else {
63  // Automatically initialize IoContextManager (default behavior)
64  AutoInitializer::ensure_io_context_running();
65  }
66 
67  auto server = external_ioc ? std::make_unique<wrapper::TcpServer>(port_, external_ioc)
68  : std::make_unique<wrapper::TcpServer>(port_);
69 
70  if (external_ioc) {
71  server->set_manage_external_context(use_independent_context_);
72  }
73 
74  // Apply client limit configuration
75  UNILINK_LOG_DEBUG(
76  "tcp_server_builder", "build",
77  "client_limit_set=" + std::to_string(client_limit_set_) + ", max_clients=" + std::to_string(max_clients_));
78  if (client_limit_set_) {
79  if (max_clients_ == 0) {
80  // Unlimited clients
81  UNILINK_LOG_DEBUG("tcp_server_builder", "build", "Setting unlimited clients");
82  server->set_unlimited_clients();
83  } else {
84  // Limited clients
85  UNILINK_LOG_DEBUG("tcp_server_builder", "build", "Setting client limit to " + std::to_string(max_clients_));
86  server->set_client_limit(max_clients_);
87  }
88  } else {
89  UNILINK_LOG_DEBUG("tcp_server_builder", "build", "No client limit set");
90  }
91 
92  // Set callbacks
93  if (on_data_) {
94  server->on_data(on_data_);
95  }
96 
97  if (on_connect_) {
98  server->on_connect(on_connect_);
99  }
100 
101  if (on_disconnect_) {
102  server->on_disconnect(on_disconnect_);
103  }
104 
105  if (on_error_) {
106  server->on_error(on_error_);
107  }
108 
109  // Multi-client callback configuration
110  if (on_multi_connect_) {
111  server->on_multi_connect(on_multi_connect_);
112  }
113 
114  if (on_multi_data_) {
115  server->on_multi_data(on_multi_data_);
116  }
117 
118  if (on_multi_disconnect_) {
119  server->on_multi_disconnect(on_multi_disconnect_);
120  }
121 
122  // Port retry configuration
123  UNILINK_LOG_DEBUG("tcp_server_builder", "build", "enable_port_retry=" + std::to_string(enable_port_retry_));
124  if (enable_port_retry_) {
125  UNILINK_LOG_DEBUG("tcp_server_builder", "build",
126  "Setting port retry: max=" + std::to_string(max_port_retries_) +
127  ", interval=" + std::to_string(port_retry_interval_ms_) + "ms");
128  server->enable_port_retry(true, max_port_retries_, port_retry_interval_ms_);
129  }
130 
131  // Apply configuration and auto-start if requested
132  server->auto_manage(auto_manage_);
133 
134  return server;
135 }
