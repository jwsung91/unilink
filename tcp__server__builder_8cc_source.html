44std::unique_ptr<wrapper::TcpServer> TcpServerBuilder::build() {
45 std::unique_ptr<wrapper::TcpServer> server;
46 if (independent_context_) {
47 server = std::make_unique<wrapper::TcpServer>(port_, std::make_shared<boost::asio::io_context>());
48 server->manage_external_context(true);
49 } else {
50 server = std::make_unique<wrapper::TcpServer>(port_);
51 }
52
53 if (on_data_) server->on_data(on_data_);
54 if (on_connect_) server->on_connect(on_connect_);
55 if (on_disconnect_) server->on_disconnect(on_disconnect_);
56 if (on_error_) server->on_error(on_error_);
57
58 if (framer_factory_) {
59 server->framer(framer_factory_);
60 }
61
62 if (on_message_) {
63 server->on_message(on_message_);
64 }
65
66 if (enable_port_retry_) {
67 server->port_retry(true, max_port_retries_, port_retry_interval_ms_);
68 }
69
70 if (idle_timeout_.count() > 0) {
71 server->idle_timeout(idle_timeout_);
72 }
73
74 if (client_limit_set_) {
75 if (max_clients_ == 0)
76 server->unlimited_clients();
77 else
78 server->max_clients(max_clients_);
79 }
80
81 if (auto_start_) {
82 server->auto_start(true);
83 }
84
85 return server;
86}

110TcpServerBuilder& TcpServerBuilder::max_clients(size_t max) {
111 if (max == 0) throw std::invalid_argument("Client limit cannot be 0");
112 if (max == 1) throw std::invalid_argument("Use single_client() for 1 client limit");
113 max_clients_ = max;
114 client_limit_set_ = true;
115 return *this;
116}
