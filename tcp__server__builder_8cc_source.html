50template <uint32_t State>
51std::unique_ptr<wrapper::TcpServer> TcpServerBuilder<State>::build() {
52 std::unique_ptr<wrapper::TcpServer> server;
53 if (independent_context_) {
54 server = std::make_unique<wrapper::TcpServer>(port_, std::make_shared<boost::asio::io_context>());
55 server->manage_external_context(true);
56 } else {
57 server = std::make_unique<wrapper::TcpServer>(port_);
58 }
59
60 if (this->on_data_) server->on_data(this->on_data_);
61 if (this->on_data_batch_) server->on_data_batch(this->on_data_batch_);
62 if (this->on_connect_) server->on_connect(this->on_connect_);
63 if (this->on_disconnect_) server->on_disconnect(this->on_disconnect_);
64 if (this->on_error_) server->on_error(this->on_error_);
65 if (this->on_backpressure_) server->on_backpressure(this->on_backpressure_);
66
67 if (client_limit_enabled_) {
68 server->max_clients(max_clients_);
69 }
70
71 server->bind_address(bind_address_);
72 server->port_retry(port_retry_enabled_, static_cast<int>(max_port_retries_),
73 static_cast<int>(port_retry_interval_ms_));
74 if (idle_timeout_set_) server->idle_timeout(idle_timeout_);
75 server->tcp_no_delay(tcp_no_delay_);
76 server->keep_alive(keep_alive_);
77 server->send_buffer_size(send_buffer_size_);
78 server->receive_buffer_size(receive_buffer_size_);
79
80 if (this->bp_strategy_set_) server->backpressure_strategy(this->bp_strategy_);
81 server->backpressure_threshold(this->get_effective_backpressure_threshold());
82
83 if (this->framer_factory_) {
84 server->framer(this->framer_factory_);
85 }
86 if (this->on_message_) {
87 server->on_message(std::move(this->on_message_));
88 }
89 if (this->on_message_batch_) {
90 server->on_message_batch(std::move(this->on_message_batch_));
91 }
92
93 if (auto_start_) {
94 server->auto_start(true);
95 }
96
97 return server;
98}
