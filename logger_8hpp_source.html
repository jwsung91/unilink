265#define UNILINK_LOG_PERF_START(component, operation) \
266 auto _perf_start_##operation = \
267 (unilink::diagnostics::Logger::instance().level() <= unilink::diagnostics::LogLevel::DEBUG) \
268 ? std::chrono::high_resolution_clock::now() \
269 : std::chrono::high_resolution_clock::time_point()
270
271#define UNILINK_LOG_PERF_END(component, operation) \
272 do { \
273 if (unilink::diagnostics::Logger::instance().level() <= unilink::diagnostics::LogLevel::DEBUG) { \
274 auto _perf_end_##operation = std::chrono::high_resolution_clock::now(); \
275 using _us_t = std::chrono::microseconds; \
276 auto _diff_##operation = _perf_end_##operation - _perf_start_##operation; \
277 auto _perf_duration_##operation = std::chrono::duration_cast<_us_t>(_diff_##operation).count(); \
278 UNILINK_LOG_DEBUG(component, operation, "Duration: " + std::to_string(_perf_duration_##operation) + " μs"); \
279 } \
280 } while (0)
