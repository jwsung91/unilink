262#define UNILINK_LOG(level, component, operation, message) \
263 do { \
264 const auto unilink_log_level = (level); \
265 if (unilink::diagnostics::Logger::instance().should_log(unilink_log_level)) { \
266 unilink::diagnostics::Logger::instance().log(unilink_log_level, component, operation, message); \
267 } \
268 } while (0)

292#define UNILINK_LOG_PERF_START(component, operation) \
293 auto _perf_start_##operation = \
294 (unilink::diagnostics::Logger::instance().should_log(unilink::diagnostics::LogLevel::DEBUG)) \
295 ? std::chrono::high_resolution_clock::now() \
296 : std::chrono::high_resolution_clock::time_point()
297
298#define UNILINK_LOG_PERF_END(component, operation) \
299 do { \
300 if (unilink::diagnostics::Logger::instance().should_log(unilink::diagnostics::LogLevel::DEBUG)) { \
301 auto _perf_end_##operation = std::chrono::high_resolution_clock::now(); \
302 using _us_t = std::chrono::microseconds; \
303 auto _diff_##operation = _perf_end_##operation - _perf_start_##operation; \
304 auto _perf_duration_##operation = std::chrono::duration_cast<_us_t>(_diff_##operation).count(); \
305 UNILINK_LOG(unilink::diagnostics::LogLevel::DEBUG, component, operation, \
306 "Duration: " + std::to_string(_perf_duration_##operation) + " μs"); \
307 } \
308 } while (0)
