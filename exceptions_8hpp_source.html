34class UNILINK_API UnilinkException : public std::runtime_error {
35 public:
36 explicit UnilinkException(const std::string& message, const std::string& component = "",
37 const std::string& operation = "")
38 : std::runtime_error(message), component_(component), operation_(operation) {}
39
40 virtual ~UnilinkException() noexcept;
41
42 const std::string& component() const noexcept { return component_; }
43 const std::string& operation() const noexcept { return operation_; }
44
45 std::string full_message() const {
46 if (component_.empty() && operation_.empty()) {
47 return what();
48 }
49 if (operation_.empty()) {
50 return std::format("[{}] {}", component_, what());
51 }
52 if (component_.empty()) {
53 return std::format("{} (operation: {})", what(), operation_);
54 }
55 return std::format("[{}] {} (operation: {})", component_, what(), operation_);
56 }
57
58 private:
59 std::string component_;
60 std::string operation_;
61};
