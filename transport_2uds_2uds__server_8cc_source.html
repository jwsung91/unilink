116 void UdsServer::start() {
117  if (impl_->state_.state() == base::LinkState::Listening) return;
118 
119  impl_->stopping_ = false;
120 
121  if (!impl_->cfg_.is_valid()) {
122  UNILINK_LOG_ERROR("uds_server", "start", "Invalid UDS server configuration or socket path");
123  impl_->state_.set_state(base::LinkState::Error);
124  impl_->notify_state();
125  return;
126  }
127 
128  // Cleanup old socket file if exists
129  std::remove(impl_->cfg_.socket_path.c_str());
130 
131  boost::system::error_code ec;
132  impl_->acceptor_->open(uds(), ec);
133  if (ec) {
134  UNILINK_LOG_ERROR("uds_server", "start", "Failed to open acceptor: " + ec.message());
135  impl_->state_.set_state(base::LinkState::Error);
136  impl_->notify_state();
137  return;
138  }
139 
140  uds::endpoint endpoint;
141  try {
142  endpoint = uds::endpoint(impl_->cfg_.socket_path);
143  } catch (const std::exception& e) {
144  UNILINK_LOG_ERROR("uds_server", "start", "Invalid UDS endpoint: " + std::string(e.what()));
145  impl_->state_.set_state(base::LinkState::Error);
146  impl_->notify_state();
147  return;
148  }
149 
150  impl_->acceptor_->bind(endpoint, ec);
151  if (ec) {
152  UNILINK_LOG_ERROR("uds_server", "start", "Failed to bind to " + impl_->cfg_.socket_path + ": " + ec.message());
153  impl_->state_.set_state(base::LinkState::Error);
154  impl_->notify_state();
155  return;
156  }
157 
158  impl_->acceptor_->listen(net::socket_base::max_listen_connections, ec);
159  if (ec) {
160  UNILINK_LOG_ERROR("uds_server", "start", "Failed to listen: " + ec.message());
161  impl_->state_.set_state(base::LinkState::Error);
162  impl_->notify_state();
163  return;
164  }
165 
166  impl_->state_.set_state(base::LinkState::Listening);
167  impl_->notify_state();
168 
169  if (impl_->owns_ioc_ && !impl_->ioc_thread_.joinable()) {
170  if (impl_->ioc_->stopped()) {
171  impl_->ioc_->restart();
172  }
173  impl_->work_guard_ =
174  std::make_unique<net::executor_work_guard<net::io_context::executor_type>>(net::make_work_guard(*impl_->ioc_));
175  impl_->ioc_thread_ = std::thread([this]() {
176  try {
177  impl_->ioc_->run();
178  } catch (...) {
179  }
180  });
181  }
182 
183  net::post(impl_->ioc_->get_executor(), [self = shared_from_this()]() { self->impl_->do_accept(self); });
184 }

186 void UdsServer::stop() {
187  bool already_stopping = impl_->stopping_.exchange(true);
188  if (already_stopping) return;
189 
190  boost::system::error_code ec;
191  impl_->acceptor_->close(ec);
192 
193  {
194  std::lock_guard<std::mutex> lock(impl_->sessions_mutex_);
195  impl_->on_bytes_ = nullptr;
196  impl_->on_state_ = nullptr;
197  impl_->on_bp_ = nullptr;
198  impl_->on_multi_connect_ = nullptr;
199  impl_->on_multi_data_ = nullptr;
200  impl_->on_multi_disconnect_ = nullptr;
201  }
202 
203  // Cleanup UDS socket file on stop
204  std::remove(impl_->cfg_.socket_path.c_str());
205 
206  // Release work guard if owned
207  if (impl_->owns_ioc_) {
208  impl_->work_guard_.reset();
209  if (impl_->ioc_) {
210  impl_->ioc_->stop();
211  }
212  }
213 
214  std::vector<std::shared_ptr<UdsServerSession>> sessions_to_stop;
215  {
216  std::lock_guard<std::mutex> lock(impl_->sessions_mutex_);
217  for (auto& pair : impl_->sessions_) {
218  sessions_to_stop.push_back(pair.second);
219  }
220  impl_->sessions_.clear();
221  }
222 
223  for (auto& session : sessions_to_stop) {
224  session->stop();
225  }
226 
227  if (impl_->owns_ioc_ && impl_->ioc_thread_.joinable()) {
228  if (std::this_thread::get_id() != impl_->ioc_thread_.get_id()) {
229  impl_->ioc_thread_.join();
230  } else {
231  impl_->ioc_thread_.detach();
232  }
233  }
234 
235  impl_->state_.set_state(base::LinkState::Idle);
236  impl_->notify_state();
237 }

324 void UdsServer::Impl::do_accept(std::shared_ptr<UdsServer> self) {
325  acceptor_->async_accept([self](const boost::system::error_code& ec, uds::socket socket) {
326  if (!self || self->impl_->stopping_) return;
327 
328  if (!ec) {
329  size_t client_id;
330  auto session = std::make_shared<UdsServerSession>(*self->impl_->ioc_, std::move(socket),
331  self->impl_->cfg_.backpressure_threshold,
332  self->impl_->cfg_.idle_timeout_ms);
333 
334  {
335  std::lock_guard<std::mutex> lock(self->impl_->sessions_mutex_);
336  client_id = self->impl_->next_client_id_++;
337  self->impl_->sessions_[client_id] = session;
338  }
339 
340  std::weak_ptr<UdsServer> weak_self = self;
341  session->on_bytes([weak_self, client_id](memory::ConstByteSpan data) {
342  auto s = weak_self.lock();
343  if (!s) return;
344  MultiClientDataHandler data_handler;
345  OnBytes bytes_handler;
346  {
347  std::lock_guard<std::mutex> lock(s->impl_->sessions_mutex_);
348  data_handler = s->impl_->on_multi_data_;
349  bytes_handler = s->impl_->on_bytes_;
350  }
351  if (data_handler) data_handler(client_id, data);
352  if (bytes_handler) bytes_handler(data);
353  });
354 
355  session->on_close([weak_self, client_id]() {
356  auto s = weak_self.lock();
357  if (!s) return;
358  MultiClientDisconnectHandler disconnect_handler;
359  {
360  std::lock_guard<std::mutex> lock(s->impl_->sessions_mutex_);
361  s->impl_->sessions_.erase(client_id);
362  disconnect_handler = s->impl_->on_multi_disconnect_;
363  }
364  if (disconnect_handler) disconnect_handler(client_id);
365  });
366 
367  session->start();
368 
369  MultiClientConnectHandler connect_handler;
370  {
371  std::lock_guard<std::mutex> lock(self->impl_->sessions_mutex_);
372  connect_handler = self->impl_->on_multi_connect_;
373  }
374  if (connect_handler) connect_handler(client_id, "UDS Client");
375 
376  // Continue accepting
377  auto* impl = self->impl_.get();
378  impl->do_accept(self);
379  } else {
380  // Log only real errors, not operation_aborted
381  if (ec != boost::asio::error::operation_aborted) {
382  UNILINK_LOG_ERROR("uds_server", "accept", "Accept failed: " + ec.message());
383  }
384  }
385  });
386 }
