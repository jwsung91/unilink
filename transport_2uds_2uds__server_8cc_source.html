116void UdsServer::start() {
117 if (impl_->state_.state() == base::LinkState::Listening) return;
118
119 impl_->stopping_ = false;
120
121 if (!impl_->cfg_.is_valid()) {
122 UNILINK_LOG_ERROR("uds_server", "start", "Invalid UDS server configuration or socket path");
123 impl_->state_.set_state(base::LinkState::Error);
124 impl_->notify_state();
125 return;
126 }
127
128 // Cleanup old socket file if exists
129 std::remove(impl_->cfg_.socket_path.c_str());
130
131 boost::system::error_code ec;
132 impl_->acceptor_->open(uds(), ec);
133 if (ec) {
134 UNILINK_LOG_ERROR("uds_server", "start", "Failed to open acceptor: " + ec.message());
135 impl_->state_.set_state(base::LinkState::Error);
136 impl_->notify_state();
137 return;
138 }
139
140 uds::endpoint endpoint;
141 try {
142 endpoint = uds::endpoint(impl_->cfg_.socket_path);
143 } catch (const std::exception& e) {
144 UNILINK_LOG_ERROR("uds_server", "start", "Invalid UDS endpoint: " + std::string(e.what()));
145 impl_->state_.set_state(base::LinkState::Error);
146 impl_->notify_state();
147 return;
148 }
149
150 impl_->acceptor_->bind(endpoint, ec);
151 if (ec) {
152 UNILINK_LOG_ERROR("uds_server", "start", "Failed to bind to " + impl_->cfg_.socket_path + ": " + ec.message());
153 impl_->state_.set_state(base::LinkState::Error);
154 impl_->notify_state();
155 return;
156 }
157
158 impl_->acceptor_->listen(net::socket_base::max_listen_connections, ec);
159 if (ec) {
160 UNILINK_LOG_ERROR("uds_server", "start", "Failed to listen: " + ec.message());
161 impl_->state_.set_state(base::LinkState::Error);
162 impl_->notify_state();
163 return;
164 }
165
166 impl_->state_.set_state(base::LinkState::Listening);
167 impl_->notify_state();
168
169 if (impl_->owns_ioc_ && !impl_->ioc_thread_.joinable()) {
170 if (impl_->ioc_->stopped()) {
171 impl_->ioc_->restart();
172 }
173 impl_->work_guard_ =
174 std::make_unique<net::executor_work_guard<net::io_context::executor_type>>(net::make_work_guard(*impl_->ioc_));
175 impl_->ioc_thread_ = std::thread([this]() {
176 try {
177 impl_->ioc_->run();
178 } catch (...) {
179 }
180 });
181 }
182
183 net::post(impl_->ioc_->get_executor(), [self = shared_from_this()]() { self->impl_->do_accept(self); });
184}

186void UdsServer::stop() {
187 bool already_stopping = impl_->stopping_.exchange(true);
188 if (already_stopping) return;
189
190 // 1. Close acceptor first to prevent new connections
191 boost::system::error_code ec;
192 impl_->acceptor_->close(ec);
193
194 // 2. Clear callbacks to prevent new user-level events
195 {
196 std::lock_guard<std::mutex> lock(impl_->sessions_mutex_);
197 impl_->on_bytes_ = nullptr;
198 impl_->on_state_ = nullptr;
199 impl_->on_bp_ = nullptr;
200 impl_->on_multi_connect_ = nullptr;
201 impl_->on_multi_data_ = nullptr;
202 impl_->on_multi_disconnect_ = nullptr;
203 }
204
205 // 3. Stop all active sessions and clear the map
206 std::vector<std::shared_ptr<UdsServerSession>> sessions_to_stop;
207 {
208 std::lock_guard<std::mutex> lock(impl_->sessions_mutex_);
209 for (auto& pair : impl_->sessions_) {
210 sessions_to_stop.push_back(pair.second);
211 }
212 impl_->sessions_.clear();
213 }
214
215 for (auto& session : sessions_to_stop) {
216 session->stop();
217 }
218
219 // 4. Cleanup UDS socket file
220 std::remove(impl_->cfg_.socket_path.c_str());
221
222 // 5. Release work guard and stop IO context if owned
223 if (impl_->owns_ioc_) {
224 impl_->work_guard_.reset();
225 if (impl_->ioc_) {
226 impl_->ioc_->stop();
227 }
228 }
229
230 if (impl_->owns_ioc_ && impl_->ioc_thread_.joinable()) {
231 if (std::this_thread::get_id() != impl_->ioc_thread_.get_id()) {
232 impl_->ioc_thread_.join();
233 } else {
234 impl_->ioc_thread_.detach();
235 }
236 }
237
238 impl_->state_.set_state(base::LinkState::Idle);
239 impl_->notify_state();
240}

295bool UdsServer::send_to_client(ClientId client_id, memory::ConstByteSpan data) {
296 std::shared_ptr<UdsServerSession> session;
297 {
298 std::lock_guard<std::mutex> lock(impl_->sessions_mutex_);
299 auto it = impl_->sessions_.find(client_id);
300 if (it != impl_->sessions_.end()) session = it->second;
301 }
302 if (session) {
303 session->async_write_copy(data);
304 return true;
305 }
306 return false;
307}

338void UdsServer::Impl::do_accept(std::shared_ptr<UdsServer> self) {
339 acceptor_->async_accept([self](const boost::system::error_code& ec, uds::socket socket) {
340 if (!self || self->impl_->stopping_) return;
341
342 if (!ec) {
343 ClientId client_id;
344 auto session = std::make_shared<UdsServerSession>(*self->impl_->ioc_, std::move(socket),
345 self->impl_->cfg_.backpressure_threshold,
346 self->impl_->cfg_.idle_timeout_ms);
347
348 {
349 std::lock_guard<std::mutex> lock(self->impl_->sessions_mutex_);
350 client_id = self->impl_->next_client_id_++;
351 self->impl_->sessions_[client_id] = session;
352 }
353
354 std::weak_ptr<UdsServer> weak_self = self;
355 session->on_bytes([weak_self, client_id](memory::ConstByteSpan data) {
356 auto s = weak_self.lock();
357 if (!s) return;
358 MultiClientDataHandler data_handler;
359 OnBytes bytes_handler;
360 {
361 std::lock_guard<std::mutex> lock(s->impl_->sessions_mutex_);
362 data_handler = s->impl_->on_multi_data_;
363 bytes_handler = s->impl_->on_bytes_;
364 }
365 if (data_handler) data_handler(client_id, data);
366 if (bytes_handler) bytes_handler(data);
367 });
368
369 session->on_close([weak_self, client_id]() {
370 auto s = weak_self.lock();
371 if (!s || s->impl_->stopping_) return;
372
373 MultiClientDisconnectHandler disconnect_handler;
374 {
375 std::lock_guard<std::mutex> lock(s->impl_->sessions_mutex_);
376 if (s->impl_->stopping_) return; // Double check inside lock
377 s->impl_->sessions_.erase(client_id);
378 disconnect_handler = s->impl_->on_multi_disconnect_;
379 }
380 if (disconnect_handler) disconnect_handler(client_id);
381 });
382
383 session->start();
384
385 MultiClientConnectHandler connect_handler;
386 {
387 std::lock_guard<std::mutex> lock(self->impl_->sessions_mutex_);
388 connect_handler = self->impl_->on_multi_connect_;
389 }
390 if (connect_handler) connect_handler(client_id, "UDS Client");
391
392 // Continue accepting
393 auto* impl = self->impl_.get();
394 impl->do_accept(self);
395 } else {
396 // Log only real errors, not operation_aborted
397 if (ec != boost::asio::error::operation_aborted) {
398 UNILINK_LOG_ERROR("uds_server", "accept", "Accept failed: " + ec.message());
399 }
400 }
401 });
402}
