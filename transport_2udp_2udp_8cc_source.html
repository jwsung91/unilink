111 void init() {
112 cfg_.validate_and_clamp();
113 bp_high_ = cfg_.backpressure_threshold;
114 bp_low_ = bp_high_ > 1 ? bp_high_ / 2 : bp_high_;
115 if (bp_low_ == 0) bp_low_ = 1;
116 bp_limit_ = std::min(std::max(bp_high_ * 4, base::constants::DEFAULT_BACKPRESSURE_THRESHOLD),
117 base::constants::MAX_BUFFER_SIZE);
118 if (bp_limit_ < bp_high_) {
119 bp_limit_ = bp_high_;
120 }
121 set_remote_from_config();
122 }

124 ~Impl() {
125 try {
126 stop_requested_.store(true);
127 stopping_.store(true);
128 if (owns_ioc_ && work_guard_) {
129 work_guard_.reset();

131 if (ioc_thread_.joinable()) {
132 if (std::this_thread::get_id() != ioc_thread_.get_id()) {
133 ioc_thread_.join();
134 } else {
135 ioc_thread_.detach();
136 }
137 }
138 perform_stop_cleanup();
139 } catch (...) {
140 }

143 void open_socket(std::shared_ptr<UdpChannel> self) {
144 if (stopping_.load() || stop_requested_.load()) return;
145
146 boost::system::error_code ec;
147 auto address = net::ip::make_address(cfg_.local_address, ec);
148 if (ec) {
149 UNILINK_LOG_ERROR("udp", "bind", "Invalid local address: " + cfg_.local_address);
150 transition_to(LinkState::Error, ec);
151 return;
152 }
153
154 local_endpoint_ = udp::endpoint(address, cfg_.local_port);
155 socket_.open(local_endpoint_.protocol(), ec);
156 if (ec) {
157 UNILINK_LOG_ERROR("udp", "open", "Socket open failed: " + ec.message());
158 transition_to(LinkState::Error, ec);
159 return;
160 }
161
162 if (cfg_.reuse_address) {
163 socket_.set_option(net::socket_base::reuse_address(true), ec);
164 if (ec) {
165 UNILINK_LOG_ERROR("udp", "open", "Failed to set reuse_address: " + ec.message());
166 transition_to(LinkState::Error, ec);
167 return;
168 }
169 }
170
171 if (cfg_.enable_broadcast) {
172 socket_.set_option(net::socket_base::broadcast(true), ec);
173 if (ec) {
174 UNILINK_LOG_ERROR("udp", "open", "Failed to set broadcast: " + ec.message());
175 transition_to(LinkState::Error, ec);
176 return;
177 }
178 }
179
180 socket_.bind(local_endpoint_, ec);
181 if (ec) {
182 UNILINK_LOG_ERROR("udp", "bind", "Bind failed: " + ec.message());
183 transition_to(LinkState::Error, ec);
184 return;
185 }
186
187 opened_.store(true);
188 if (remote_endpoint_) {
189 connected_.store(true);
190 transition_to(LinkState::Connected);
191 } else {
192 transition_to(LinkState::Listening);
193 }
194 start_receive(self);
195 }

197 void start_receive(std::shared_ptr<UdpChannel> self) {
198 if (stopping_.load() || stop_requested_.load() || state_.is_state(LinkState::Closed) ||
199 state_.is_state(LinkState::Error) || !socket_.is_open()) {
200 return;
201 }
202
203 socket_.async_receive_from(net::buffer(rx_), recv_endpoint_,
204 [self](const boost::system::error_code& ec, std::size_t bytes) {
205 auto impl = self->get_impl();
206 impl->handle_receive(self, ec, bytes);
207 });
208 }
209
210 void handle_receive(std::shared_ptr<UdpChannel> self, const boost::system::error_code& ec, std::size_t bytes) {
211 if (ec == boost::asio::error::operation_aborted) {
212 return;
213 }
214
215 if (stopping_.load() || stop_requested_.load() || state_.is_state(LinkState::Closed) ||
216 state_.is_state(LinkState::Error)) {
217 return;
218 }
219
220 if (ec == boost::asio::error::message_size || bytes >= rx_.size()) {
221 UNILINK_LOG_ERROR("udp", "receive", "Datagram truncated (buffer too small)");
222 transition_to(LinkState::Error, ec);
223 return;
224 }
225
226 if (ec) {
227 UNILINK_LOG_ERROR("udp", "receive", "Receive failed: " + ec.message());
228 transition_to(LinkState::Error, ec);
229 return;
230 }
231
232 if (!remote_endpoint_) {
233 remote_endpoint_ = recv_endpoint_;
234 connected_.store(true);
235 transition_to(LinkState::Connected);
236 }
237
238 if (bytes > 0) {
239 if (on_bytes_) {
240 try {
241 on_bytes_(memory::ConstByteSpan(rx_.data(), bytes));
242 } catch (const std::exception& e) {
243 UNILINK_LOG_ERROR("udp", "on_bytes", "Exception in bytes callback: " + std::string(e.what()));
244 if (cfg_.stop_on_callback_exception) {
245 transition_to(LinkState::Error);
246 return;
247 }
248 } catch (...) {
249 UNILINK_LOG_ERROR("udp", "on_bytes", "Unknown exception in bytes callback");
250 if (cfg_.stop_on_callback_exception) {
251 transition_to(LinkState::Error);
252 return;
253 }
254 }
255 }
256
257 if (on_bytes_from_) {
258 try {
259 on_bytes_from_(memory::ConstByteSpan(rx_.data(), bytes), recv_endpoint_);
260 } catch (const std::exception& e) {
261 UNILINK_LOG_ERROR("udp", "on_bytes_from", "Exception in bytes callback: " + std::string(e.what()));
262 if (cfg_.stop_on_callback_exception) {
263 transition_to(LinkState::Error);
264 return;
265 }
266 } catch (...) {
267 UNILINK_LOG_ERROR("udp", "on_bytes_from", "Unknown exception in bytes callback");
268 if (cfg_.stop_on_callback_exception) {
269 transition_to(LinkState::Error);
270 return;
271 }
272 }
273 }
274 }
275
276 start_receive(self);
277 }

279 void do_write(std::shared_ptr<UdpChannel> self) {
280 if (writing_ || tx_.empty()) return;
281 if (stop_requested_.load() || stopping_.load() || state_.is_state(LinkState::Closed) ||
282 state_.is_state(LinkState::Error)) {
283 tx_.clear();
284 queue_bytes_ = 0;
285 writing_ = false;
286 backpressure_active_ = false;
287 report_backpressure(queue_bytes_);
288 return;
289 }
290
291 auto current = std::move(tx_.front());
292 tx_.pop_front();
293
294 const auto& dest_endpoint = current.destination ? current.destination : remote_endpoint_;
295
296 if (!dest_endpoint) {
297 UNILINK_LOG_WARNING("udp", "write", "Remote endpoint not set; dropping write request");
298 writing_ = false;
299 do_write(self); // Process next in queue
300 return;
301 }
302
303 writing_ = true;
304
305 auto bytes_queued = std::visit(
306 [](auto&& buf) -> size_t {
307 using Buffer = std::decay_t<decltype(buf)>;
308 if constexpr (std::is_same_v<Buffer, std::shared_ptr<const std::vector<uint8_t>>>) {
309 return buf ? buf->size() : 0;
310 } else {
311 return buf.size();
312 }
313 },
314 current.buffer);
315
316 auto on_write = [self, bytes_queued](const boost::system::error_code& ec, std::size_t) {
317 auto impl = self->get_impl();
318 impl->queue_bytes_ = (impl->queue_bytes_ > bytes_queued) ? (impl->queue_bytes_ - bytes_queued) : 0;
319 impl->report_backpressure(impl->queue_bytes_);
320
321 if (ec == boost::asio::error::operation_aborted) {
322 impl->writing_ = false;
323 return;
324 }
325
326 if (impl->stop_requested_.load() || impl->stopping_.load() || impl->state_.is_state(LinkState::Closed) ||
327 impl->state_.is_state(LinkState::Error)) {
328 impl->writing_ = false;
329 impl->tx_.clear();
330 impl->queue_bytes_ = 0;
331 impl->report_backpressure(impl->queue_bytes_);
332 return;
333 }
334
335 if (ec) {
336 UNILINK_LOG_ERROR("udp", "write", "Send failed: " + ec.message());
337 impl->transition_to(LinkState::Error, ec);
338 impl->writing_ = false;
339 return;
340 }
341
342 impl->writing_ = false;
343 impl->do_write(self);
344 };
345
346 std::visit(
347 [&](auto&& buf) {
348 using T = std::decay_t<decltype(buf)>;
349
350 auto* data_ptr = [&]() {
351 if constexpr (std::is_same_v<T, std::shared_ptr<const std::vector<uint8_t>>>) {
352 return buf->data();
353 } else {
354 return buf.data();
355 }
356 }();
357
358 auto size = [&]() {
359 if constexpr (std::is_same_v<T, std::shared_ptr<const std::vector<uint8_t>>>) {
360 return buf->size();
361 } else {
362 return buf.size();
363 }
364 }();
365
366 socket_.async_send_to(
367 net::buffer(data_ptr, size), *dest_endpoint,
368 [buf_captured = std::move(buf), on_write = std::move(on_write)](
369 const boost::system::error_code& ec, std::size_t bytes) mutable { on_write(ec, bytes); });
370 },
371 std::move(current.buffer));
372 }

380 void notify_state() {
381 if (!on_state_) return;
382 try {
383 on_state_(state_.state());
384 } catch (const std::exception& e) {
385 UNILINK_LOG_ERROR("udp", "on_state", "Exception in state callback: " + std::string(e.what()));
386 } catch (...) {
387 UNILINK_LOG_ERROR("udp", "on_state", "Unknown exception in state callback");
388 }
389 }
390
391 void report_backpressure(size_t queued_bytes) {
392 if (stop_requested_.load() || !on_bp_) return;
393
394 if (!backpressure_active_ && queued_bytes >= bp_high_) {
395 backpressure_active_ = true;
396 try {
397 on_bp_(queued_bytes);
398 } catch (const std::exception& e) {
399 UNILINK_LOG_ERROR("udp", "on_backpressure", "Exception in backpressure callback: " + std::string(e.what()));
400 } catch (...) {
401 UNILINK_LOG_ERROR("udp", "on_backpressure", "Unknown exception in backpressure callback");
402 }
403 } else if (backpressure_active_ && queued_bytes <= bp_low_) {
404 backpressure_active_ = false;
405 try {
406 on_bp_(queued_bytes);
407 } catch (const std::exception& e) {
408 UNILINK_LOG_ERROR("udp", "on_backpressure", "Exception in backpressure callback: " + std::string(e.what()));
409 } catch (...) {
410 UNILINK_LOG_ERROR("udp", "on_backpressure", "Unknown exception in backpressure callback");
411 }
412 }
413 }
414
415 bool enqueue_buffer(BufferVariant&& buffer, size_t size, std::optional<udp::endpoint> dest = std::nullopt) {
416 if (stopping_.load() || stop_requested_.load() || state_.is_state(LinkState::Closed) ||
417 state_.is_state(LinkState::Error)) {
418 return false;
419 }
420
421 if (queue_bytes_ + size > bp_limit_) {
422 UNILINK_LOG_ERROR("udp", "write", "Queue limit exceeded (" + std::to_string(queue_bytes_ + size) + " bytes)");
423 report_backpressure(queue_bytes_ + size);
424 return false;
425 }
426 queue_bytes_ += size;
427 tx_.push_back({std::move(buffer), dest});
428 report_backpressure(queue_bytes_);
429 return true;
430 }

432 void set_remote_from_config() {
433 if (!cfg_.remote_address || !cfg_.remote_port) return;
434 boost::system::error_code ec;
435 auto addr = net::ip::make_address(*cfg_.remote_address, ec);
436 if (ec) {
437 throw std::runtime_error("Invalid remote address: " + *cfg_.remote_address);
438 }
439 remote_endpoint_ = udp::endpoint(addr, *cfg_.remote_port);
440 }
441
442 void transition_to(LinkState target, const boost::system::error_code& ec = {}) {
443 if (ec == net::error::operation_aborted) {
444 return;
445 }
446
447 const auto current = state_.state();
448 if ((current == LinkState::Closed || current == LinkState::Error) &&
449 (target == LinkState::Closed || target == LinkState::Error)) {
450 return;
451 }
452
453 if (target == LinkState::Closed || target == LinkState::Error) {
454 if (terminal_state_notified_.exchange(true)) {
455 return;
456 }
457 } else if (current == target) {
458 return;
459 }
460
461 state_.set_state(target);
462 notify_state();
463 }
464
465 void perform_stop_cleanup() {
466 try {
467 close_socket();
468 tx_.clear();
469 queue_bytes_ = 0;
470 writing_ = false;
471 const bool had_backpressure = backpressure_active_;
472 backpressure_active_ = false;
473 if (had_backpressure && on_bp_) {
474 try {
475 on_bp_(queue_bytes_);
476 } catch (...) {
477 }
478 }
479 connected_.store(false);
480 opened_.store(false);
481 if (owns_ioc_ && work_guard_) {
482 work_guard_->reset();
483 }
484 transition_to(LinkState::Closed);
485 on_bytes_ = nullptr;
486 on_state_ = nullptr;
487 on_bp_ = nullptr;
488 } catch (...) {
489 }
490 }
491
492 void join_ioc_thread(bool allow_detach) {
493 if (!owns_ioc_ || !ioc_thread_.joinable()) {
494 return;
495 }
496
497 if (std::this_thread::get_id() == ioc_thread_.get_id()) {
498 if (allow_detach) {
499 ioc_thread_.detach();
500 }
501 return;
502 }
503
504 try {
505 ioc_thread_.join();
506 } catch (...) {
507 }
508 }
509};

536void UdpChannel::start() {
537 auto impl = get_impl();
538 if (impl->started_) return;
539 if (!impl->cfg_.is_valid()) {
540 throw std::runtime_error("Invalid UDP configuration");
541 }
542
543 if (impl->owns_ioc_ && impl->owned_ioc_ && impl->owned_ioc_->stopped()) {
544 impl->owned_ioc_->restart();
545 }
546
547 if (impl->ioc_thread_.joinable()) {
548 impl->join_ioc_thread(false);
549 }
550
551 if (impl->owns_ioc_) {
552 impl->work_guard_ =
553 std::make_unique<net::executor_work_guard<net::io_context::executor_type>>(impl->ioc_->get_executor());
554 }
555
556 auto self = shared_from_this();
557 net::dispatch(impl->strand_, [self]() {
558 auto impl = self->get_impl();
559 impl->stop_requested_.store(false);
560 impl->stopping_.store(false);
561 impl->terminal_state_notified_.store(false);
562 impl->connected_.store(false);
563 impl->opened_.store(false);
564 impl->writing_ = false;
565 impl->queue_bytes_ = 0;
566 impl->backpressure_active_ = false;
567 impl->state_.set_state(LinkState::Idle);
568
569 impl->transition_to(LinkState::Connecting);
570 impl->open_socket(self);
571 });
572
573 if (impl->owns_ioc_) {
574 impl->ioc_thread_ = std::thread([impl]() {
575 try {
576 impl->ioc_->run();
577 } catch (...) {
578 }
579 });
580 }
581
582 impl->started_ = true;
583}

585void UdpChannel::stop() {
586 auto impl = get_impl();
587 if (impl->stop_requested_.exchange(true)) return;
588
589 if (!impl->started_) {
590 impl->transition_to(LinkState::Closed);
591 impl->on_bytes_ = nullptr;
592 impl->on_state_ = nullptr;
593 impl->on_bp_ = nullptr;
594 return;
595 }
596
597 impl->stopping_.store(true);
598 auto self = shared_from_this();
599 net::post(impl->strand_, [self]() { self->get_impl()->perform_stop_cleanup(); });
600
601 impl->join_ioc_thread(false);
602
603 if (impl->owns_ioc_ && impl->owned_ioc_) {
604 impl->owned_ioc_->restart();
605 }
606
607 impl->started_ = false;
608}

612void UdpChannel::async_write_copy(memory::ConstByteSpan data) {
613 auto impl = get_impl();
614 if (data.empty()) return;
615 if (impl->stop_requested_.load()) return;
616 if (impl->stopping_.load() || impl->state_.is_state(LinkState::Closed) || impl->state_.is_state(LinkState::Error))
617 return;
618
619 size_t size = data.size();
620 if (size > base::constants::MAX_BUFFER_SIZE) {
621 UNILINK_LOG_ERROR("udp", "write", "Write size exceeds maximum allowed");
622 return;
623 }
624
625 if (impl->cfg_.enable_memory_pool && size <= 65536) {
626 memory::PooledBuffer pooled(size);
627 if (pooled.valid()) {
628 base::safe_memory::safe_memcpy(pooled.data(), data.data(), size);
629 net::post(impl->strand_, [self = shared_from_this(), buf = std::move(pooled), size]() mutable {
630 auto impl = self->get_impl();
631 if (!impl->enqueue_buffer(std::move(buf), size)) return;
632 impl->do_write(self);
633 });
634 return;
635 }
636 }
637
638 std::vector<uint8_t> copy(data.begin(), data.end());
639 net::post(impl->strand_, [self = shared_from_this(), buf = std::move(copy), size]() mutable {
640 auto impl = self->get_impl();
641 if (!impl->enqueue_buffer(std::move(buf), size)) return;
642 impl->do_write(self);
643 });
644}

698void UdpChannel::async_write_to(memory::ConstByteSpan data, const boost::asio::ip::udp::endpoint& destination) {
699 auto impl = get_impl();
700 if (data.empty()) return;
701 if (impl->stop_requested_.load()) return;
702 if (impl->stopping_.load() || impl->state_.is_state(LinkState::Closed) || impl->state_.is_state(LinkState::Error))
703 return;
704
705 size_t size = data.size();
706 if (impl->cfg_.enable_memory_pool && size <= 65536) {
