82 void UdpChannel::start() {
83  if (started_) return;
84  if (!cfg_.is_valid()) {
85  throw std::runtime_error("Invalid UDP configuration");
86  }
87 
88  if (owns_ioc_ && owned_ioc_ && owned_ioc_->stopped()) {
89  owned_ioc_->restart();
90  }
91 
92  if (ioc_thread_.joinable()) {
93  join_ioc_thread(false);
94  }
95 
96  if (owns_ioc_) {
97  work_guard_ = std::make_unique<net::executor_work_guard<net::io_context::executor_type>>(ioc_->get_executor());
98  }
99 
100  auto self = shared_from_this();
101  net::dispatch(strand_, [self]() {
102  self->reset_start_state();
103  self->transition_to(LinkState::Connecting);
104  self->open_socket();
105  });
106 
107  if (owns_ioc_) {
108  ioc_thread_ = std::thread([this]() {
109  try {
110  ioc_->run();
111  } catch (const std::exception& e) {
112  UNILINK_LOG_ERROR("udp", "run", "Unhandled exception: " + std::string(e.what()));
113  } catch (...) {
114  UNILINK_LOG_ERROR("udp", "run", "Unknown unhandled exception");
115  }
116  });
117  }
118 
119  started_ = true;
120 }

122 void UdpChannel::stop() {
123  if (stop_requested_.exchange(true)) return;
124 
125  if (!started_) {
126  transition_to(LinkState::Closed);
127  clear_callbacks();
128  return;
129  }
130 
131  stopping_.store(true);
132  auto weak = weak_from_this();
133  if (!ioc_) {
134  perform_stop_cleanup();
135  } else if (auto self = weak.lock()) {
136  net::post(strand_, [self]() { self->perform_stop_cleanup(); });
137  } else {
138  perform_stop_cleanup();
139  }
140 
141  join_ioc_thread(false);
142 
143  if (owns_ioc_ && owned_ioc_) {
144  owned_ioc_->restart();
145  }
146 
147  started_ = false;
148 }

152 void UdpChannel::async_write_copy(const uint8_t* data, size_t size) {
153  if (!data || size == 0) return;
154  if (stop_requested_.load()) return;
155  if (stopping_.load() || state_.is_state(LinkState::Closed) || state_.is_state(LinkState::Error)) return;
156  if (!remote_endpoint_) {
157  UNILINK_LOG_WARNING("udp", "write", "Remote endpoint not set; dropping write request");
158  return;
159  }
160 
161  if (size > common::constants::MAX_BUFFER_SIZE) {
162  UNILINK_LOG_ERROR("udp", "write", "Write size exceeds maximum allowed");
163  return;
164  }
165 
166  if (size > bp_limit_) {
167  UNILINK_LOG_ERROR("udp", "write", "Queue limit exceeded by single write");
168  transition_to(LinkState::Error);
169  return;
170  }
171 
172  if (cfg_.enable_memory_pool && size <= 65536) {
173  memory::PooledBuffer pooled(size);
174  if (pooled.valid()) {
175  common::safe_memory::safe_memcpy(pooled.data(), data, size);
176  net::post(strand_, [self = weak_from_this(), buf = std::move(pooled), size]() mutable {
177  if (auto s = self.lock()) {
178  if (!s->enqueue_buffer(std::move(buf), size)) return;
179  s->do_write();
180  }
181  });
182  return;
183  }
184  }
185 
186  std::vector<uint8_t> copy(data, data + size);
187  net::post(strand_, [self = weak_from_this(), buf = std::move(copy), size]() mutable {
188  if (auto s = self.lock()) {
189  if (!s->enqueue_buffer(std::move(buf), size)) return;
190  s->do_write();
191  }
192  });
193 }

249 void UdpChannel::open_socket() {
250  if (stopping_.load() || stop_requested_.load()) return;
251 
252  boost::system::error_code ec;
253  auto address = net::ip::make_address(cfg_.local_address, ec);
254  if (ec) {
255  UNILINK_LOG_ERROR("udp", "bind", "Invalid local address: " + cfg_.local_address);
256  transition_to(LinkState::Error, ec);
257  return;
258  }
259 
260  local_endpoint_ = udp::endpoint(address, cfg_.local_port);
261  socket_.open(local_endpoint_.protocol(), ec);
262  if (ec) {
263  UNILINK_LOG_ERROR("udp", "open", "Socket open failed: " + ec.message());
264  transition_to(LinkState::Error, ec);
265  return;
266  }
267 
268  socket_.bind(local_endpoint_, ec);
269  if (ec) {
270  UNILINK_LOG_ERROR("udp", "bind", "Bind failed: " + ec.message());
271  transition_to(LinkState::Error, ec);
272  return;
273  }
274 
275  opened_.store(true);
276  if (remote_endpoint_) {
277  connected_.store(true);
278  transition_to(LinkState::Connected);
279  } else {
280  transition_to(LinkState::Listening);
281  }
282  start_receive();
283 }

297 void UdpChannel::handle_receive(const boost::system::error_code& ec, std::size_t bytes) {
298  if (ec == boost::asio::error::operation_aborted) {
299  return;
300  }
301 
302  if (stopping_.load() || stop_requested_.load() || state_.is_state(LinkState::Closed) ||
303  state_.is_state(LinkState::Error)) {
304  return;
305  }
306 
307  // Fail-fast on truncation: treat both message_size and full-buffer reads as truncated datagrams.
308  // We set Error once and exit without re-arming the receive loop, so user callbacks are not spammed.
309  if (ec == boost::asio::error::message_size || bytes >= rx_.size()) {
310  UNILINK_LOG_ERROR("udp", "receive", "Datagram truncated (buffer too small)");
311  transition_to(LinkState::Error, ec);
312  return;
313  }
314 
315  if (ec) {
316  UNILINK_LOG_ERROR("udp", "receive", "Receive failed: " + ec.message());
317  transition_to(LinkState::Error, ec);
318  return;
319  }
320 
321  if (!remote_endpoint_) {
322  remote_endpoint_ = recv_endpoint_;
323  connected_.store(true);
324  transition_to(LinkState::Connected);
325  }
326 
327  if (bytes > 0 && on_bytes_) {
328  try {
329  on_bytes_(rx_.data(), bytes);
330  } catch (const std::exception& e) {
331  UNILINK_LOG_ERROR("udp", "on_bytes", "Exception in bytes callback: " + std::string(e.what()));
332  if (cfg_.stop_on_callback_exception) {
333  transition_to(LinkState::Error);
334  return;
335  }
336  } catch (...) {
337  UNILINK_LOG_ERROR("udp", "on_bytes", "Unknown exception in bytes callback");
338  if (cfg_.stop_on_callback_exception) {
339  transition_to(LinkState::Error);
340  return;
341  }
342  }
343  }
344 
345  start_receive();
346 }

348 void UdpChannel::do_write() {
349  if (writing_ || tx_.empty()) return;
350  if (stop_requested_.load() || stopping_.load() || state_.is_state(LinkState::Closed) ||
351  state_.is_state(LinkState::Error)) {
352  tx_.clear();
353  queue_bytes_ = 0;
354  writing_ = false;
355  backpressure_active_ = false;
356  report_backpressure(queue_bytes_);
357  return;
358  }
359  if (!remote_endpoint_) {
360  writing_ = false;
361  return;
362  }
363 
364  writing_ = true;
365 
366  auto current = std::move(tx_.front());
367  tx_.pop_front();
368  auto bytes_queued = std::visit(
369  [](auto&& buf) -> size_t {
370  using Buffer = std::decay_t<decltype(buf)>;
371  if constexpr (std::is_same_v<Buffer, std::shared_ptr<const std::vector<uint8_t>>>) {
372  return buf ? buf->size() : 0;
373  } else {
374  return buf.size();
375  }
376  },
377  current);
378 
379  auto on_write = [self = shared_from_this(), bytes_queued](const boost::system::error_code& ec, std::size_t) {
380  self->queue_bytes_ = (self->queue_bytes_ > bytes_queued) ? (self->queue_bytes_ - bytes_queued) : 0;
381  self->report_backpressure(self->queue_bytes_);
382 
383  if (ec == boost::asio::error::operation_aborted) {
384  self->writing_ = false;
385  return;
386  }
387 
388  if (self->stop_requested_.load() || self->stopping_.load() || self->state_.is_state(LinkState::Closed) ||
389  self->state_.is_state(LinkState::Error)) {
390  self->writing_ = false;
391  self->tx_.clear();
392  self->queue_bytes_ = 0;
393  self->report_backpressure(self->queue_bytes_);
394  return;
395  }
396 
397  if (ec) {
398  UNILINK_LOG_ERROR("udp", "write", "Send failed: " + ec.message());
399  self->transition_to(LinkState::Error, ec);
400  self->writing_ = false;
401  return;
402  }
403 
404  self->writing_ = false;
405  self->do_write();
406  };
407 
408  if (std::holds_alternative<memory::PooledBuffer>(current)) {
409  auto pooled = std::get<memory::PooledBuffer>(std::move(current));
410  // Optimization: Move PooledBuffer directly into lambda to avoid std::make_shared allocation
411  auto data = pooled.data();
412  auto size = pooled.size();
413  socket_.async_send_to(net::buffer(data, size), *remote_endpoint_,
414  [buf = std::move(pooled), on_write = std::move(on_write)](
415  const boost::system::error_code& ec, std::size_t bytes) mutable { on_write(ec, bytes); });
416  } else if (std::holds_alternative<std::shared_ptr<const std::vector<uint8_t>>>(current)) {
417  auto shared_vec = std::get<std::shared_ptr<const std::vector<uint8_t>>>(std::move(current));
418  auto data = shared_vec->data();
419  auto size = shared_vec->size();
420  socket_.async_send_to(net::buffer(data, size), *remote_endpoint_,
421  [shared_vec = std::move(shared_vec), on_write = std::move(on_write)](
422  const boost::system::error_code& ec, std::size_t bytes) mutable { on_write(ec, bytes); });
423  } else {
424  auto vec = std::get<std::vector<uint8_t>>(std::move(current));
425  // Optimization: Move vector directly into lambda to avoid std::make_shared allocation
426  auto data = vec.data();
427  auto size = vec.size();
428  socket_.async_send_to(net::buffer(data, size), *remote_endpoint_,
429  [buf = std::move(vec), on_write = std::move(on_write)](
430  const boost::system::error_code& ec, std::size_t bytes) mutable { on_write(ec, bytes); });
431  }
432 }

451 void UdpChannel::report_backpressure(size_t queued_bytes) {
452  if (stop_requested_.load() || !on_bp_) return;
453 
454  if (!backpressure_active_ && queued_bytes >= bp_high_) {
455  backpressure_active_ = true;
456  try {
457  on_bp_(queued_bytes);
458  } catch (const std::exception& e) {
459  UNILINK_LOG_ERROR("udp", "on_backpressure", "Exception in backpressure callback: " + std::string(e.what()));
460  } catch (...) {
461  UNILINK_LOG_ERROR("udp", "on_backpressure", "Unknown exception in backpressure callback");
462  }
463  } else if (backpressure_active_ && queued_bytes <= bp_low_) {
464  backpressure_active_ = false;
465  try {
466  on_bp_(queued_bytes);
467  } catch (const std::exception& e) {
468  UNILINK_LOG_ERROR("udp", "on_backpressure", "Exception in backpressure callback: " + std::string(e.what()));
469  } catch (...) {
470  UNILINK_LOG_ERROR("udp", "on_backpressure", "Unknown exception in backpressure callback");
471  }
472  }

488 bool UdpChannel::enqueue_buffer(
489  std::variant<memory::PooledBuffer, std::vector<uint8_t>, std::shared_ptr<const std::vector<uint8_t>>>&& buffer,
490  size_t size) {
491  if (stopping_.load() || stop_requested_.load() || state_.is_state(LinkState::Closed) ||
492  state_.is_state(LinkState::Error)) {
493  return false;
494  }
495 
496  if (queue_bytes_ + size > bp_limit_) {
497  UNILINK_LOG_ERROR("udp", "write", "Queue limit exceeded");
498  transition_to(LinkState::Error);
499  return false;
500  }
501  queue_bytes_ += size;
502  tx_.push_back(std::move(buffer));
503  report_backpressure(queue_bytes_);
504  return true;
505 }

517 void UdpChannel::transition_to(LinkState target, const boost::system::error_code& ec) {
518  if (ec == net::error::operation_aborted) {
519  return;
520  }
521 
522  const auto current = state_.get_state();
523  if ((current == LinkState::Closed || current == LinkState::Error) &&
524  (target == LinkState::Closed || target == LinkState::Error)) {
525  return;
526  }
527 
528  if (target == LinkState::Closed || target == LinkState::Error) {
529  if (terminal_state_notified_.exchange(true)) {
530  return;
531  }
532  } else if (current == target) {
533  return;
534  }
535 
536  state_.set_state(target);
537  notify_state();
538 }

546 void UdpChannel::perform_stop_cleanup() {
547  try {
548  close_socket();
549  tx_.clear();
550  queue_bytes_ = 0;
551  writing_ = false;
552  const bool had_backpressure = backpressure_active_;
553  backpressure_active_ = false;
554  if (had_backpressure && on_bp_) {
555  try {
556  on_bp_(queue_bytes_);
557  } catch (const std::exception& e) {
558  UNILINK_LOG_ERROR("udp", "on_backpressure", "Exception in backpressure callback: " + std::string(e.what()));
559  } catch (...) {
560  UNILINK_LOG_ERROR("udp", "on_backpressure", "Unknown exception in backpressure callback");
561  }
562  }
563  connected_.store(false);
564  opened_.store(false);
565  if (owns_ioc_ && ioc_) {
566  ioc_->stop();
567  }
568  if (owns_ioc_ && work_guard_) {
569  work_guard_->reset();
570  }
571  transition_to(LinkState::Closed);
572  clear_callbacks();
573  } catch (const std::exception& e) {
574  UNILINK_LOG_ERROR("udp", "stop_cleanup", "Cleanup error: " + std::string(e.what()));
575  } catch (...) {
576  UNILINK_LOG_ERROR("udp", "stop_cleanup", "Unknown error in stop cleanup");
577  }
578 }
