46 std::unique_ptr<wrapper::Serial> SerialBuilder::build() {
47  try {
48  // Final validation before building
49  util::InputValidator::validate_device_path(device_);
50  util::InputValidator::validate_baud_rate(baud_rate_);
51  util::InputValidator::validate_retry_interval(retry_interval_ms_);
52 
53  std::shared_ptr<boost::asio::io_context> external_ioc;
54  if (use_independent_context_) {
55  // Use independent IoContext (for test isolation)
56  auto independent_context = concurrency::IoContextManager::instance().create_independent_context();
57  external_ioc = std::shared_ptr<boost::asio::io_context>(std::move(independent_context));
58  } else {
59  // Default behavior uses shared IoContextManager - ensure it is running
60  AutoInitializer::ensure_io_context_running();
61  }
62 
63  auto serial = external_ioc ? std::make_unique<wrapper::Serial>(device_, baud_rate_, external_ioc)
64  : std::make_unique<wrapper::Serial>(device_, baud_rate_);
65 
66  // Apply configuration with exception safety
67  try {
68  if (external_ioc) {
69  serial->set_manage_external_context(use_independent_context_);
70  }
71 
72  // Apply framing if configured
73  if (framer_) {
74  std::shared_ptr<framer::IFramer> shared_framer = std::move(framer_);
75  if (on_message_) {
76  shared_framer->set_on_message(on_message_);
77  }
78  serial->on_bytes([shared_framer](memory::ConstByteSpan data) { shared_framer->push_bytes(data); });
79  }
80 
81  // Set callbacks with exception safety
82  if (on_data_) {
83  serial->on_data(on_data_);
84  }
85 
86  if (on_connect_) {
87  serial->on_connect(on_connect_);
88  }
89 
90  if (on_disconnect_) {
91  serial->on_disconnect(on_disconnect_);
92  }
93 
94  if (on_error_) {
95  serial->on_error(on_error_);
96  }
97 
98  // Set retry interval
99  serial->set_retry_interval(std::chrono::milliseconds(retry_interval_ms_));
100 
101  serial->auto_manage(auto_manage_);
102 
103  } catch (const std::exception& e) {
104  // If configuration fails, ensure serial is properly cleaned up
105  serial.reset();
106  throw diagnostics::BuilderException("Failed to configure Serial: " + std::string(e.what()), "SerialBuilder",
107  "build");
108  }
109 
110  return serial;
111 
112  } catch (const diagnostics::ValidationException& e) {
113  throw diagnostics::BuilderException("Validation failed during Serial build: " + e.get_full_message(),
114  "SerialBuilder", "build");
115  } catch (const std::bad_alloc& e) {
116  throw diagnostics::BuilderException("Memory allocation failed during Serial build: " + std::string(e.what()),
117  "SerialBuilder", "build");
118  } catch (const std::exception& e) {
119  throw diagnostics::BuilderException("Unexpected error during Serial build: " + std::string(e.what()),
120  "SerialBuilder", "build");
121  }
122 }
