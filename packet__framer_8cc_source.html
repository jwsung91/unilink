34 void PacketFramer::push_bytes(memory::ConstByteSpan data) {
35  if (data.empty()) return;
36 
37  buffer_.insert(buffer_.end(), data.begin(), data.end());
38 
39  while (true) {
40  if (state_ == State::Sync) {
41  if (start_pattern_.empty()) {
42  state_ = State::Collect;
43  continue;
44  }
45 
46  auto it = std::search(buffer_.begin(), buffer_.end(), start_pattern_.begin(), start_pattern_.end());
47  if (it != buffer_.end()) {
48  // Found start pattern.
49  // Discard everything before start pattern.
50  if (it != buffer_.begin()) {
51  buffer_.erase(buffer_.begin(), it);
52  }
53  state_ = State::Collect;
54  // Start scanning for end pattern after the start pattern we just found
55  scanned_idx_ = start_pattern_.size();
56  // Continue to check for end pattern immediately
57  } else {
58  // Start pattern not found.
59  // Keep partial match at the end.
60  if (start_pattern_.size() > 1) {
61  size_t keep_len = start_pattern_.size() - 1;
62  if (buffer_.size() > keep_len) {
63  buffer_.erase(buffer_.begin(), buffer_.end() - static_cast<std::ptrdiff_t>(keep_len));
64  }
65  } else {
66  buffer_.clear();
67  }
68  break; // Need more data
69  }
70  } else if (state_ == State::Collect) {
71  if (end_pattern_.empty()) {
72  // If end pattern is empty, packet ends immediately after start pattern?
73  // Assume minimal packet is start pattern only
74  size_t packet_len = start_pattern_.size();
75  if (on_message_) {
76  on_message_(memory::ConstByteSpan(buffer_.data(), packet_len));
77  }
78  if (buffer_.empty()) return;
79 
80  buffer_.erase(buffer_.begin(), buffer_.begin() + static_cast<std::ptrdiff_t>(packet_len));
81  state_ = State::Sync;
82  continue;
83  }
84 
85  // Search for end pattern *after* start pattern
86  // Optimization: use scanned_idx_ to avoid re-scanning
87  size_t search_offset = std::max(start_pattern_.size(), scanned_idx_);
88 
89  // Back up slightly to catch split end pattern if we are resuming search
90  if (search_offset > start_pattern_.size()) {
91  size_t overlap = (end_pattern_.size() > 1) ? (end_pattern_.size() - 1) : 0;
92  if (search_offset >= overlap) {
93  search_offset -= overlap;
94  } else {
95  search_offset = 0;
96  }
97  }
98 
99  // Safety clamp to ensure we don't search inside start pattern
100  if (search_offset < start_pattern_.size()) {
101  search_offset = start_pattern_.size();
102  }
103 
104  if (buffer_.size() < search_offset) {
105  // Should not happen if Sync worked correctly
106  break;
107  }
108 
109  auto search_start = buffer_.begin() + static_cast<std::ptrdiff_t>(search_offset);
110  auto it = std::search(search_start, buffer_.end(), end_pattern_.begin(), end_pattern_.end());
111 
112  if (it != buffer_.end()) {
113  // Found end pattern.
114  size_t packet_len = static_cast<size_t>(std::distance(buffer_.begin(), it)) + end_pattern_.size();
115 
116  if (packet_len <= max_length_) {
117  if (on_message_) {
118  on_message_(memory::ConstByteSpan(buffer_.data(), packet_len));
119  }
120  if (buffer_.empty()) return;
121 
122  buffer_.erase(buffer_.begin(), buffer_.begin() + static_cast<std::ptrdiff_t>(packet_len));
123  state_ = State::Sync;
124  scanned_idx_ = 0;
125  } else {
126  // Exceeded max length, discard packet
127  buffer_.erase(buffer_.begin(), buffer_.begin() + static_cast<std::ptrdiff_t>(packet_len));
128  state_ = State::Sync;
129  scanned_idx_ = 0;
130  }
131  } else {
132  // End pattern not found.
133  scanned_idx_ = buffer_.size();
134  if (buffer_.size() > max_length_) {
135  // Exceeded limit while collecting. Reset.
136  buffer_.clear();
137  state_ = State::Sync;
138  scanned_idx_ = 0;
139  }
140  break; // Need more data
141  }
142  }
143  }
144 }
