76void TcpServerSession::async_write_copy(memory::ConstByteSpan data) {
77 if (!alive_ || closing_) return; // Don't queue writes if session is not alive
78
79 size_t size = data.size();
80 if (size > base::constants::MAX_BUFFER_SIZE) {
81 UNILINK_LOG_ERROR("tcp_server_session", "write", "Write size exceeds maximum allowed");
82 return;
83 }
84
85 // Use memory pool for better performance (only for reasonable sizes)
86 if (size <= base::constants::LARGE_BUFFER_THRESHOLD) { // Only use pool for buffers <= 64KB
87 memory::PooledBuffer pooled_buffer(size);
88 if (pooled_buffer.valid()) {
89 // Copy data to pooled buffer safely
90 base::safe_memory::safe_memcpy(pooled_buffer.data(), data.data(), size);
91
92 net::post(strand_, [self = shared_from_this(), buf = std::move(pooled_buffer)]() mutable {
93 if (!self->alive_ || self->closing_) return; // Double-check in case session was closed
94 const auto added = buf.size();
95 self->maybe_flush_for_keep_latest(added);
96 if (self->queue_bytes_ + added > self->bp_limit_) {
97 UNILINK_LOG_ERROR("tcp_server_session", "write", "Queue limit exceeded, dropping message");
98 self->report_backpressure(self->queue_bytes_ + added);
99 return;
100 }
101
102 self->queue_bytes_ += added;
103 self->tx_.emplace_back(std::move(buf));
104 self->report_backpressure(self->queue_bytes_);
105 if (!self->writing_) self->do_write();
106 });
107 return;
108 }
109 }
110
111 // Fallback to regular allocation for large buffers or pool exhaustion
112 std::vector<uint8_t> fallback(data.begin(), data.end());
113
114 net::post(strand_, [self = shared_from_this(), buf = std::move(fallback)]() mutable {
115 if (!self->alive_ || self->closing_) return; // Double-check in case session was closed
116 const auto added = buf.size();
117 self->maybe_flush_for_keep_latest(added);
118 if (self->queue_bytes_ + added > self->bp_limit_) {
119 UNILINK_LOG_ERROR("tcp_server_session", "write", "Queue limit exceeded, dropping message");
120 self->report_backpressure(self->queue_bytes_ + added);
121 return;
122 }
123
124 self->queue_bytes_ += added;
125 self->tx_.emplace_back(std::move(buf));
126 self->report_backpressure(self->queue_bytes_);
127 if (!self->writing_) self->do_write();
128 });
129}

201void TcpServerSession::stop() {
202 if (closing_.exchange(true)) return;
203 auto self = shared_from_this();
204 net::post(strand_, [self] {
205 // Clear callbacks on the strand to block further user callbacks after stop.
206 self->on_bytes_ = nullptr;
207 self->on_bp_ = nullptr;
208 self->on_close_ = nullptr;
209 self->idle_timer_.cancel();
210 self->do_close();
211 });

214void TcpServerSession::cancel() {
215 auto self = shared_from_this();
216 net::dispatch(strand_, [self] {
217 self->idle_timer_.cancel();
218 boost::system::error_code ec;
219 // Cancelling the socket via close() causes ongoing operations to complete with operation_aborted.
220 // Unlike stop(), this does NOT set closing_ flag immediately, allowing the
221 // error handler to run normally and trigger do_close() via the error path.
222 if (self->socket_) {
223 self->socket_->close(ec);
224 }
225 });
226}

228void TcpServerSession::start_read() {
229 auto self = shared_from_this();
230 socket_->async_read_some(
231 net::buffer(rx_.data(), rx_.size()), net::bind_executor(strand_, [self](auto ec, std::size_t n) {
232 if (self->closing_ || !self->alive_) return;
233 if (ec) {
234 self->do_close();
235 return;
236 }
237 self->reset_idle_timer();
238 if (self->on_bytes_) {
239 try {
240 self->on_bytes_(memory::ConstByteSpan(self->rx_.data(), n));
241 } catch (const std::exception& e) {
242 UNILINK_LOG_ERROR("tcp_server_session", "on_bytes",
243 "Exception in on_bytes callback: " + std::string(e.what()));
244 self->do_close();
245 return;
246 } catch (...) {
247 UNILINK_LOG_ERROR("tcp_server_session", "on_bytes", "Unknown exception in on_bytes callback");
248 self->do_close();
249 return;
250 }
251 }
252 self->start_read();
253 }));
254}

256void TcpServerSession::do_write() {
257 if (tx_.empty()) {
258 writing_ = false;
259 return;
260 }
261 writing_ = true;
262 auto self = shared_from_this();
263
264 // Move buffer out of queue immediately to ensure lifetime safety during async op
265 // Optimization: Move into current_write_buffer_ to keep it alive during async op
266 // without allocating a shared_ptr control block.
267 current_write_buffer_ = std::move(tx_.front());
268 tx_.pop_front();
269
270 auto& current = *current_write_buffer_;
271
272 auto on_write = [self](const boost::system::error_code& ec, std::size_t n) {
273 // Release the buffer immediately
274 self->current_write_buffer_.reset();
275
276 if (self->closing_ || !self->alive_) return;
277 if (self->queue_bytes_ >= n) {
278 self->queue_bytes_ -= n;
279 } else {
280 self->queue_bytes_ = 0;
281 }
282 self->report_backpressure(self->queue_bytes_);
283
284 if (ec) {
285 self->do_close();
286 return;
287 }
288 self->reset_idle_timer();
289 self->do_write();
290 };
291
292 std::visit(
293 [&](const auto& buf) {
294 using T = std::decay_t<decltype(buf)>;
295 if constexpr (std::is_same_v<T, std::shared_ptr<const std::vector<uint8_t>>>) {
296 socket_->async_write(net::buffer(buf->data(), buf->size()), net::bind_executor(strand_, on_write));
297 } else {
298 socket_->async_write(net::buffer(buf.data(), buf.size()), net::bind_executor(strand_, on_write));
299 }
300 },
301 current);
302}

304void TcpServerSession::do_close() {
305 if (cleanup_done_.exchange(true)) return; // Ensures cleanup runs only once
306
307 alive_.store(false);
308 closing_.store(true); // Redundant, but ensures consistency
309
310 // Safely invoke on_close callback
311 auto close_cb = std::move(on_close_);
312
313 // Clear all callbacks to prevent any further invocations
314 on_bytes_ = nullptr;
315 on_bp_ = nullptr;
316 on_close_ = nullptr;
317 idle_timer_.cancel();
318
319 UNILINK_LOG_INFO("tcp_server_session", "disconnect", "Client disconnected");
320 boost::system::error_code ec;
321 socket_->shutdown(tcp::socket::shutdown_both, ec);
322 socket_->close(ec);
323
324 // Release memory immediately
325 tx_.clear();
326 queue_bytes_ = 0;
327
328 if (close_cb) {
329 try {
330 close_cb();
331 } catch (const std::exception& e) {
332 UNILINK_LOG_ERROR("tcp_server_session", "on_close", "Exception in on_close callback: " + std::string(e.what()));
333 } catch (...) {
334 UNILINK_LOG_ERROR("tcp_server_session", "on_close", "Unknown exception in on_close callback");
335 }
336 }
337}

339void TcpServerSession::maybe_flush_for_keep_latest(size_t added) {
340 if (bp_strategy_ != base::constants::BackpressureStrategy::KeepLatest) return;
341 if (backpressure_active_ || queue_bytes_ + added > bp_high_) {
342 tx_.clear();
343 queue_bytes_ = 0;

347void TcpServerSession::report_backpressure(size_t queued_bytes) {
348 if (closing_ || !alive_ || !on_bp_) return;
349 if (!backpressure_active_ && queued_bytes >= bp_high_) {
350 backpressure_active_ = true;
351 try {
352 on_bp_(queued_bytes);
353 } catch (const std::exception& e) {
354 UNILINK_LOG_ERROR("tcp_server_session", "on_backpressure",
355 "Exception in backpressure callback: " + std::string(e.what()));
356 } catch (...) {
357 UNILINK_LOG_ERROR("tcp_server_session", "on_backpressure", "Unknown exception in backpressure callback");
358 }
359 } else if (backpressure_active_ && queued_bytes <= bp_low_) {
360 backpressure_active_ = false;
361 try {
362 on_bp_(queued_bytes);
363 } catch (const std::exception& e) {
364 UNILINK_LOG_ERROR("tcp_server_session", "on_backpressure",
365 "Exception in backpressure callback: " + std::string(e.what()));
366 } catch (...) {
367 UNILINK_LOG_ERROR("tcp_server_session", "on_backpressure", "Unknown exception in backpressure callback");
368 }
369 }

372void TcpServerSession::reset_idle_timer() {
373 if (idle_timeout_ms_ <= 0) return;
374
375 // Cancel any existing timer
376 idle_timer_.cancel();
377
378 // Reset timer
379 idle_timer_.expires_after(std::chrono::milliseconds(idle_timeout_ms_));
380
381 auto self = shared_from_this();
382 idle_timer_.async_wait(net::bind_executor(strand_, [self](const boost::system::error_code& ec) {
383 if (ec == boost::asio::error::operation_aborted) return;
384 if (!self->alive_ || self->closing_) return;
385
386 UNILINK_LOG_WARNING("tcp_server_session", "timeout", "Connection idle timeout expired, closing session");
387 self->do_close();
388 }));
389}
