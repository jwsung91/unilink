75  void fulfill_all_locked(bool value) {
76  for (auto& promise : pending_promises) {
77  try {
78  promise.set_value(value);
79  } catch (...) {
80  }
81  }
82  pending_promises.clear();

85  std::future<bool> start() {
86  std::unique_lock<std::mutex> lock(mutex_);
87  if (channel && channel->is_connected()) {
88  std::promise<bool> p;
89  p.set_value(true);
90  return p.get_future();
91  }
92 
93  std::promise<bool> p;
94  auto future = p.get_future();
95  pending_promises.emplace_back(std::move(p));
96 
97  if (started.exchange(true)) {
98  return future;
99  }
100 
101  if (!channel) {
102  channel = factory::ChannelFactory::create(cfg, external_ioc);
103  setup_internal_handlers();
104  }
105 
106  lock.unlock();
107  channel->start();
108  if (use_external_context && manage_external_context && !external_thread.joinable()) {
109  if (external_ioc->stopped()) external_ioc->restart();
110  work_guard = std::make_unique<boost::asio::executor_work_guard<boost::asio::io_context::executor_type>>(
111  boost::asio::make_work_guard(*external_ioc));
112  external_thread = std::thread([this, ioc = external_ioc]() {
113  try {
114  ioc->run();
115  } catch (...) {
116  }
117  });
118  }
119  return future;
120  }

122  void stop() {
123  std::unique_lock<std::mutex> lock(mutex_);
124  if (!started.exchange(false)) {
125  fulfill_all_locked(false);
126  return;
127  }
128 
129  if (channel) {
130  channel->on_bytes(nullptr);
131  channel->on_state(nullptr);
132  lock.unlock();
133  channel->stop();
134  lock.lock();
135  }
136 
137  if (work_guard) {
138  work_guard.reset();
139  }
140 
141  if (use_external_context && manage_external_context && external_thread.joinable()) {
142  if (external_ioc) {
143  external_ioc->stop();
144  }
145  if (std::this_thread::get_id() != external_thread.get_id()) {
146  lock.unlock();
147  external_thread.join();
148  lock.lock();
149  } else {
150  external_thread.detach();
151  }
152  }
153 
154  fulfill_all_locked(false);
155 
156  if (framer) {
157  framer->reset();
158  }
159  }

161  void setup_internal_handlers() {
162  if (!channel) return;
163 
164  std::weak_ptr<bool> weak_alive = alive_marker;
165 
166  channel->on_bytes([this, weak_alive](memory::ConstByteSpan data) {
167  auto alive = weak_alive.lock();
168  if (!alive) return;
169 
170  // 1. Raw data handler
171  MessageHandler handler;
172  {
173  std::lock_guard<std::mutex> lock(mutex_);
174  handler = data_handler;
175  }
176  if (handler) {
177  std::string str_data = base::safe_convert::uint8_to_string(data.data(), data.size());
178  handler(MessageContext(0, str_data));
179  }
180 
181  // 2. Framer integration
182  std::lock_guard<std::mutex> lock(mutex_);
183  if (framer) {
184  framer->push_bytes(data);
185  }
186  });
187 
188  channel->on_state([this, weak_alive](base::LinkState state) {
189  auto alive = weak_alive.lock();
190  if (!alive) return;
191 
192  switch (state) {
193  case base::LinkState::Connected:
194  case base::LinkState::Listening: {
195  ConnectionHandler handler;
196  {
197  std::lock_guard<std::mutex> lock(mutex_);
198  fulfill_all_locked(true);
199  handler = connect_handler;
200  }
201  if (handler) handler(ConnectionContext(0));
202  break;
203  }
204  case base::LinkState::Closed:
205  case base::LinkState::Error:
206  case base::LinkState::Idle: {
207  ConnectionHandler disconnect_handler_snapshot;
208  ErrorHandler error_handler_snapshot;
209  {
210  std::lock_guard<std::mutex> lock(mutex_);
211  fulfill_all_locked(false);
212  if (state == base::LinkState::Error) {
213  error_handler_snapshot = error_handler;
214  } else {
215  disconnect_handler_snapshot = disconnect_handler;
216  }
217  }
218  if (disconnect_handler_snapshot) {
219  disconnect_handler_snapshot(ConnectionContext(0));
220  }
221  if (error_handler_snapshot) {
222  error_handler_snapshot(ErrorContext(ErrorCode::IoError, "Connection error"));
223  }
224  break;
225  }
226  default:
227  break;
228  }
229  });
230  }
231 
232  // Attach the stored message_handler to framer->on_message().
233  // Must be called with mutex_ already held.
234  void attach_framer_callback() {
235  if (!framer) return;
236  framer->on_message([this](memory::ConstByteSpan msg) {
237  MessageHandler handler;
238  {
239  std::lock_guard<std::mutex> lock(mutex_);
240  handler = message_handler;
241  }
242  if (handler) {
243  handler(MessageContext(0, base::safe_convert::uint8_to_string(msg.data(), msg.size())));
244  }
245  });
246  }
