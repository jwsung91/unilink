46 void Udp::start() {
47  if (started_) return;
48 
49  if (use_external_context_) {
50  if (!external_ioc_) {
51  throw std::runtime_error("External io_context is not set");
52  }
53  if (manage_external_context_ && external_ioc_->stopped()) {
54  external_ioc_->restart();
55  }
56  }
57 
58  if (!channel_) {
59  channel_ = factory::ChannelFactory::create(cfg_, external_ioc_);
60  setup_internal_handlers();
61  }
62 
63  channel_->start();
64  if (use_external_context_ && manage_external_context_ && !external_thread_.joinable()) {
65  external_thread_ = std::thread([ioc = external_ioc_]() {
66  boost::asio::executor_work_guard<boost::asio::io_context::executor_type> guard(ioc->get_executor());
67  ioc->run();
68  });
69  }
70  started_ = true;
71 }

73 void Udp::stop() {
74  if (!started_ || !channel_) return;
75 
76  channel_->stop();
77  std::this_thread::sleep_for(std::chrono::milliseconds(50));
78  channel_.reset();
79  started_ = false;

83  external_ioc_->stop();

144 void Udp::notify_state_change(base::LinkState state) {
145  switch (state) {
146  case base::LinkState::Connected:

148  break;
149  case base::LinkState::Closed:
150  if (disconnect_handler_) disconnect_handler_();
151  break;
152  case base::LinkState::Error:
153  if (error_handler_) error_handler_("UDP channel error occurred");
154  break;
155  default:
156  break;
