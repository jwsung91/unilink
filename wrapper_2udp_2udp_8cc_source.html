61  void stop() {
62  if (!started) return;
63 
64  if (channel) {
65  // Clear handlers first to prevent callbacks during shutdown
66  channel->on_bytes(nullptr);
67  channel->on_state(nullptr);
68  channel->stop();
69  }
70 
71  if (use_external_context && manage_external_context) {
72  if (external_ioc) {
73  external_ioc->stop();
74  }
75  if (external_thread.joinable()) {
76  try {
77  external_thread.join();
78  } catch (...) {
79  }
80  }
81  }
82 
83  started = false;
84  }

86  void setup_internal_handlers(Udp* parent) {
87  if (!channel) return;

89  channel->on_bytes([this](memory::ConstByteSpan data) {

102  void notify_state_change(base::LinkState state) {
103  switch (state) {
104  case base::LinkState::Connected:
105  if (connect_handler) connect_handler();
106  break;
107  case base::LinkState::Closed:
108  if (disconnect_handler) disconnect_handler();
109  break;
110  case base::LinkState::Error:
111  if (error_handler) error_handler("Connection error");
112  break;
113  default:
114  break;
115  }

130 void Udp::start() {
131  if (pimpl_->started) return;
132 
133  if (pimpl_->use_external_context) {
134  if (!pimpl_->external_ioc) {
135  throw std::runtime_error("External io_context is not set");
136  }
137  if (pimpl_->manage_external_context) {
138  if (pimpl_->external_ioc->stopped()) {
139  pimpl_->external_ioc->restart();
140  }
141  }
142  }
143 
144  if (!pimpl_->channel) {
145  pimpl_->channel = factory::ChannelFactory::create(pimpl_->cfg, pimpl_->external_ioc);
146  pimpl_->setup_internal_handlers(this);
147  }
148 
149  pimpl_->channel->start();
150  if (pimpl_->use_external_context && pimpl_->manage_external_context && !pimpl_->external_thread.joinable()) {
151  pimpl_->external_thread = std::thread([ioc = pimpl_->external_ioc]() {
152  boost::asio::executor_work_guard<boost::asio::io_context::executor_type> guard(ioc->get_executor());
153  ioc->run();
154  });
155  }
156  pimpl_->started = true;
157 }
