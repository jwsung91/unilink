49 void Udp::start() {
50  if (started_) return;
51 
52  if (use_external_context_) {
53  if (!external_ioc_) {
54  throw std::runtime_error("External io_context is not set");
55  }
56  if (manage_external_context_) {
57  if (external_ioc_->stopped()) {
58  external_ioc_->restart();
59  }
60  }
61  }
62 
63  if (!channel_) {
64  channel_ = factory::ChannelFactory::create(cfg_, external_ioc_);
65  }
66 
67  // Ensure handlers are set up (safe to call multiple times)
68  setup_internal_handlers();
69 
70  channel_->start();
71  if (use_external_context_ && manage_external_context_ && !external_thread_.joinable()) {
72  external_thread_ = std::thread([ioc = external_ioc_]() {
73  boost::asio::executor_work_guard<boost::asio::io_context::executor_type> guard(ioc->get_executor());
74  ioc->run();
75  });
76  }
77  started_ = true;
78 }

80 void Udp::stop() {
81  if (!started_ || !channel_) return;
82 
83  // Clear callbacks to prevent use-after-free
84  channel_->on_bytes({});
85  channel_->on_state({});
86  channel_->on_backpressure({});
87 
88  channel_->stop();
89 
90  // Manually notify closed state since we detached callbacks
91  notify_state_change(base::LinkState::Closed);
92 
93  if (use_external_context_ && manage_external_context_) {
94  if (external_ioc_) {
95  external_ioc_->stop();
96  }
97  if (external_thread_.joinable()) {
98  external_thread_.join();
99  }
100  }
101 
102  started_ = false;
103 }

153 void Udp::setup_internal_handlers() {
154  if (!channel_) return;
155 
156  auto self = weak_from_this();
157 
158  channel_->on_bytes([self](memory::ConstByteSpan data) {
159  if (auto s = self.lock()) {
160  if (s->bytes_handler_) {
161  s->bytes_handler_(data);
162  }
163  if (s->data_handler_) {
164  std::string str_data = common::safe_convert::uint8_to_string(data.data(), data.size());
165  s->data_handler_(str_data);
166  }
167  }
168  });
169 
170  channel_->on_state([self](base::LinkState state) {
171  if (auto s = self.lock()) {
172  s->notify_state_change(state);
173  }
174  });
175 }

177 void Udp::notify_state_change(base::LinkState state) {
178  switch (state) {
179  case base::LinkState::Connected:
180  if (connect_handler_) connect_handler_();
181  break;
182  case base::LinkState::Closed:
183  if (disconnect_handler_) disconnect_handler_();
184  break;
185  case base::LinkState::Error:
186  if (error_handler_) error_handler_("Connection error");
187  break;
188  default:
189  break;
190  }
191 }
