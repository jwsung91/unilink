144 void fulfill_all_locked(bool value) {
145 for (auto& p : pending_promises_) {
146 try {
147 p.set_value(value);
148 } catch (...) {
149 }
150 }
151 pending_promises_.clear();

154 void flush_batches() {
155 std::unique_lock<std::shared_mutex> lock(mutex_);
156 if (!data_batch_queue_.empty()) {
157 auto handler = data_batch_handler_;
158 auto batch = std::move(data_batch_queue_);
159 data_batch_queue_.clear();
160 if (handler) {
161 lock.unlock();
162 handler(batch);
163 lock.lock();
164 }
165 }
166 if (!message_batch_queue_.empty()) {
167 auto handler = message_batch_handler_;
168 auto batch = std::move(message_batch_queue_);
169 message_batch_queue_.clear();
170 if (handler) {
171 lock.unlock();
172 handler(batch);
173 lock.lock();
174 }
175 }
176 if (batch_timer_) {
177 batch_timer_->cancel();
178 }
179 }

181 void schedule_batch_timer() {
182 if (!batch_timer_) return;
183 batch_timer_->expires_after(max_batch_latency_);
184 batch_timer_->async_wait(
185 [this, weak_alive = std::weak_ptr<bool>(alive_marker_)](const boost::system::error_code& ec) {
186 if (ec) return;
187 auto alive = weak_alive.lock();
188 if (!alive) return;
189 flush_batches();
190 });
191 }

193 std::future<bool> start() {
194 std::unique_lock<std::shared_mutex> lock(mutex_);
195 if (is_listening_.load()) {
196 std::promise<bool> p;
197 p.set_value(true);
198 return p.get_future();
199 }
200 std::promise<bool> p;
201 auto f = p.get_future();
202 pending_promises_.push_back(std::move(p));
203 if (started_.exchange(true)) return f;
204
205 if (!channel_) {
206 config::TcpServerConfig config;
207 config.port = port_;
208 config.enable_port_retry = port_retry_enabled_.load();
209 config.max_port_retries = max_port_retries_.load();
210 config.port_retry_interval_ms = port_retry_interval_ms_.load();
211 config.idle_timeout_ms = idle_timeout_ms_.load();
212 config.backpressure_threshold = backpressure_threshold_.load();
213 config.backpressure_strategy = backpressure_strategy_.load();
214
215 channel_ = factory::ChannelFactory::create(config, external_ioc_);
216 transport_cache_ = std::dynamic_pointer_cast<transport::TcpServer>(channel_);
217 setup_internal_handlers();
218
219 if (client_limit_enabled_.load()) {
220 auto transport_server = std::dynamic_pointer_cast<transport::TcpServer>(channel_);
221 if (transport_server) {
222 if (max_clients_.load() == 0)
223 transport_server->set_unlimited_clients();
224 else
225 transport_server->set_client_limit(max_clients_.load());
226 }
227 }
228 }
229 lock.unlock();
230 channel_->start();
231 if (use_external_context_.load() && manage_external_context_.load() && !external_thread_.joinable()) {
232 if (external_ioc_ && external_ioc_->stopped()) {
233 external_ioc_->restart();
234 }
235 work_guard_ = std::make_unique<boost::asio::executor_work_guard<boost::asio::io_context::executor_type>>(
236 boost::asio::make_work_guard(*external_ioc_));
237 external_thread_ = std::thread([ioc = external_ioc_]() {
238 try {
239 ioc->run();
240 } catch (...) {
241 }
242 });
243 }
244 return f;
245 }

247 void stop() {
248 bool should_join = false;
249 {
250 std::unique_lock<std::shared_mutex> lock(mutex_);
251 if (!started_.exchange(false)) {
252 is_listening_.store(false);
253 fulfill_all_locked(false);
254 return;
255 }
256 bp_cv_.notify_all();
257 if (batch_timer_) {
258 batch_timer_->cancel();
259 batch_timer_.reset();
260 }
261 if (channel_) {
262 channel_->on_bytes(nullptr);
263 channel_->on_state(nullptr);
264 auto transport_server = std::dynamic_pointer_cast<transport::TcpServer>(channel_);
265 if (transport_server) transport_server->request_stop();
266 lock.unlock();
267 channel_->stop();
268 lock.lock();
269 }
270 if (use_external_context_.load() && manage_external_context_.load()) {
271 if (work_guard_) work_guard_.reset();
272 if (external_ioc_) external_ioc_->stop();
273 should_join = true;
274 }
275 fulfill_all_locked(false);
276 is_listening_.store(false);
277 }
278 if (should_join && external_thread_.joinable()) {
279 try {
280 if (std::this_thread::get_id() != external_thread_.get_id()) {
281 external_thread_.join();
282 } else {
283 external_thread_.detach();
284 }
285 } catch (...) {
286 }
287 }
288 std::unique_lock<std::shared_mutex> lock(mutex_);
289 channel_.reset();
290 }

310 bool broadcast(std::string_view data) {
311 if (backpressure_strategy_.load() == base::constants::BackpressureStrategy::Reliable) {
312 std::vector<ClientId> clients;
313 {
314 std::shared_lock<std::shared_mutex> lock(mutex_);
315 if (transport_cache_) clients = transport_cache_->connected_clients();
316 }
317 bool any_sent = false;
318 for (auto id : clients) {
319 if (send_to_blocking(id, data)) any_sent = true;
320 }
321 return any_sent;
322 }
323 return try_broadcast(data);
324 }
325
326 bool send_to_blocking(ClientId client_id, std::string_view data) {
327 std::unique_lock<std::mutex> lock(bp_mutex_);
328 bp_cv_.wait(lock, [this, client_id]() {
329 std::shared_lock<std::shared_mutex> rlock(mutex_);
330 const auto& ts = transport_cache_;
331 return !started_.load() || !ts || !ts->is_backpressure_active(client_id);
332 });
333 return try_send_to(client_id, data);
334 }

336 void setup_internal_handlers() {
337 if (!channel_) return;
338
339 batch_timer_ = std::make_unique<boost::asio::steady_timer>(channel_->get_executor());
340
341 std::weak_ptr<bool> weak_alive = alive_marker_;
342 auto transport_server = std::dynamic_pointer_cast<transport::TcpServer>(channel_);
343 if (transport_server) {
344 transport_server->on_multi_connect([this, weak_alive](ClientId id, const std::string& info) {
345 auto alive = weak_alive.lock();
346 if (!alive) return;
347
348 ConnectionHandler handler;
349 {
350 std::unique_lock<std::shared_mutex> lock(mutex_);
351 if (framer_factory_) {
352 auto framer = framer_factory_();
353 if (framer) {
354 framer->on_message([this, id](memory::ConstByteSpan msg) {
355 std::unique_lock<std::shared_mutex> lock(mutex_);
356 if (message_batch_handler_) {
357 message_batch_queue_.emplace_back(id, memory::SafeDataBuffer(msg));
358 if (message_batch_queue_.size() >= max_batch_size_) {
359 auto handler = message_batch_handler_;
360 auto batch = std::move(message_batch_queue_);
361 message_batch_queue_.clear();
362 lock.unlock();
363 handler(batch);
364 } else if (message_batch_queue_.size() == 1) {
365 schedule_batch_timer();
366 }
367 return;
368 }
369
370 MessageHandler on_message_handler = on_message_;
371 if (on_message_handler) {
372 lock.unlock();
373 on_message_handler(MessageContext(id, memory::SafeDataBuffer(msg)));
374 }
375 });
376 framers_[id] = std::move(framer);
377 }
378 }
379 handler = on_client_connect_;
380 }
381 if (handler) handler(ConnectionContext(id, info));
382 });
383 transport_server->on_multi_data([this, weak_alive](ClientId id, memory::ConstByteSpan data_span) {
384 auto alive = weak_alive.lock();
385 if (!alive) return;
386
387 std::unique_lock<std::shared_mutex> lock(mutex_);
388 if (data_batch_handler_) {
389 data_batch_queue_.emplace_back(id, memory::SafeDataBuffer(data_span));
390 if (data_batch_queue_.size() >= max_batch_size_) {
391 auto handler = data_batch_handler_;
392 auto batch = std::move(data_batch_queue_);
393 data_batch_queue_.clear();
394 lock.unlock();
395 handler(batch);
396 } else if (data_batch_queue_.size() == 1) {
397 schedule_batch_timer();
398 }
399 } else {
400 MessageHandler handler = on_data_;
401 if (handler) {
402 lock.unlock();
403 handler(MessageContext(id, memory::SafeDataBuffer(data_span)));
404 lock.lock();
405 }
406 }
407
408 auto it = framers_.find(id);
409 if (it != framers_.end()) {
410 it->second->push_bytes(data_span);
411 }
412 });
413 transport_server->on_multi_disconnect([this, weak_alive](ClientId id) {
414 auto alive = weak_alive.lock();
415 if (!alive) return;
416
417 ConnectionHandler handler;
418 {
419 std::unique_lock<std::shared_mutex> lock(mutex_);
420 framers_.erase(id);
421 handler = on_disconnect_;
422 }
423 if (handler) handler(ConnectionContext(id));
424 });
425
426 transport_server->on_backpressure([this, weak_alive](size_t queued) {
427 bp_cv_.notify_all();
428 auto alive = weak_alive.lock();
429 if (!alive) return;
430 std::shared_lock<std::shared_mutex> lock(mutex_);
431 if (on_backpressure_) on_backpressure_(queued);
432 });
433 }
434 channel_->on_state([this, weak_alive](base::LinkState state) {
435 auto alive = weak_alive.lock();
436 if (!alive) return;
437
438 if (state == base::LinkState::Listening) {
439 is_listening_.store(true);
440 std::unique_lock<std::shared_mutex> lock(mutex_);
441 fulfill_all_locked(true);
442 } else if (state == base::LinkState::Error || state == base::LinkState::Closed ||
443 state == base::LinkState::Idle) {
444 ErrorHandler handler;
445 is_listening_.store(false);
446 {
447 std::unique_lock<std::shared_mutex> lock(mutex_);
448 fulfill_all_locked(false);
449 if (state == base::LinkState::Error) {
450 handler = on_error_;
451 }
452 }
453 if (handler) {
454 handler(ErrorContext(ErrorCode::IoError, "Server error"));
455 }
456 }
457 });
458 }
