122 void fulfill_all_locked(bool value) {
123 for (auto& p : pending_promises_) {
124 try {
125 p.set_value(value);
126 } catch (...) {
127 }
128 }
129 pending_promises_.clear();
130 }

132 std::future<bool> start() {
133 std::unique_lock<std::shared_mutex> lock(mutex_);
134 if (is_listening_.load()) {
135 std::promise<bool> p;
136 p.set_value(true);
137 return p.get_future();
138 }
139 std::promise<bool> p;
140 auto f = p.get_future();
141 pending_promises_.push_back(std::move(p));
142 if (started_.exchange(true)) return f;
143
144 if (!channel_) {
145 config::TcpServerConfig config;
146 config.port = port_;
147 config.enable_port_retry = port_retry_enabled_;
148 config.max_port_retries = max_port_retries_;
149 config.port_retry_interval_ms = port_retry_interval_ms_;
150 config.idle_timeout_ms = idle_timeout_ms_;
151
152 channel_ = factory::ChannelFactory::create(config, external_ioc_);
153 transport_cache_ = std::dynamic_pointer_cast<transport::TcpServer>(channel_);
154 setup_internal_handlers();
155
156 if (client_limit_enabled_) {
157 auto transport_server = std::dynamic_pointer_cast<transport::TcpServer>(channel_);
158 if (transport_server) {
159 if (max_clients_ == 0)
160 transport_server->set_unlimited_clients();
161 else
162 transport_server->set_client_limit(max_clients_);
163 }
164 }
165 }
166 lock.unlock();
167 channel_->start();
168 if (use_external_context_ && manage_external_context_ && !external_thread_.joinable()) {
169 if (external_ioc_ && external_ioc_->stopped()) {
170 external_ioc_->restart();
171 }
172 work_guard_ = std::make_unique<boost::asio::executor_work_guard<boost::asio::io_context::executor_type>>(
173 boost::asio::make_work_guard(*external_ioc_));
174 external_thread_ = std::thread([ioc = external_ioc_]() {
175 try {
176 ioc->run();
177 } catch (...) {
178 }
179 });
180 }
181 return f;
182 }

184 void stop() {
185 bool should_join = false;
186 {
187 std::unique_lock<std::shared_mutex> lock(mutex_);
188 if (!started_.exchange(false)) {
189 is_listening_ = false;
190 fulfill_all_locked(false);
191 return;
192 }
193 if (channel_) {
194 channel_->on_bytes(nullptr);
195 channel_->on_state(nullptr);
196 auto transport_server = std::dynamic_pointer_cast<transport::TcpServer>(channel_);
197 if (transport_server) transport_server->request_stop();
198 lock.unlock();
199 channel_->stop();
200 lock.lock();
201 }
202 if (use_external_context_ && manage_external_context_) {
203 if (work_guard_) work_guard_.reset();
204 if (external_ioc_) external_ioc_->stop();
205 should_join = true;
206 }
207 fulfill_all_locked(false);
208 is_listening_ = false;
209 }
210 if (should_join && external_thread_.joinable()) {
211 try {
212 if (std::this_thread::get_id() != external_thread_.get_id()) {
213 external_thread_.join();
214 } else {
215 external_thread_.detach();
216 }
217 } catch (...) {
218 }
219 }
220 std::lock_guard<std::shared_mutex> lock(mutex_);
221 channel_.reset();
222 }

224 void setup_internal_handlers() {
225 if (!channel_) return;
226 std::weak_ptr<bool> weak_alive = alive_marker_;
227 auto transport_server = std::dynamic_pointer_cast<transport::TcpServer>(channel_);
228 if (transport_server) {
229 transport_server->on_multi_connect([this, weak_alive](ClientId id, const std::string& info) {
230 auto alive = weak_alive.lock();
231 if (!alive) return;
232
233 ConnectionHandler handler;
234 {
235 std::lock_guard<std::shared_mutex> lock(mutex_);
236 if (framer_factory_) {
237 auto framer = framer_factory_();
238 if (framer) {
239 framer->on_message([this, id](memory::ConstByteSpan msg) {
240 MessageHandler on_message_handler;
241 {
242 std::lock_guard<std::shared_mutex> lock(mutex_);
243 on_message_handler = on_message_;
244 }
245 if (on_message_handler) {
246 std::string str_msg = base::safe_convert::uint8_to_string(msg.data(), msg.size());
247 on_message_handler(MessageContext(id, str_msg));
248 }
249 });
250 framers_[id] = std::move(framer);
251 }
252 }
253 handler = on_client_connect_;
254 }
255 if (handler) handler(ConnectionContext(id, info));
256 });
257 transport_server->on_multi_data([this, weak_alive](ClientId id, memory::ConstByteSpan data_span) {
258 auto alive = weak_alive.lock();
259 if (!alive) return;
260
261 std::string data(reinterpret_cast<const char*>(data_span.data()), data_span.size());
262 MessageHandler handler;
263 {
264 std::shared_lock<std::shared_mutex> lock(mutex_);
265 handler = on_data_;
266 }
267 if (handler) handler(MessageContext(id, data));
268
269 std::shared_lock<std::shared_mutex> lock(mutex_);
270 auto it = framers_.find(id);
271 if (it != framers_.end()) {
272 it->second->push_bytes(data_span);
273 }
274 });
275 transport_server->on_multi_disconnect([this, weak_alive](ClientId id) {
276 auto alive = weak_alive.lock();
277 if (!alive) return;
278
279 ConnectionHandler handler;
280 {
281 std::lock_guard<std::shared_mutex> lock(mutex_);
282 framers_.erase(id);
283 handler = on_disconnect_;
284 }
285 if (handler) handler(ConnectionContext(id));
286 });
287 }
288 channel_->on_state([this, weak_alive](base::LinkState state) {
289 auto alive = weak_alive.lock();
290 if (!alive) return;
291
292 if (state == base::LinkState::Listening) {
293 is_listening_ = true;
294 std::lock_guard<std::shared_mutex> lock(mutex_);
295 fulfill_all_locked(true);
296 } else if (state == base::LinkState::Error || state == base::LinkState::Closed ||
297 state == base::LinkState::Idle) {
298 ErrorHandler handler;
299 is_listening_ = false;
300 {
301 std::lock_guard<std::shared_mutex> lock(mutex_);
302 fulfill_all_locked(false);
303 if (state == base::LinkState::Error) {
304 handler = on_error_;
305 }
306 }
307 if (handler) {
308 handler(ErrorContext(ErrorCode::IoError, "Server error"));
309 }
310 }
311 });
312 }
