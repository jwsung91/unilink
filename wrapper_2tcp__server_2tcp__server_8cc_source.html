54 void TcpServer::start() {
55  std::lock_guard<std::mutex> lock(mutex_);
56  if (started_) return;
57 
58  if (use_external_context_) {
59  if (!external_ioc_) {
60  throw std::runtime_error("External io_context is not set");
61  }
62  if (manage_external_context_) {
63  if (external_ioc_->stopped()) {
64  external_ioc_->restart();
65  }
66  }
67  }
68 
69  if (!channel_) {
70  // Use improved Factory
71  config::TcpServerConfig config;
72  config.port = port_;
73  config.enable_port_retry = port_retry_enabled_;
74  config.max_port_retries = max_port_retries_;
75  config.port_retry_interval_ms = port_retry_interval_ms_;
76 
77  channel_ = factory::ChannelFactory::create(config, external_ioc_);
78  setup_internal_handlers();
79 
80  // Apply stored client limit configuration
81  if (client_limit_enabled_) {
82  auto transport_server = std::dynamic_pointer_cast<transport::TcpServer>(channel_);
83  if (transport_server) {
84  if (max_clients_ == 0) {
85  transport_server->set_unlimited_clients();
86  } else {
87  transport_server->set_client_limit(max_clients_);
88  }
89  }
90  }
91  }
92 
93  channel_->start();
94  if (use_external_context_ && manage_external_context_ && !external_thread_.joinable()) {
95  external_thread_ = std::thread([ioc = external_ioc_]() {
96  boost::asio::executor_work_guard<boost::asio::io_context::executor_type> guard(ioc->get_executor());
97  ioc->run();
98  });
99  }
100  started_ = true;
101 }

103 void TcpServer::stop() {
104  std::shared_ptr<interface::Channel> local_channel;
105  bool posted_transport_stop = false;
106  {
107  std::lock_guard<std::mutex> lock(mutex_);
108  if (!started_ && !channel_) {
109  return;
110  }
111 
112  // Detach callbacks to prevent use-after-free if background events arrive
113  if (channel_) {
114  channel_->on_bytes({});
115  channel_->on_state({});
116  channel_->on_backpressure({});
117 
118  // If underlying transport is TcpServer, prefer async stop to avoid callback reentrancy issues
119  auto transport_server = std::dynamic_pointer_cast<transport::TcpServer>(channel_);
120  if (transport_server) {
121  transport_server->request_stop();
122  posted_transport_stop = true;
123  }
124  }
125 
126  started_ = false;
127  is_listening_ = false;
128  if (!posted_transport_stop) {
129  local_channel = std::move(channel_);
130  }
131  }
132 
133  if (local_channel) {
134  local_channel->stop();
135  }
136 
137  if (use_external_context_) {
138  if (manage_external_context_ && external_ioc_) {
139  external_ioc_->stop();
140  }
141  if (manage_external_context_ && external_thread_.joinable()) {
142  external_thread_.join();
143  }
144  }

147 void TcpServer::send(const std::string& data) {
148  std::shared_ptr<interface::Channel> channel;
149  bool connected = false;
150  {
151  std::lock_guard<std::mutex> lock(mutex_);
152  channel = channel_;
153  connected = channel && channel->is_connected();
154  }
155 
156  if (connected && channel) {
157  auto binary_view = common::safe_convert::string_to_bytes(data);
158  channel->async_write_copy(binary_view.first, binary_view.second);
159  }

213 void TcpServer::setup_internal_handlers() {
214  if (!channel_) return;
215 
216  channel_->on_bytes([this](const uint8_t* data, size_t size) { handle_bytes(data, size); });
217 
218  channel_->on_state([this](base::LinkState state) { handle_state(state); });
219 
220  // Set multi-client callbacks
221  auto transport_server = std::dynamic_pointer_cast<transport::TcpServer>(channel_);
222  if (transport_server) {
223  if (on_multi_connect_) {
224  transport_server->on_multi_connect([this](size_t client_id, const std::string& client_info) {
225  if (on_multi_connect_) {
226  on_multi_connect_(client_id, client_info);
227  }
228  });
229  }
230 
231  if (on_multi_data_) {
232  transport_server->on_multi_data([this](size_t client_id, const std::string& data) {
233  if (on_multi_data_) {
234  on_multi_data_(client_id, data);
235  }
236  });
237  }
238 
239  if (on_multi_disconnect_) {
240  transport_server->on_multi_disconnect([this](size_t client_id) {
241  if (on_multi_disconnect_) {
242  on_multi_disconnect_(client_id);
243  }
244  });
245  }
246  }

259 void TcpServer::handle_state(base::LinkState state) {
260  {
261  std::lock_guard<std::mutex> lock(mutex_);
262  // Update listening state based on server state
263  if (state == base::LinkState::Listening) {
264  is_listening_ = true;
265  } else if (state == base::LinkState::Error || state == base::LinkState::Closed) {
266  is_listening_ = false;
267  }
268  }
269 
270  switch (state) {
271  case base::LinkState::Connected:
272  // For TCP server, Connected state means a client connected
273  // This should only be called when a client actually connects, not when server starts listening
274  if (on_connect_) {
275  on_connect_();
276  }
277  break;
278  case base::LinkState::Closed:
279  if (on_disconnect_) {
280  on_disconnect_();
281  }
282  break;
283  case base::LinkState::Error:
284  if (on_error_) {
285  on_error_("Connection error");
286  }
287  break;
288  case base::LinkState::Listening:
289  // Server is now listening - this is not a connection event
290  // Do not call on_connect_ here
291  break;
292  default:
293  break;
294  }
295 }
