58 inline std::string ts_now() {
59  using namespace std::chrono;
60  const auto now = system_clock::now();
61  const auto tt = system_clock::to_time_t(now);
62  std::tm tm{};
63 #if defined(UNILINK_PLATFORM_WINDOWS)
64  localtime_s(&tm, &tt);
65 #else
66  localtime_r(&tt, &tm);
67 #endif
68  const auto ms = duration_cast<milliseconds>(now.time_since_epoch()) % 1000;
69  std::ostringstream oss;
70  oss << std::put_time(&tm, "%F %T") << '.' << std::setw(3) << std::setfill('0') << ms.count();
71  return oss.str(); // e.g., 2025-09-15 13:07:42.123
72 }

96 // Safe memory operations
97 namespace safe_memory {
105 inline void safe_memcpy(uint8_t* dest, const uint8_t* src, size_t size) {
106  if (!dest) {
107  throw std::invalid_argument("Destination pointer is null");
108  }
109  if (!src) {
110  throw std::invalid_argument("Source pointer is null");
111  }
112  if (size == 0) {
113  return; // Nothing to copy
114  }
115  if (size > 1024 * 1024) { // 1MB limit
116  throw std::invalid_argument("Copy size too large (max 1MB)");
117  }
118 
119  std::memcpy(dest, src, size);
120 }
