83  void fulfill_all(bool value) {
84  std::lock_guard<std::mutex> lock(mutex_);
85  for (auto& p : pending_promises_) {
86  try {
87  p.set_value(value);
88  } catch (...) {
89  }
90  }
91  pending_promises_.clear();
92  }
93 
94  std::future<bool> start() {
95  std::lock_guard<std::mutex> lock(mutex_);
96  if (channel_ && channel_->is_connected()) {
97  std::promise<bool> p;
98  p.set_value(true);
99  return p.get_future();
100  }
101  std::promise<bool> p;
102  auto f = p.get_future();
103  pending_promises_.push_back(std::move(p));
104  if (started_) return f;
105 
106  if (!channel_) {
107  config::TcpClientConfig config;
108  config.host = host_;
109  config.port = port_;
110  config.retry_interval_ms = static_cast<unsigned int>(retry_interval_.count());
111  config.max_retries = max_retries_;
112  config.connection_timeout_ms = static_cast<unsigned>(connection_timeout_.count());
113  channel_ = factory::ChannelFactory::create(config, external_ioc_);
114  setup_internal_handlers();
115  }
116  channel_->start();
117  if (use_external_context_ && manage_external_context_ && !external_thread_.joinable()) {
118  work_guard_ = std::make_unique<boost::asio::executor_work_guard<boost::asio::io_context::executor_type>>(
119  external_ioc_->get_executor());
120  external_thread_ = std::thread([this, ioc = external_ioc_]() {
121  try {
122  while (started_.load() && !ioc->stopped()) {
123  if (ioc->run_one_for(std::chrono::milliseconds(50)) == 0) {
124  std::this_thread::yield();
125  }
126  }
127  } catch (...) {
128  }
129  });
130  }
131  started_ = true;
132  return f;
133  }
134 
135  void stop() {
136  bool should_join = false;
137  {
138  std::unique_lock<std::mutex> lock(mutex_);
139  if (!started_.load()) {
140  for (auto& p : pending_promises_) {
141  try {
142  p.set_value(false);
143  } catch (...) {
144  }
145  }
146  pending_promises_.clear();
147  return;
148  }
149  started_ = false;
150  if (channel_) {
151  channel_->on_bytes(nullptr);
152  channel_->on_state(nullptr);
153  channel_->stop();
154  }
155  if (use_external_context_ && manage_external_context_) {
156  if (work_guard_) work_guard_.reset();
157  if (external_ioc_) external_ioc_->stop();
158  should_join = true;
159  }
160  for (auto& p : pending_promises_) {
161  try {
162  p.set_value(false);
163  } catch (...) {
164  }
165  }
166  pending_promises_.clear();
167  }
168  if (should_join && external_thread_.joinable()) {
169  try {
170  external_thread_.join();
171  } catch (...) {
172  }
173  }
174  std::lock_guard<std::mutex> lock(mutex_);
175  channel_.reset();
176  }

178  void send(std::string_view data) {
179  if (channel_ && channel_->is_connected()) {
180  auto binary_view = common::safe_convert::string_to_bytes(data);
181  channel_->async_write_copy(memory::ConstByteSpan(binary_view.first, binary_view.second));
182  }
183  }

187  void setup_internal_handlers() {
188  if (!channel_) return;
189 
190  std::weak_ptr<bool> weak_alive = alive_marker_;
191 
192  // Explicitly do not use try-catch here to allow exceptions from handlers
193  // to propagate to transport layer for error handling (e.g., auto-reconnect)
194  channel_->on_bytes([this, weak_alive](memory::ConstByteSpan data) {
195  if (weak_alive.expired()) return;
196  if (data_handler_) {
197  std::string str_data = common::safe_convert::uint8_to_string(data.data(), data.size());
198  data_handler_(MessageContext(0, str_data));
199  }
200  });
201 
202  channel_->on_state([this, weak_alive](base::LinkState state) {
203  if (weak_alive.expired()) return;
204  if (state == base::LinkState::Connected) {
205  fulfill_all(true);
206  if (connect_handler_) connect_handler_(ConnectionContext(0));
207  } else if (state == base::LinkState::Closed || state == base::LinkState::Error) {
208  fulfill_all(false);
209  if (state == base::LinkState::Closed && disconnect_handler_) {
210  disconnect_handler_(ConnectionContext(0));
211  } else if (state == base::LinkState::Error && error_handler_) {
212  error_handler_(ErrorContext(ErrorCode::IoError, "Connection state error"));
213  }
214  }
215  });
216  }
