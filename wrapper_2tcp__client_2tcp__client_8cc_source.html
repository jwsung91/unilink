86  void start() {
87  if (started_) return;
88 
89  if (use_external_context_) {
90  if (!external_ioc_) {
91  throw std::runtime_error("External io_context is not set");
92  }
93  if (manage_external_context_) {
94  if (external_ioc_->stopped()) {
95  external_ioc_->restart();
96  }
97  }
98  }
99 
100  if (!channel_) {
101  // Create Channel
102  config::TcpClientConfig config;
103  config.host = host_;
104  config.port = port_;
105  config.retry_interval_ms = static_cast<unsigned int>(retry_interval_.count());
106  config.max_retries = max_retries_;
107  config.connection_timeout_ms = static_cast<unsigned>(connection_timeout_.count());
108  channel_ = factory::ChannelFactory::create(config, external_ioc_);
109  setup_internal_handlers();
110  }
111 
112  channel_->start();
113  if (use_external_context_ && manage_external_context_ && !external_thread_.joinable()) {
114  work_guard_.emplace(external_ioc_->get_executor());
115  external_thread_ = std::thread([ioc = external_ioc_]() { ioc->run(); });
116  }
117  started_ = true;
118  }
119 
120  void stop() {
121  if (!started_ || !channel_) return;
122 
123  channel_->stop();
124 
125  if (use_external_context_ && manage_external_context_) {
126  if (work_guard_) {
127  work_guard_.reset();
128  }
129  if (external_thread_.joinable()) {
130  external_thread_.join();
131  }
132  }
133 
134  channel_.reset();
135  started_ = false;
136  }

138  void send(std::string_view data) {
139  if (is_connected() && channel_) {
140  auto binary_view = common::safe_convert::string_to_bytes(data);
141  channel_->async_write_copy(memory::ConstByteSpan(binary_view.first, binary_view.second));
142  }
143  }

145  bool is_connected() const { return channel_ && channel_->is_connected(); }

147  void setup_internal_handlers() {
148  if (!channel_) return;
149 
150  // Convert byte data to string and pass it
151  channel_->on_bytes([this](memory::ConstByteSpan data) {
152  if (bytes_handler_) {
153  bytes_handler_(data);
154  }
155  if (data_handler_) {
156  std::string str_data = common::safe_convert::uint8_to_string(data.data(), data.size());
157  data_handler_(str_data);
158  }
159  });
160 
161  // Handle state changes
162  channel_->on_state([this](base::LinkState state) { notify_state_change(state); });
163  }
164 
165  void notify_state_change(base::LinkState state) {
166  switch (state) {
167  case base::LinkState::Connected:
168  if (connect_handler_) connect_handler_();
169  break;
170  case base::LinkState::Closed:
171  if (disconnect_handler_) disconnect_handler_();
172  break;
173  case base::LinkState::Error:
174  if (error_handler_) error_handler_("Connection error occurred");
175  break;
176  default:
177  break;
178  }
179  }
180 };
