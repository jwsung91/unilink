99 void fulfill_all_locked(bool value) {
100 for (auto& p : pending_promises_) {
101 try {
102 p.set_value(value);
103 } catch (...) {
104 }
105 }
106 pending_promises_.clear();
107 }
108
109 void flush_batches() {
110 std::unique_lock<std::shared_mutex> lock(mutex_);
111 if (!data_batch_queue_.empty()) {
112 auto handler = data_batch_handler_;
113 auto batch = std::move(data_batch_queue_);
114 data_batch_queue_.clear();
115 if (handler) {
116 lock.unlock();
117 handler(batch);
118 lock.lock();
119 }
120 }
121 if (!message_batch_queue_.empty()) {
122 auto handler = message_batch_handler_;
123 auto batch = std::move(message_batch_queue_);
124 message_batch_queue_.clear();
125 if (handler) {
126 lock.unlock();
127 handler(batch);
128 lock.lock();
129 }
130 }
131 if (batch_timer_) {
132 batch_timer_->cancel();
133 }
134 }
135
136 void schedule_batch_timer() {
137 if (!batch_timer_) return;
138 batch_timer_->expires_after(max_batch_latency_);
139 batch_timer_->async_wait(
140 [this, weak_alive = std::weak_ptr<bool>(alive_marker_)](const boost::system::error_code& ec) {
141 if (ec) return;
142 auto alive = weak_alive.lock();
143 if (!alive) return;
144 flush_batches();
145 });
146 }

148 std::future<bool> start() {
149 std::unique_lock<std::shared_mutex> lock(mutex_);
150 if (channel_ && channel_->is_connected()) {
151 std::promise<bool> p;
152 p.set_value(true);
153 return p.get_future();
154 }
155 std::promise<bool> p;
156 auto f = p.get_future();
157 pending_promises_.push_back(std::move(p));
158 if (started_.load()) return f;
159
160 if (!channel_) {
161 config::TcpClientConfig config;
162 config.host = host_;
163 config.port = port_;
164 config.retry_interval_ms = static_cast<unsigned int>(retry_interval_.count());
165 config.max_retries = max_retries_;
166 config.connection_timeout_ms = static_cast<unsigned>(connection_timeout_.count());
167 channel_ = factory::ChannelFactory::create(config, external_ioc_);
168 setup_internal_handlers();
169 }
170
171 started_.store(true);
172 channel_->start();
173 if (use_external_context_.load() && manage_external_context_.load() && !external_thread_.joinable()) {
174 if (external_ioc_ && external_ioc_->stopped()) {
175 external_ioc_->restart();
176 }
177 work_guard_ = std::make_unique<boost::asio::executor_work_guard<boost::asio::io_context::executor_type>>(
178 external_ioc_->get_executor());
179 external_thread_ = std::thread([this, ioc = external_ioc_]() {
180 try {
181 while (started_.load() && !ioc->stopped()) {
182 if (ioc->run_one_for(std::chrono::milliseconds(50)) == 0) {
183 std::this_thread::yield();
184 }
185 }
186 } catch (...) {
187 }
188 });
189 }
190 return f;
191 }
192
193 void stop() {
194 bool should_join = false;
195 {
196 std::unique_lock<std::shared_mutex> lock(mutex_);
197 if (!started_.load()) {
198 fulfill_all_locked(false);
199 return;
200 }
201 started_.store(false);
202 if (batch_timer_) {
203 batch_timer_->cancel();
204 batch_timer_.reset();
205 }
206 if (channel_) {
207 channel_->on_bytes(nullptr);
208 channel_->on_state(nullptr);
209 lock.unlock();
210 channel_->stop();
211 lock.lock();
212 }
213 if (use_external_context_.load() && manage_external_context_.load()) {
214 if (work_guard_) work_guard_.reset();
215 if (external_ioc_) external_ioc_->stop();
216 should_join = true;
217 }
218 fulfill_all_locked(false);
219 }
220 if (should_join && external_thread_.joinable()) {
221 try {
222 external_thread_.join();
223 } catch (...) {
224 }
225 }
226 std::unique_lock<std::shared_mutex> lock(mutex_);
227 channel_.reset();
228 if (framer_) framer_->reset();
229 }

246 void setup_internal_handlers() {
247 if (!channel_) return;
248
249 batch_timer_ = std::make_unique<boost::asio::steady_timer>(channel_->get_executor());
250
251 std::weak_ptr<bool> weak_alive = alive_marker_;
252
253 channel_->on_bytes([this, weak_alive](memory::ConstByteSpan data) {
254 auto alive = weak_alive.lock();
255 if (!alive) return;
256
257 std::unique_lock<std::shared_mutex> lock(mutex_);
258 if (data_batch_handler_) {
259 data_batch_queue_.emplace_back(0, memory::SafeDataBuffer(data));
260 if (data_batch_queue_.size() >= max_batch_size_) {
261 auto handler = data_batch_handler_;
262 auto batch = std::move(data_batch_queue_);
263 data_batch_queue_.clear();
264 lock.unlock();
265 handler(batch);
266 } else if (data_batch_queue_.size() == 1) {
267 schedule_batch_timer();
268 }
269 return;
270 }
271
272 MessageHandler handler = data_handler_;
273 if (handler) {
274 lock.unlock();
275 handler(MessageContext(0, memory::SafeDataBuffer(data)));
276 lock.lock();
277 }
278
279 if (framer_) {
280 framer_->push_bytes(data);
281 }
282 });
283
284 channel_->on_state([this, weak_alive](base::LinkState state) {
285 auto alive = weak_alive.lock();
286 if (!alive) return;
287 ConnectionHandler connect_handler;
288 ConnectionHandler disconnect_handler;
289 ErrorHandler error_handler;
290 std::shared_ptr<interface::Channel> channel_snapshot;
291
292 if (state == base::LinkState::Connected) {
293 {
294 std::unique_lock<std::shared_mutex> lock(mutex_);
295 fulfill_all_locked(true);
296 connect_handler = connect_handler_;
297 }
298 if (connect_handler) connect_handler(ConnectionContext(0));
299 } else if (state == base::LinkState::Closed || state == base::LinkState::Error) {
300 {
301 std::unique_lock<std::shared_mutex> lock(mutex_);
302 fulfill_all_locked(false);
303 if (state == base::LinkState::Closed) {
304 disconnect_handler = disconnect_handler_;
305 } else {
306 error_handler = error_handler_;
307 channel_snapshot = channel_;
308 }
309 }
310 if (state == base::LinkState::Closed && disconnect_handler) {
311 disconnect_handler(ConnectionContext(0));
312 } else if (state == base::LinkState::Error && error_handler) {
313 bool handled = false;
314 if (auto transport = std::dynamic_pointer_cast<transport::TcpClient>(channel_snapshot)) {
315 if (auto info = transport->last_error_info()) {
316 error_handler(diagnostics::to_error_context(*info));
317 handled = true;
318 }
319 }
320 if (!handled) {
321 error_handler(ErrorContext(ErrorCode::IoError, "Connection error"));
322 }
323 }
324 }
325 });
326 }
327
328 void attach_framer_callback() {
329 if (!framer_) return;
330 framer_->on_message([this](memory::ConstByteSpan msg) {
331 std::unique_lock<std::shared_mutex> lock(mutex_);
332 if (message_batch_handler_) {
333 message_batch_queue_.emplace_back(0, memory::SafeDataBuffer(msg));
334 if (message_batch_queue_.size() >= max_batch_size_) {
335 auto handler = message_batch_handler_;
336 auto batch = std::move(message_batch_queue_);
337 message_batch_queue_.clear();
338 lock.unlock();
339 handler(batch);
340 } else if (message_batch_queue_.size() == 1) {
341 schedule_batch_timer();
342 }
343 return;
344 }
345
346 MessageHandler handler = message_handler_;
347 if (handler) {
348 lock.unlock();
349 handler(MessageContext(0, memory::SafeDataBuffer(msg)));
350 }
351 });
352 }
