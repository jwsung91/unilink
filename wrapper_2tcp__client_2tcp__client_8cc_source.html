87  void fulfill_all_locked(bool value) {
88  for (auto& p : pending_promises_) {
89  try {
90  p.set_value(value);
91  } catch (...) {
92  }
93  }
94  pending_promises_.clear();

97  std::future<bool> start() {
98  std::lock_guard<std::mutex> lock(mutex_);
99  if (channel_ && channel_->is_connected()) {
100  std::promise<bool> p;
101  p.set_value(true);
102  return p.get_future();
103  }
104  std::promise<bool> p;
105  auto f = p.get_future();
106  pending_promises_.push_back(std::move(p));
107  if (started_) return f;
108 
109  if (!channel_) {
110  config::TcpClientConfig config;
111  config.host = host_;
112  config.port = port_;
113  config.retry_interval_ms = static_cast<unsigned int>(retry_interval_.count());
114  config.max_retries = max_retries_;
115  config.connection_timeout_ms = static_cast<unsigned>(connection_timeout_.count());
116  channel_ = factory::ChannelFactory::create(config, external_ioc_);
117  setup_internal_handlers();
118  }
119  started_ = true;
120  channel_->start();
121  if (use_external_context_ && manage_external_context_ && !external_thread_.joinable()) {
122  if (external_ioc_ && external_ioc_->stopped()) {
123  external_ioc_->restart();
124  }
125  work_guard_ = std::make_unique<boost::asio::executor_work_guard<boost::asio::io_context::executor_type>>(
126  external_ioc_->get_executor());
127  external_thread_ = std::thread([this, ioc = external_ioc_]() {
128  try {
129  while (started_.load() && !ioc->stopped()) {
130  if (ioc->run_one_for(std::chrono::milliseconds(50)) == 0) {
131  std::this_thread::yield();
132  }
133  }
134  } catch (...) {
135  }
136  });
137  }
138  return f;
139  }

141  void stop() {
142  bool should_join = false;
143  {
144  std::unique_lock<std::mutex> lock(mutex_);
145  if (!started_.load()) {
146  fulfill_all_locked(false);
147  return;
148  }
149  started_ = false;
150  if (channel_) {
151  channel_->on_bytes(nullptr);
152  channel_->on_state(nullptr);
153  // RELEASE LOCK before channel_->stop(): stop() may trigger callbacks
154  // (e.g. on_state -> fulfill_all_locked) that try to acquire mutex_.
155  lock.unlock();
156  channel_->stop();
157  lock.lock();
158  }
159  if (use_external_context_ && manage_external_context_) {
160  if (work_guard_) work_guard_.reset();
161  if (external_ioc_) external_ioc_->stop();
162  should_join = true;
163  }
164  fulfill_all_locked(false);
165  }
166  if (should_join && external_thread_.joinable()) {
167  try {
168  external_thread_.join();
169  } catch (...) {
170  }
171  }
172  std::lock_guard<std::mutex> lock(mutex_);
173  channel_.reset();
174  if (framer_) framer_->reset();
175  }

186  void setup_internal_handlers() {
187  if (!channel_) return;
188 
189  std::weak_ptr<bool> weak_alive = alive_marker_;
190 
191  // Explicitly do not use try-catch here to allow exceptions from handlers
192  // to propagate to transport layer for error handling (e.g., auto-reconnect)
193  channel_->on_bytes([this, weak_alive](memory::ConstByteSpan data) {
194  auto alive = weak_alive.lock();
195  if (!alive) return;
196 
197  // 1. Raw data handler
198  MessageHandler data_handler;
199  {
200  std::lock_guard<std::mutex> lock(mutex_);
201  data_handler = data_handler_;
202  }
203  if (data_handler) {
204  std::string str_data = common::safe_convert::uint8_to_string(data.data(), data.size());
205  data_handler(MessageContext(0, str_data));
206  }
207 
208  // 2. Framer integration
209  std::lock_guard<std::mutex> lock(mutex_);
210  if (framer_) {
211  framer_->push_bytes(data);
212  }
213  });
214 
215  channel_->on_state([this, weak_alive](base::LinkState state) {
216  auto alive = weak_alive.lock();
217  if (!alive) return;
218  ConnectionHandler connect_handler;
219  ConnectionHandler disconnect_handler;
220  ErrorHandler error_handler;
221  std::shared_ptr<interface::Channel> channel_snapshot;
222 
223  if (state == base::LinkState::Connected) {
224  {
225  std::lock_guard<std::mutex> lock(mutex_);
226  fulfill_all_locked(true);
227  connect_handler = connect_handler_;
228  }
229  if (connect_handler) connect_handler(ConnectionContext(0));
230  } else if (state == base::LinkState::Closed || state == base::LinkState::Error) {
231  {
232  std::lock_guard<std::mutex> lock(mutex_);
233  fulfill_all_locked(false);
234  if (state == base::LinkState::Closed) {
235  disconnect_handler = disconnect_handler_;
236  } else {
237  error_handler = error_handler_;
238  channel_snapshot = channel_;
239  }
240  }
241  if (state == base::LinkState::Closed && disconnect_handler) {
242  disconnect_handler(ConnectionContext(0));
243  } else if (state == base::LinkState::Error && error_handler) {
244  bool handled = false;
245  if (auto transport = std::dynamic_pointer_cast<transport::TcpClient>(channel_snapshot)) {
246  if (auto info = transport->last_error_info()) {
247  error_handler(diagnostics::to_error_context(*info));
248  handled = true;
249  }
250  }
251  if (!handled) {
252  error_handler(ErrorContext(ErrorCode::IoError, "Connection error"));
253  }
254  }
255  }
256  });
257  }
258 
259  // Attach the stored message_handler_ to framer_->set_on_message().
260  // Must be called with mutex_ already held.
261  void attach_framer_callback() {
262  if (!framer_) return;
263  framer_->set_on_message([this](memory::ConstByteSpan msg) {
264  MessageHandler handler;
265  {
266  std::lock_guard<std::mutex> lock(mutex_);
267  handler = message_handler_;
268  }
269  if (handler) {
270  handler(MessageContext(0, common::safe_convert::uint8_to_string(msg.data(), msg.size())));
271  }
272  });
273  }
