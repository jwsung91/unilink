58 void TcpClient::start() {
59  if (started_) return;
60 
61  if (use_external_context_) {
62  if (!external_ioc_) {
63  throw std::runtime_error("External io_context is not set");
64  }
65  if (manage_external_context_) {
66  if (external_ioc_->stopped()) {
67  external_ioc_->restart();
68  }
69  }
70  }
71 
72  if (!channel_) {
73  // Create Channel
74  config::TcpClientConfig config;
75  config.host = host_;
76  config.port = port_;
77  config.retry_interval_ms = static_cast<unsigned int>(retry_interval_.count());
78  config.max_retries = max_retries_;
79  config.connection_timeout_ms = static_cast<unsigned>(connection_timeout_.count());
80  channel_ = factory::ChannelFactory::create(config, external_ioc_);
81  setup_internal_handlers();
82  }
83 
84  channel_->start();
85  if (use_external_context_ && manage_external_context_ && !external_thread_.joinable()) {
86  work_guard_.emplace(external_ioc_->get_executor());
87  external_thread_ = std::thread([ioc = external_ioc_]() { ioc->run(); });
88  }
89  started_ = true;
90 }

92 void TcpClient::stop() {
93  if (!started_ || !channel_) return;
94 
95  channel_->stop();
96 
97  if (use_external_context_ && manage_external_context_) {
98  if (work_guard_) {
99  work_guard_.reset();
100  }
101  if (external_thread_.joinable()) {
102  external_thread_.join();
103  }
104  }
105 
106  channel_.reset();
107  started_ = false;
108 }

179 void TcpClient::setup_internal_handlers() {
180  if (!channel_) return;
181 
182  // Convert byte data to string and pass it
183  channel_->on_bytes([this](const uint8_t* p, size_t n) {
184  if (bytes_handler_) {
185  bytes_handler_(p, n);
186  }
187  if (data_handler_) {
188  std::string data = common::safe_convert::uint8_to_string(p, n);
189  data_handler_(data);
190  }
191  });
192 
193  // Handle state changes
194  channel_->on_state([this](base::LinkState state) { notify_state_change(state); });
195 }

197 void TcpClient::notify_state_change(base::LinkState state) {
198  switch (state) {
199  case base::LinkState::Connected:
200  if (connect_handler_) connect_handler_();
201  break;
202  case base::LinkState::Closed:
203  if (disconnect_handler_) disconnect_handler_();
204  break;
205  case base::LinkState::Error:
206  if (error_handler_) error_handler_("Connection error occurred");
207  break;
208  default:
209  break;
210  }
211 }
