38MemoryPool::MemoryPool(size_t initial_pool_size, size_t max_pool_size) {
39 // Suppress unused parameter warning since we reserve based on max_pool_size
40 (void)initial_pool_size;
41
42 // Initialize 4 fixed-size pools
43 static constexpr std::array<size_t, 4> BUCKET_SIZES = {
44 static_cast<size_t>(BufferSize::SMALL), // 1KB
45 static_cast<size_t>(BufferSize::MEDIUM), // 4KB
46 static_cast<size_t>(BufferSize::LARGE), // 16KB
47 static_cast<size_t>(BufferSize::XLARGE) // 64KB
48 };
49
50 for (size_t i = 0; i < buckets_.size(); ++i) {
51 buckets_[i].size_ = BUCKET_SIZES[i];
52 buckets_[i].capacity_ = max_pool_size / buckets_.size();
53 buckets_[i].buffers_.reserve(buckets_[i].capacity_);
54 }
55}

57std::unique_ptr<uint8_t[]> MemoryPool::acquire(size_t size) {
58 validate_size(size);
59
60 // Optimization: Bypass pool for large allocations > 64KB (XLARGE)
61 // This avoids mutex contention and fixes truncation bug for sizes > 64KB.
62 if (size > static_cast<size_t>(BufferSize::XLARGE)) {
63 auto buffer = create_buffer(size);
64 // create_buffer throws on failure, so if we are here, we succeeded.
65 total_allocations_.fetch_add(1, std::memory_order_relaxed);
66 return buffer;
67 }
68
69 auto& bkt = bucket(size);
70 return acquire_from_bucket(bkt);
71}

77void MemoryPool::release(std::unique_ptr<uint8_t[]> buffer, size_t size) {
78 if (!buffer) return;
79
80 validate_size(size);
81
82 if (size > static_cast<size_t>(BufferSize::XLARGE)) {
83 MEMORY_TRACK_DEALLOCATION(buffer.get());
84 return; // unique_ptr destructor handles cleanup
85 }
86
87 auto& bkt = bucket(size);
88 release_to_bucket(bkt, std::move(buffer));
89}

98double MemoryPool::hit_rate() const {
99 size_t total = total_allocations_.load(std::memory_order_relaxed);
100 if (total == 0) return 0.0;
101
102 size_t hits = pool_hits_.load(std::memory_order_relaxed);
103 return static_cast<double>(hits) / static_cast<double>(total);
104}

139size_t MemoryPool::bucket_index(size_t size) const {
140 // Optimized: Unrolled checks for faster lookup
141 if (size <= static_cast<size_t>(BufferSize::SMALL)) return 0;
142 if (size <= static_cast<size_t>(BufferSize::MEDIUM)) return 1;
143 if (size <= static_cast<size_t>(BufferSize::LARGE)) return 2;
144 return 3; // XLARGE
145}

147std::unique_ptr<uint8_t[]> MemoryPool::acquire_from_bucket(PoolBucket& bucket) {
148 std::unique_ptr<uint8_t[]> buffer;
149
150 {
151 std::lock_guard<std::mutex> lock(bucket.mutex_);
152
153 // Get from stack
154 if (!bucket.buffers_.empty()) {
155 buffer = std::move(bucket.buffers_.back());
156 bucket.buffers_.pop_back();
157 }
158 }
159
160 if (buffer) {
161 pool_hits_.fetch_add(1, std::memory_order_relaxed);
162 total_allocations_.fetch_add(1, std::memory_order_relaxed);
163 return buffer;
164 }
165
166 // Create new buffer outside lock
167 buffer = create_buffer(bucket.size_);
168 if (buffer) {
169 total_allocations_.fetch_add(1, std::memory_order_relaxed);
170 }
171
172 return buffer;
173}

175void MemoryPool::release_to_bucket(PoolBucket& bucket, std::unique_ptr<uint8_t[]> buffer) {
176 {
177 std::lock_guard<std::mutex> lock(bucket.mutex_);
178
179 // Add buffer back to pool (stack)
180 if (bucket.buffers_.size() < bucket.capacity_) {
181 bucket.buffers_.push_back(std::move(buffer));
182 return;
183 }
184 }
185
186 // If pool is full, discard buffer (auto-release)
187 // buffer is unique_ptr so it will be automatically deleted when out of scope
188 MEMORY_TRACK_DEALLOCATION(buffer.get());
189}

191std::unique_ptr<uint8_t[]> MemoryPool::create_buffer(size_t size) {
192 uint8_t* raw_buffer = new (std::nothrow) uint8_t[size];
193 if (raw_buffer) {
194 MEMORY_TRACK_ALLOCATION(raw_buffer, size);
195 } else {
196 throw std::bad_alloc(); // Or handle error appropriately
197 }
198 return std::unique_ptr<uint8_t[]>(raw_buffer);
199}
