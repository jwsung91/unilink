32 MemoryPool::MemoryPool(size_t initial_pool_size, size_t max_pool_size) {
33  // Suppress unused parameter warning since we reserve based on max_pool_size
34  (void)initial_pool_size;
35 
36  // Initialize 4 fixed-size pools
37  static constexpr std::array<size_t, 4> BUCKET_SIZES = {
38  static_cast<size_t>(BufferSize::SMALL), // 1KB
39  static_cast<size_t>(BufferSize::MEDIUM), // 4KB
40  static_cast<size_t>(BufferSize::LARGE), // 16KB
41  static_cast<size_t>(BufferSize::XLARGE) // 64KB
42  };
43 
44  for (size_t i = 0; i < buckets_.size(); ++i) {
45  buckets_[i].size_ = BUCKET_SIZES[i];
46  buckets_[i].capacity_ = max_pool_size / buckets_.size();
47  buckets_[i].buffers_.reserve(buckets_[i].capacity_);
48  }
49 }

119 size_t MemoryPool::get_bucket_index(size_t size) const {
120  static constexpr std::array<size_t, 4> BUCKET_SIZES = {
121  static_cast<size_t>(BufferSize::SMALL), // 1KB
122  static_cast<size_t>(BufferSize::MEDIUM), // 4KB
123  static_cast<size_t>(BufferSize::LARGE), // 16KB
124  static_cast<size_t>(BufferSize::XLARGE) // 64KB
125  };
126 
127  for (size_t i = 0; i < BUCKET_SIZES.size(); ++i) {
128  if (size <= BUCKET_SIZES[i]) {
129  return i;
130  }
131  }
132  return BUCKET_SIZES.size() - 1; // use largest size
133 }

135 std::unique_ptr<uint8_t[]> MemoryPool::acquire_from_bucket(PoolBucket& bucket) {
136  std::unique_ptr<uint8_t[]> buffer;
137 
138  {
139  std::lock_guard<std::mutex> lock(bucket.mutex_);
140 
141  // Get from stack
142  if (!bucket.buffers_.empty()) {
143  buffer = std::move(bucket.buffers_.back());
144  bucket.buffers_.pop_back();
145  }
146  }
147 
148  if (buffer) {
149  pool_hits_.fetch_add(1, std::memory_order_relaxed);
150  total_allocations_.fetch_add(1, std::memory_order_relaxed);
151  return buffer;
152  }
153 
154  // Create new buffer outside lock
155  buffer = create_buffer(bucket.size_);
156  if (buffer) {
157  total_allocations_.fetch_add(1, std::memory_order_relaxed);
158  }
159 
160  return buffer;
161 }

163 void MemoryPool::release_to_bucket(PoolBucket& bucket, std::unique_ptr<uint8_t[]> buffer) {
164  {
165  std::lock_guard<std::mutex> lock(bucket.mutex_);
166 
167  // Add buffer back to pool (stack)
168  if (bucket.buffers_.size() < bucket.capacity_) {
169  bucket.buffers_.push_back(std::move(buffer));
170  return;
171  }
172  }
173 
174  // If pool is full, discard buffer (auto-release)
175  // buffer is unique_ptr so it will be automatically deleted when out of scope
176  MEMORY_TRACK_DEALLOCATION(buffer.get());
177 }
