195 template <typename StateType>
196 bool ThreadSafeState<StateType>::compare_and_set(const State& expected, const State& desired) {
197  std::unique_lock<std::shared_mutex> lock(state_mutex_);
198  if (state_ == expected) {
199  state_ = desired;
200  state_changed_.store(true);
201  lock.unlock();
202  notify_callbacks(desired);
203  state_cv_.notify_all();
204  return true;
205  }
206  return false;
207 }

209 template <typename StateType>
210 StateType ThreadSafeState<StateType>::exchange(const State& new_state) {
211  State old_state;
212  {
213  std::unique_lock<std::shared_mutex> lock(state_mutex_);
214  old_state = state_;
215  state_ = new_state;
216  state_changed_.store(true);
217  }
218  notify_callbacks(new_state);
219  state_cv_.notify_all();
220  return old_state;

265 template <typename StateType>
266 void ThreadSafeState<StateType>::notify_callbacks(const State& new_state) {
267  std::lock_guard<std::mutex> lock(callbacks_mutex_);
268  for (const auto& callback : callbacks_) {
269  try {
270  callback(new_state);
271  } catch (...) {
272  // Ignore callback exceptions to prevent state corruption
273  }
274  }
275 }
