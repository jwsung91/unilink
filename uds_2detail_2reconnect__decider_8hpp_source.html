53 inline UdsReconnectLogicDecision decide_reconnect_uds(const config::UdsClientConfig& cfg,
54  const diagnostics::ErrorInfo& error_info, uint32_t attempt_count,
55  const std::optional<ReconnectPolicy>& policy) {
56  if (!error_info.retryable) {
57  return {false, std::nullopt};
58  }
59 
60  if (cfg.max_retries == 0) {
61  return {false, std::nullopt};
62  }
63 
64  if (cfg.max_retries > 0 && attempt_count >= static_cast<uint32_t>(cfg.max_retries)) {
65  return {false, std::nullopt};
66  }
67 
68  if (policy) {
69  auto policy_decision = (*policy)(error_info, attempt_count);
70  if (!policy_decision.retry) {
71  return {false, std::nullopt};
72  }
73  return {true, clamp_reconnect_delay_uds(policy_decision.delay)};
74  }
75 
76  return {true, clamp_reconnect_delay_uds(std::chrono::milliseconds(cfg.retry_interval_ms))};
