32 void LineFramer::push_bytes(memory::ConstByteSpan data) {
33  if (data.empty()) return;
34 
35  // Append new data to buffer
36  buffer_.insert(buffer_.end(), data.begin(), data.end());
37 
38  size_t processed_offset = 0;
39 
40  while (processed_offset < buffer_.size()) {
41  auto it_start = buffer_.begin() + static_cast<std::ptrdiff_t>(processed_offset);
42  auto delimiter_pos = std::search(it_start, buffer_.end(), delimiter_.begin(), delimiter_.end());
43 
44  if (delimiter_pos == buffer_.end()) {
45  break;
46  }
47 
48  size_t msg_len = static_cast<size_t>(std::distance(it_start, delimiter_pos));
49  size_t total_len = msg_len + delimiter_.length();
50 
51  if (on_message_) {
52  size_t extract_len = include_delimiter_ ? total_len : msg_len;
53  if (extract_len > 0) {
54  // Address of element is safe as long as buffer is not reallocated
55  on_message_(memory::ConstByteSpan(&(*it_start), extract_len));
56  } else {
57  // Empty message (only if msg_len=0 and !include_delimiter_)
58  on_message_(memory::ConstByteSpan());
59  }
60  }
61 
62  // Check if reset was called during callback (buffer cleared)
63  if (buffer_.empty()) {
64  return;
65  }
66 
67  processed_offset += total_len;
68  }
69 
70  // Remove processed data
71  if (processed_offset > 0) {
72  if (processed_offset >= buffer_.size()) {
73  buffer_.clear();
74  } else {
75  buffer_.erase(buffer_.begin(), buffer_.begin() + static_cast<std::ptrdiff_t>(processed_offset));
76  }
77  }
78 
79  // Check max length on remaining buffer
80  if (buffer_.size() > max_length_) {
81  // Buffer overflow - clear buffer to prevent attack/leak
82  buffer_.clear();
83  }
84 }
