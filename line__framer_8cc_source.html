33 void LineFramer::push_bytes(memory::ConstByteSpan data) {
34  if (data.empty()) return;
35 
36  // Process data in chunks to prevent large memory allocations (DoS protection)
37  // We use max(max_length_, 4096) as a reasonable chunk size.
38  // This ensures that even if the user sends a huge payload, we only
39  // allocate memory incrementally and have a chance to clear the buffer
40  // if limits are exceeded.
41  const size_t chunk_limit = std::max(max_length_, size_t(4096));
42 
43  size_t offset = 0;
44  while (offset < data.size()) {
45  size_t len = std::min(data.size() - offset, chunk_limit);
46  push_bytes_internal(data.subspan(offset, len));
47  offset += len;
48  }
49 }

51 void LineFramer::push_bytes_internal(memory::ConstByteSpan data) {
52  if (data.empty()) return;
53 
54  // Fast Path: If buffer is empty, process data directly (zero-copy)
55  if (buffer_.empty()) {
56  size_t processed_count = scan_and_process(data, 0);
57 
58  // If we haven't processed everything, append the remainder to the buffer
59  if (processed_count < data.size()) {
60  buffer_.insert(buffer_.end(), data.begin() + processed_count, data.end());
61  scanned_idx_ = buffer_.size(); // We scanned all of it
62 
63  // DoS protection for partial message overflow
64  if (buffer_.size() > max_length_) {
65  buffer_.clear();
66  scanned_idx_ = 0;
67  }
68  } else {
69  // All processed, buffer remains empty
70  scanned_idx_ = 0;
71  }
72  return;
73  }
74 
75  // Slow Path: Append new data to buffer and process
76  buffer_.insert(buffer_.end(), data.begin(), data.end());
77 
78  // Determine where to start searching to avoid re-scanning
79  // We back up by delimiter length - 1 to catch split delimiters
80  size_t search_start_idx = scanned_idx_;
81  if (search_start_idx >= delimiter_.length()) {
82  search_start_idx -= (delimiter_.length() - 1);
83  } else {
84  search_start_idx = 0;
85  }
86 
87  size_t processed_count = scan_and_process(memory::ConstByteSpan(buffer_), search_start_idx);
88 
89  // Batch erase all processed data to ensure O(N) erase complexity
90  if (processed_count > 0) {
91  buffer_.erase(buffer_.begin(), buffer_.begin() + static_cast<std::ptrdiff_t>(processed_count));
92  }
93 
94  // Update scanned_idx_ for the next call.
95  // The buffer size is now reduced. We have scanned everything that remains.
96  scanned_idx_ = buffer_.size();
97 
98  // Final check: if the *remaining* partial message in the buffer already exceeds max_length_,
99  // we must reset to prevent unbound growth (DoS protection).
100  if (buffer_.size() > max_length_) {
101  buffer_.clear();
102  scanned_idx_ = 0;
103  }
104 }

106 size_t LineFramer::scan_and_process(memory::ConstByteSpan data, size_t search_start_offset) {
107  // Safety clamp
108  if (search_start_offset > data.size()) {
109  search_start_offset = data.size();
110  }
111 
112  // processed_count tracks the number of bytes from the start of the buffer
113  // that have been either emitted as messages or skipped due to overflow.
114  size_t processed_count = 0;
115 
116  // Search cursor
117  size_t search_cursor = search_start_offset;
118 
119  // O(N) scan loop
120  while (true) {
121  if (search_cursor > data.size()) break;
122 
123  // Perform search using iterators derived from current buffer state
124  auto search_begin = data.begin() + static_cast<std::ptrdiff_t>(search_cursor);
125  decltype(data.begin()) it;
126 
127  if (delimiter_.size() == 1) {
128  // Optimization: Use std::find for single-byte delimiter
129  it = std::find(search_begin, data.end(), static_cast<uint8_t>(delimiter_[0]));
130  } else {
131  it = std::search(search_begin, data.end(), delimiter_.begin(), delimiter_.end());
132  }
133 
134  if (it == data.end()) {
135  break;
136  }
137 
138  // Found a delimiter
139  size_t match_start_idx = static_cast<size_t>(std::distance(data.begin(), it));
140  size_t match_end_idx = match_start_idx + delimiter_.length();
141 
142  // Calculate message length (from end of previous processed data to end of current delimiter)
143  size_t current_msg_total_len = match_end_idx - processed_count;
144 
145  // Check strict max_length constraint
146  if (current_msg_total_len > max_length_) {
147  // Message exceeds limit. Skip it.
148  } else {
149  // Valid message
150  if (on_message_) {
151  size_t payload_len = include_delimiter_ ? current_msg_total_len : (current_msg_total_len - delimiter_.length());
152  on_message_(memory::ConstByteSpan(data.data() + processed_count, payload_len));
153  }
154  }
155 
156  // Mark these bytes as processed
157  processed_count = match_end_idx;
158 
159  // Advance search cursor to start strictly after the current delimiter
160  search_cursor = processed_count;
161  }
162 
163  return processed_count;
164 }
