35 std::unique_ptr<wrapper::UdsClient> UdsClientBuilder::build() {

37 
38  std::unique_ptr<wrapper::UdsClient> client;
39  if (independent_context_) {
40  auto ioc = std::make_shared<boost::asio::io_context>();
41  client = std::make_unique<wrapper::UdsClient>(socket_path_, ioc);
42  client->manage_external_context(true);
43  } else {
44  client = std::make_unique<wrapper::UdsClient>(socket_path_);
45  }
46 
47  if (on_data_) client->on_data(on_data_);
48  if (on_connect_) client->on_connect(on_connect_);
49  if (on_disconnect_) client->on_disconnect(on_disconnect_);
50  if (on_error_) client->on_error(on_error_);
51 
52  client->retry_interval(retry_interval_);
53  client->max_retries(max_retries_);
54  client->connection_timeout(connection_timeout_);
55 
56  if (framer_factory_) {
57  client->framer(framer_factory_());
58  }
59  if (on_message_) {
60  client->on_message(std::move(on_message_));
61  }
62 
63  if (auto_manage_) {
64  client->auto_manage(true);
65  }
66 
67  return client;
