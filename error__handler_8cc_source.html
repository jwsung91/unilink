39 void ErrorHandler::report_error(const ErrorInfo& error) {
40  if (!enabled_.load()) {
41  return;
42  }
43 
44  if (error.level < min_level_.load()) {
45  return;
46  }
47 
48  std::vector<ErrorCallback> callbacks_copy;
49  {
50  std::lock_guard<std::mutex> lock(mutex_);
51  update_stats(error);
52  add_to_recent_errors(error);
53  add_to_component_errors(error);
54  callbacks_copy = callbacks_;
55  }
56  notify_callbacks(callbacks_copy, error);
57 }

114 size_t ErrorHandler::get_error_count(const std::string& component, ErrorLevel level) const {
115  std::lock_guard<std::mutex> lock(mutex_);
116  auto it = errors_by_component_.find(component);
117  if (it == errors_by_component_.end()) {
118  return 0;
119  }
120 
121  return static_cast<size_t>(std::count_if(it->second.begin(), it->second.end(),
122  [level](const ErrorInfo& error) { return error.level == level; }));
123 }

125 void ErrorHandler::update_stats(const ErrorInfo& error) {
126  std::lock_guard<std::mutex> lock(stats_mutex_);
127 
128  stats_.total_errors++;
129  stats_.errors_by_level[static_cast<int>(error.level)]++;
130  stats_.errors_by_category[static_cast<int>(error.category)]++;
131 
132  if (error.retryable) {
133  stats_.retryable_errors++;
134  }
135 
136  if (stats_.first_error == std::chrono::system_clock::time_point{}) {
137  stats_.first_error = error.timestamp;
138  }
139  stats_.last_error = error.timestamp;
140 }

142 void ErrorHandler::notify_callbacks(const std::vector<ErrorCallback>& callbacks, const ErrorInfo& error) {
143  for (const auto& callback : callbacks) {
144  try {
145  callback(error);
146  } catch (const std::exception& e) {
147  // Avoid infinite recursion - use Logger directly instead of ErrorHandler
148  UNILINK_LOG_ERROR("error_handler", "callback", "Error in error callback: " + std::string(e.what()));
149  } catch (...) {
150  UNILINK_LOG_ERROR("error_handler", "callback", "Unknown error in error callback");
151  }
152  }
153 }

155 void ErrorHandler::add_to_recent_errors(const ErrorInfo& error) {
156  recent_errors_.push_back(error);
157 
158  // Keep only the most recent errors
159  if (recent_errors_.size() > MAX_RECENT_ERRORS) {
160  recent_errors_.erase(
161  recent_errors_.begin(),
162  recent_errors_.begin() + static_cast<std::ptrdiff_t>(recent_errors_.size() - MAX_RECENT_ERRORS));
163  }
164 }

166 void ErrorHandler::add_to_component_errors(const ErrorInfo& error) {
167  errors_by_component_[error.component].push_back(error);
168 
169  // Limit component error history to prevent memory growth
170  constexpr size_t MAX_COMPONENT_ERRORS = 100;
171  auto& component_errors = errors_by_component_[error.component];
172  if (component_errors.size() > MAX_COMPONENT_ERRORS) {
173  component_errors.erase(
174  component_errors.begin(),
175  component_errors.begin() + static_cast<std::ptrdiff_t>(component_errors.size() - MAX_COMPONENT_ERRORS));
176  }
177 }
