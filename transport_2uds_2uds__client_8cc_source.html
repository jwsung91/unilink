186 void UdsClient::start() {
187  auto current_state = impl_->state_.get_state();
188  if (current_state == LinkState::Connecting || current_state == LinkState::Connected) {
189  return;
190  }
191 
192  impl_->stop_requested_ = false;
193  impl_->stopping_ = false;
194  impl_->current_seq_++;
195  uint64_t seq = impl_->current_seq_.load();
196 
197  if (impl_->owns_ioc_ && !impl_->ioc_thread_.joinable()) {
198  if (impl_->ioc_->stopped()) {
199  impl_->ioc_->restart();
200  }
201  impl_->work_guard_ =
202  std::make_unique<net::executor_work_guard<net::io_context::executor_type>>(net::make_work_guard(*impl_->ioc_));
203  impl_->ioc_thread_ = std::thread([this]() {
204  try {
205  impl_->ioc_->run();
206  } catch (...) {
207  }
208  });
209  }
210 
211  net::post(impl_->strand_, [self = shared_from_this(), seq]() {
212  self->impl_->transition_to(LinkState::Connecting);
213  self->impl_->do_connect(self, seq);
214  });
215 }

217 void UdsClient::stop() {
218  bool already_stopping = impl_->stopping_.exchange(true);
219  if (already_stopping) return;
220 
221  impl_->stop_requested_ = true;
222 
223  // Immediate cancellation of operations
224  impl_->retry_timer_.cancel();
225  impl_->connect_timer_.cancel();
226  impl_->close_socket();
227 
228  // Release work guard and allow io_context to run out of work
229  if (impl_->ioc_) {
230  if (impl_->owns_ioc_) {
231  impl_->work_guard_.reset();
232  }
233  }
234 
235  // Transition to Idle state (lock-free or safe call)
236  impl_->state_.set_state(LinkState::Idle);
237 }

246 void UdsClient::async_write_move(std::vector<uint8_t>&& data) {
247  net::post(impl_->strand_, [this, self = shared_from_this(), data = std::move(data)]() mutable {
248  size_t added = data.size();
249  if (impl_->queue_bytes_ + added > impl_->bp_limit_) {
250  return;

254  impl_->report_backpressure(impl_->queue_bytes_);

289 void UdsClient::Impl::do_connect(std::shared_ptr<UdsClient> self, uint64_t seq) {
290  if (!cfg_.is_valid()) {
291  self->impl_->record_error(diagnostics::ErrorLevel::ERROR, diagnostics::ErrorCategory::CONFIGURATION, "connect",
292  make_error_code(boost::system::errc::invalid_argument), "Invalid UDS socket path", false,
293  self->impl_->reconnect_attempt_count_);
294  self->impl_->transition_to(LinkState::Error);
295  return;
296  }
297 
298  std::shared_ptr<uds::endpoint> endpoint;
299  try {
300  endpoint = std::make_shared<uds::endpoint>(cfg_.socket_path);
301  } catch (const std::exception& e) {
302  self->impl_->record_error(diagnostics::ErrorLevel::ERROR, diagnostics::ErrorCategory::CONFIGURATION, "connect",
303  make_error_code(boost::system::errc::filename_too_long),
304  "Invalid UDS endpoint: " + std::string(e.what()), false,
305  self->impl_->reconnect_attempt_count_);
306  self->impl_->transition_to(LinkState::Error);
307  return;
308  }
309 
310  connect_timer_.expires_after(std::chrono::milliseconds(cfg_.connection_timeout_ms));
311  connect_timer_.async_wait(net::bind_executor(strand_, [self, seq](const boost::system::error_code& ec) {
312  if (ec == net::error::operation_aborted || seq != self->impl_->current_seq_.load()) return;
313  if (!ec) {
314  self->impl_->record_error(diagnostics::ErrorLevel::ERROR, diagnostics::ErrorCategory::CONNECTION, "connect",
315  boost::asio::error::timed_out, "Connection timed out", true,
316  self->impl_->reconnect_attempt_count_);
317  self->impl_->handle_close(self, seq, boost::asio::error::timed_out);
318  }
319  }));
320 
321  socket_->async_connect(*endpoint, net::bind_executor(strand_, [self, seq,
322  endpoint](const boost::system::error_code& ec) {
323  self->impl_->connect_timer_.cancel();
324  if (ec == net::error::operation_aborted || seq != self->impl_->current_seq_.load()) return;
325  if (ec) {
326  self->impl_->record_error(diagnostics::ErrorLevel::ERROR, diagnostics::ErrorCategory::CONNECTION, "connect", ec,
327  "Connect failed: " + ec.message(), diagnostics::is_retryable_uds_connect_error(ec),
328  self->impl_->reconnect_attempt_count_);
329  self->impl_->schedule_retry(self, seq);
330  return;
331  }
332 
333  self->impl_->connected_ = true;
334  self->impl_->reconnect_attempt_count_ = 0;
335  self->impl_->retry_attempts_ = 0;
336  self->impl_->transition_to(LinkState::Connected);
337  self->impl_->start_read(self, seq);
338  if (!self->impl_->tx_.empty() && !self->impl_->writing_) {
339  self->impl_->do_write(self, seq);
340  }
341  }));
342 }

344 void UdsClient::Impl::schedule_retry(std::shared_ptr<UdsClient> self, uint64_t seq) {
345  transition_to(LinkState::Error);
346 
347  diagnostics::ErrorInfo dummy_err(diagnostics::ErrorLevel::ERROR, diagnostics::ErrorCategory::CONNECTION, "uds_client",
348  "connect", "Retry pending", boost::system::error_code(), true);
349  auto decision = detail::decide_reconnect_uds(cfg_, dummy_err, reconnect_attempt_count_, reconnect_policy_);
350 
351  if (!decision.should_retry || stop_requested_.load()) {
352  transition_to(LinkState::Idle);
353  return;
354  }
355 
356  reconnect_attempt_count_++;
357  retry_timer_.expires_after(decision.delay.value_or(std::chrono::milliseconds(cfg_.retry_interval_ms)));
358  retry_timer_.async_wait(net::bind_executor(strand_, [self, seq](const boost::system::error_code& ec) {
359  if (ec == net::error::operation_aborted || seq != self->impl_->current_seq_.load()) return;
360  self->impl_->do_connect(self, seq);
361  }));
362 }

364 void UdsClient::Impl::start_read(std::shared_ptr<UdsClient> self, uint64_t seq) {
365  socket_->async_read_some(net::buffer(rx_),
366  net::bind_executor(strand_, [self, seq](const boost::system::error_code& ec, size_t bytes) {
367  if (ec == net::error::operation_aborted || seq != self->impl_->current_seq_.load()) return;
368  if (ec) {
369  self->impl_->handle_close(self, seq, ec);
370  return;
371  }
372 
373  OnBytes cb;
374  {
375  std::lock_guard<std::mutex> lock(self->impl_->callback_mtx_);
376  cb = self->impl_->on_bytes_;
377  }
378  if (cb) cb(memory::ConstByteSpan(self->impl_->rx_.data(), bytes));
379  self->impl_->start_read(self, seq);
380  }));
381 }

383 void UdsClient::Impl::do_write(std::shared_ptr<UdsClient> self, uint64_t seq) {
384  if (tx_.empty() || writing_) return;
385  writing_ = true;
386  current_write_buffer_ = std::move(tx_.front());
387  tx_.pop_front();
388 
389  net::const_buffer buffer;
390  std::visit(
391  [&buffer](auto&& arg) {
392  using T = std::decay_t<decltype(arg)>;
393  if constexpr (std::is_same_v<T, std::vector<uint8_t>>)
394  buffer = net::buffer(arg);
395  else if constexpr (std::is_same_v<T, std::shared_ptr<const std::vector<uint8_t>>>)
396  buffer = net::buffer(*arg);
397  else if constexpr (std::is_same_v<T, memory::PooledBuffer>)
398  buffer = net::buffer(arg.data(), arg.size());
399  },
400  *current_write_buffer_);
401 
402  size_t bytes_to_write = buffer.size();
403  socket_->async_write(
404  buffer, net::bind_executor(strand_, [self, seq, bytes_to_write](const boost::system::error_code& ec, size_t) {
405  if (ec == net::error::operation_aborted || seq != self->impl_->current_seq_.load()) return;
406  self->impl_->writing_ = false;
407  self->impl_->current_write_buffer_ = std::nullopt;
408  self->impl_->queue_bytes_ -= bytes_to_write;
409  self->impl_->report_backpressure(self->impl_->queue_bytes_);
410 
411  if (ec) {
412  self->impl_->handle_close(self, seq, ec);
413  return;
414  }
415  if (!self->impl_->tx_.empty()) self->impl_->do_write(self, seq);
416  }));
417 }
