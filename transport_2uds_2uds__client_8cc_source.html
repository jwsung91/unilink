189void UdsClient::start() {
190 auto current_state = impl_->state_.state();
191 if (current_state == LinkState::Connecting || current_state == LinkState::Connected) {
192 return;
193 }
194
195 impl_->stop_requested_ = false;
196 impl_->stopping_ = false;
197 impl_->current_seq_++;
198 uint64_t seq = impl_->current_seq_.load();
199
200 if (impl_->owns_ioc_ && !impl_->ioc_thread_.joinable()) {
201 if (impl_->ioc_->stopped()) {
202 impl_->ioc_->restart();
203 }
204 impl_->work_guard_ =
205 std::make_unique<net::executor_work_guard<net::io_context::executor_type>>(net::make_work_guard(*impl_->ioc_));
206 impl_->ioc_thread_ = std::thread([ioc = impl_->owned_ioc_]() {
207 try {
208 ioc->run();
209 } catch (...) {
210 }
211 });
212 }
213
214 net::post(impl_->strand_, [self = shared_from_this(), seq]() {
215 self->impl_->transition_to(LinkState::Connecting);
216 self->impl_->do_connect(self, seq);
217 });
218}

220void UdsClient::stop() {
221 bool already_stopping = impl_->stopping_.exchange(true);
222 if (already_stopping) return;
223
224 impl_->stop_requested_ = true;
225
226 // Immediate cancellation of operations
227 impl_->retry_timer_.cancel();
228 impl_->connect_timer_.cancel();
229 impl_->close_socket();
230
231 // Release work guard and allow io_context to run out of work
232 if (impl_->ioc_) {
233 if (impl_->owns_ioc_) {
234 impl_->work_guard_.reset();
235 }
236 }
237
238 // Transition to Idle state (lock-free or safe call)
239 impl_->state_.set_state(LinkState::Idle);
240}

251void UdsClient::async_write_move(std::vector<uint8_t>&& data) {
252 net::post(impl_->strand_, [this, self = shared_from_this(), data = std::move(data)]() mutable {
253 size_t added = data.size();
254 impl_->maybe_flush_for_keep_latest(added);
255 if (impl_->queue_bytes_ + added > impl_->bp_limit_) {
256 UNILINK_LOG_ERROR("uds_client", "write",
257 "Queue limit exceeded (" + std::to_string(impl_->queue_bytes_ + added) + " bytes)");
258 impl_->report_backpressure(impl_->queue_bytes_ + added);
259 return;
260 }
261 impl_->queue_bytes_ += added;
262 impl_->tx_.emplace_back(std::move(data));
263 impl_->report_backpressure(impl_->queue_bytes_);
264 if (!impl_->writing_) {
265 impl_->do_write(self, impl_->current_seq_.load());
266 }
267 });
268}

316void UdsClient::Impl::do_connect(std::shared_ptr<UdsClient> self, uint64_t seq) {
317 if (!cfg_.is_valid()) {
318 self->impl_->record_error(diagnostics::ErrorLevel::ERROR, diagnostics::ErrorCategory::CONFIGURATION, "connect",
319 make_error_code(boost::system::errc::invalid_argument), "Invalid UDS socket path", false,
320 self->impl_->reconnect_attempt_count_);
321 self->impl_->transition_to(LinkState::Error);
322 return;
323 }
324
325 std::shared_ptr<uds::endpoint> endpoint;
326 try {
327 endpoint = std::make_shared<uds::endpoint>(cfg_.socket_path);
328 } catch (const std::exception& e) {
329 self->impl_->record_error(diagnostics::ErrorLevel::ERROR, diagnostics::ErrorCategory::CONFIGURATION, "connect",
330 make_error_code(boost::system::errc::filename_too_long),
331 "Invalid UDS endpoint: " + std::string(e.what()), false,
332 self->impl_->reconnect_attempt_count_);
333 self->impl_->transition_to(LinkState::Error);
334 return;
335 }
336
337 connect_timer_.expires_after(std::chrono::milliseconds(cfg_.connection_timeout_ms));
338 connect_timer_.async_wait(net::bind_executor(strand_, [self, seq](const boost::system::error_code& ec) {
339 if (ec == net::error::operation_aborted || seq != self->impl_->current_seq_.load()) return;
340 if (!ec) {
341 self->impl_->record_error(diagnostics::ErrorLevel::ERROR, diagnostics::ErrorCategory::CONNECTION, "connect",
342 boost::asio::error::timed_out, "Connection timed out", true,
343 self->impl_->reconnect_attempt_count_);
344 self->impl_->handle_close(self, seq, boost::asio::error::timed_out);
345 }
346 }));
347
348 socket_->async_connect(*endpoint, net::bind_executor(strand_, [self, seq,
349 endpoint](const boost::system::error_code& ec) {
350 self->impl_->connect_timer_.cancel();
351 if (ec == net::error::operation_aborted || seq != self->impl_->current_seq_.load()) return;
352 if (ec) {
353 self->impl_->record_error(diagnostics::ErrorLevel::ERROR, diagnostics::ErrorCategory::CONNECTION, "connect", ec,
354 "Connect failed: " + ec.message(), diagnostics::is_retryable_uds_connect_error(ec),
355 self->impl_->reconnect_attempt_count_);
356 self->impl_->schedule_retry(self, seq);
357 return;
358 }
359
360 self->impl_->connected_ = true;
361 self->impl_->reconnect_attempt_count_ = 0;
362 self->impl_->retry_attempts_ = 0;
363 self->impl_->transition_to(LinkState::Connected);
364 self->impl_->start_read(self, seq);
365 if (!self->impl_->tx_.empty() && !self->impl_->writing_) {
366 self->impl_->do_write(self, seq);
367 }
368 }));
369}

371void UdsClient::Impl::schedule_retry(std::shared_ptr<UdsClient> self, uint64_t seq) {
372 transition_to(LinkState::Error);
373
374 diagnostics::ErrorInfo dummy_err(diagnostics::ErrorLevel::ERROR, diagnostics::ErrorCategory::CONNECTION, "uds_client",
375 "connect", "Retry pending", boost::system::error_code(), true);
376 auto decision = detail::decide_reconnect_uds(cfg_, dummy_err, reconnect_attempt_count_, reconnect_policy_);
377
378 if (!decision.should_retry || stop_requested_.load()) {
379 transition_to(LinkState::Idle);
380 return;
381 }
382
383 reconnect_attempt_count_++;
384 retry_timer_.expires_after(decision.delay.value_or(std::chrono::milliseconds(cfg_.retry_interval_ms)));
385 retry_timer_.async_wait(net::bind_executor(strand_, [self, seq](const boost::system::error_code& ec) {
386 if (ec == net::error::operation_aborted || seq != self->impl_->current_seq_.load()) return;
387 self->impl_->do_connect(self, seq);
388 }));
389}

391void UdsClient::Impl::start_read(std::shared_ptr<UdsClient> self, uint64_t seq) {
392 socket_->async_read_some(net::buffer(rx_),
393 net::bind_executor(strand_, [self, seq](const boost::system::error_code& ec, size_t bytes) {
394 if (ec == net::error::operation_aborted || seq != self->impl_->current_seq_.load()) return;
395 if (ec) {
396 self->impl_->handle_close(self, seq, ec);
397 return;
398 }
399
400 OnBytes cb;
401 {
402 std::lock_guard<std::mutex> lock(self->impl_->callback_mtx_);
403 cb = self->impl_->on_bytes_;
404 }
405 if (cb) cb(memory::ConstByteSpan(self->impl_->rx_.data(), bytes));
406 self->impl_->start_read(self, seq);
407 }));
408}

410void UdsClient::Impl::do_write(std::shared_ptr<UdsClient> self, uint64_t seq) {
411 if (tx_.empty() || writing_) return;
412 writing_ = true;
413 current_write_buffer_ = std::move(tx_.front());
414 tx_.pop_front();
415
416 net::const_buffer buffer;
417 std::visit(
418 [&buffer](auto&& arg) {
419 using T = std::decay_t<decltype(arg)>;
420 if constexpr (std::is_same_v<T, std::vector<uint8_t>>)
421 buffer = net::buffer(arg);
422 else if constexpr (std::is_same_v<T, std::shared_ptr<const std::vector<uint8_t>>>)
423 buffer = net::buffer(*arg);
424 else if constexpr (std::is_same_v<T, memory::PooledBuffer>)
425 buffer = net::buffer(arg.data(), arg.size());
426 },
427 *current_write_buffer_);
428
429 size_t bytes_to_write = buffer.size();
430 socket_->async_write(
431 buffer, net::bind_executor(strand_, [self, seq, bytes_to_write](const boost::system::error_code& ec, size_t) {
432 if (ec == net::error::operation_aborted || seq != self->impl_->current_seq_.load()) return;
433 self->impl_->writing_ = false;
434 self->impl_->current_write_buffer_ = std::nullopt;
435 self->impl_->queue_bytes_ -= bytes_to_write;
436 self->impl_->report_backpressure(self->impl_->queue_bytes_);
437
438 if (ec) {
439 self->impl_->handle_close(self, seq, ec);
440 return;
441 }
442 if (!self->impl_->tx_.empty()) self->impl_->do_write(self, seq);
443 }));
444}

446void UdsClient::Impl::handle_close(std::shared_ptr<UdsClient> self, uint64_t seq, const boost::system::error_code&) {
447 connected_ = false;
448 close_socket();
449 retry_timer_.cancel();
450 connect_timer_.cancel();
451
452 if (stop_requested_) {
453 transition_to(LinkState::Idle);
454 } else {
455 schedule_retry(self, seq);
456 }
