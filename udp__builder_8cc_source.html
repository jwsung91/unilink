33std::unique_ptr<wrapper::UdpClient> UdpClientBuilder::build() {
34 std::shared_ptr<boost::asio::io_context> ioc = nullptr;
35 if (independent_context_) {
36 ioc = std::make_shared<boost::asio::io_context>();
37 }
38
39 auto udp = std::make_unique<wrapper::UdpClient>(cfg_, ioc);
40 if (independent_context_) {
41 udp->manage_external_context(true);
42 }
43
44 if (on_data_) udp->on_data(on_data_);
45 if (on_connect_) udp->on_connect(on_connect_);
46 if (on_disconnect_) udp->on_disconnect(on_disconnect_);
47 if (on_error_) udp->on_error(on_error_);
48
49 if (framer_factory_) {
50 udp->framer(framer_factory_());
51 }
52 if (on_message_) {
53 udp->on_message(std::move(on_message_));
54 }
55
56 if (auto_start_) {
57 udp->auto_start(true);
58 }
59
60 return udp;
61}
