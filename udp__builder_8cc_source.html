33std::unique_ptr<wrapper::UdpClient> UdpClientBuilder::build() {
34 std::shared_ptr<boost::asio::io_context> ioc = nullptr;
35 if (independent_context_) {
36 ioc = std::make_shared<boost::asio::io_context>();
37 }
38
39 if (bp_strategy_set_) cfg_.backpressure_strategy = bp_strategy_;
40 cfg_.backpressure_threshold = get_effective_backpressure_threshold();
41
42 auto udp = std::make_unique<wrapper::UdpClient>(cfg_, ioc);
43 if (independent_context_) {
44 udp->manage_external_context(true);
45 }
46
47 if (on_data_) udp->on_data(on_data_);
48 if (on_connect_) udp->on_connect(on_connect_);
49 if (on_disconnect_) udp->on_disconnect(on_disconnect_);
50 if (on_error_) udp->on_error(on_error_);
51
52 if (framer_factory_) {
53 udp->framer(framer_factory_());
54 }
55 if (on_message_) {
56 udp->on_message(std::move(on_message_));
57 }
58
59 if (auto_start_) {
60 udp->auto_start(true);
61 }
62
63 return udp;
64}
