31 std::unique_ptr<wrapper::Udp> UdpClientBuilder::build() {
32  std::shared_ptr<boost::asio::io_context> ioc = nullptr;
33  if (use_independent_context_) {
34  ioc = std::make_shared<boost::asio::io_context>();
35  }
36 
37  auto udp = std::make_unique<wrapper::Udp>(cfg_, ioc);
38  if (use_independent_context_) {
39  udp->set_manage_external_context(true);
40  }
41 
42  if (on_data_) udp->on_data(on_data_);
43  if (on_connect_) udp->on_connect(on_connect_);
44  if (on_disconnect_) udp->on_disconnect(on_disconnect_);
45  if (on_error_) udp->on_error(on_error_);
46 
47  if (framer_factory_) {
48  udp->set_framer(framer_factory_());
49  }
50  if (on_message_) {
51  udp->on_message(std::move(on_message_));
52  }
53 
54  if (auto_manage_) {
55  udp->auto_manage(true);
56  }
57 
58  return udp;
59 }
