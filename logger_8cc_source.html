91  void parse_format(const std::string& format) {
92  auto new_format = std::make_shared<LogFormat>();
93  new_format->format_string = format;
94 
95  size_t start = 0;
96  size_t pos = 0;
97 
98  while ((pos = format.find('{', start)) != std::string::npos) {
99  if (pos > start) {
100  new_format->parsed_format.push_back({FormatPart::LITERAL, format.substr(start, pos - start)});
101  }
102 
103  size_t end = format.find('}', pos);
104  if (end == std::string::npos) {
105  new_format->parsed_format.push_back({FormatPart::LITERAL, format.substr(pos)});
106  start = format.length();
107  break;
108  }
109 
110  std::string placeholder = format.substr(pos + 1, end - pos - 1);
111  if (placeholder == "timestamp") {
112  new_format->parsed_format.push_back({FormatPart::TIMESTAMP, ""});
113  } else if (placeholder == "level") {
114  new_format->parsed_format.push_back({FormatPart::LEVEL, ""});
115  } else if (placeholder == "component") {
116  new_format->parsed_format.push_back({FormatPart::COMPONENT, ""});
117  } else if (placeholder == "operation") {
118  new_format->parsed_format.push_back({FormatPart::OPERATION, ""});
119  } else if (placeholder == "message") {
120  new_format->parsed_format.push_back({FormatPart::MESSAGE, ""});
121  } else {
122  new_format->parsed_format.push_back({FormatPart::LITERAL, format.substr(pos, end - pos + 1)});
123  }
124 
125  start = end + 1;
126  }
127 
128  if (start < format.length()) {
129  new_format->parsed_format.push_back({FormatPart::LITERAL, format.substr(start)});
130  }
131 
132  std::lock_guard<std::mutex> lock(mutex_);
133  log_format_ = std::move(new_format);
134  }

145  std::string format_message(std::chrono::system_clock::time_point timestamp_val, LogLevel level,
146  std::string_view component, std::string_view operation, std::string_view message) {
147  TimestampBuffer timestamp = get_timestamp(timestamp_val);
148  std::string_view level_str = level_to_string(level);
149 
150  std::shared_ptr<LogFormat> current_format;
151  {
152  std::lock_guard<std::mutex> lock(mutex_);
153  current_format = log_format_;
154  }
155 
156  std::string result;
157  if (current_format) {
158  result.reserve(current_format->format_string.length() + message.length() + 32);
159 
160  for (const auto& part : current_format->parsed_format) {
161  switch (part.type) {
162  case FormatPart::LITERAL:
163  result.append(part.value);
164  break;
165  case FormatPart::TIMESTAMP:
166  result.append(timestamp.view());
167  break;
168  case FormatPart::LEVEL:
169  result.append(level_str);
170  break;
171  case FormatPart::COMPONENT:
172  result.append(component);
173  break;
174  case FormatPart::OPERATION:
175  result.append(operation);
176  break;
177  case FormatPart::MESSAGE:
178  result.append(message);
179  break;
180  }
181  }
182  }
183 
184  return result;
185  }

187  void write_to_sinks(LogLevel level, const std::string& formatted_message) {
188  std::lock_guard<std::mutex> lock(mutex_);
189  int current_outputs = outputs_.load();
190 
191  if (current_outputs & static_cast<int>(LogOutput::CONSOLE)) {
192  write_to_console(formatted_message);
193  }
194 
195  if (current_outputs & static_cast<int>(LogOutput::FILE)) {
196  check_and_rotate_log();
197  write_to_file(formatted_message);
198  }
199 
200  if (current_outputs & static_cast<int>(LogOutput::CALLBACK)) {
201  call_callback(level, formatted_message);
202  }
203  }

221  TimestampBuffer get_timestamp(std::chrono::system_clock::time_point timestamp) const {
222  auto time_t = std::chrono::system_clock::to_time_t(timestamp);
223  auto ms = std::chrono::duration_cast<std::chrono::milliseconds>(timestamp.time_since_epoch()) % 1000;
224 
225  static thread_local std::time_t last_t = -1;
226  static thread_local char date_buf[32] = {0};
227 
228  if (time_t != last_t) {
229  std::tm time_info{};
230 #if defined(_WIN32)
231  ::localtime_s(&time_info, &time_t);
232 #else
233  ::localtime_r(&time_t, &time_info);
234 #endif
235  std::strftime(date_buf, sizeof(date_buf), "%Y-%m-%d %H:%M:%S", &time_info);
236  last_t = time_t;
237  }
238 
239  TimestampBuffer result;
240  int len = std::snprintf(result.data, sizeof(result.data), "%s.%03d", date_buf, static_cast<int>(ms.count()));
241  if (len > 0) {
242  if (static_cast<size_t>(len) >= sizeof(result.data)) {
243  result.length = sizeof(result.data) - 1;
244  } else {
245  result.length = static_cast<size_t>(len);
246  }
247  } else {
248  result.length = 0;
249  }
250  return result;
251  }

283  void check_and_rotate_log() {
284  if (!log_rotation_ || current_log_file_.empty()) {
285  return;
286  }
287 
288  bool should_rotate = false;
289  if (file_output_ && file_output_->is_open()) {
290  std::streampos current_pos = file_output_->tellp();
291  if (current_pos != static_cast<std::streampos>(-1) &&
292  static_cast<size_t>(current_pos) >= log_rotation_->get_config().max_file_size_bytes) {
293  should_rotate = true;
294  }
295  } else {
296  if (log_rotation_->should_rotate(current_log_file_)) {
297  should_rotate = true;
298  }
299  }
300 
301  if (!should_rotate) {
302  return;
303  }
304 
305  if (file_output_) {
306  file_output_->flush();
307  file_output_->close();
308  file_output_.reset();
309  }
310 
311  log_rotation_->rotate(current_log_file_);
312  open_log_file(current_log_file_);
313  }

340  void teardown_async_logging() {
341  std::thread worker_to_join;
342  bool notify_worker = false;
343  {
344  std::lock_guard<std::mutex> lock(mutex_);
345  async_enabled_.store(false);
346 
347  if (running_.load()) {
348  shutdown_requested_.store(true);
349  notify_worker = true;
350  }
351 
352  if (worker_thread_.joinable()) {
353  worker_to_join = std::move(worker_thread_);
354  }
355  }
356 
357  if (notify_worker) {
358  queue_cv_.notify_all();
359  }
360 
361  if (worker_to_join.joinable()) {
362  worker_to_join.join();
363  }
364 
365  running_.store(false);
366  shutdown_requested_.store(false);
367  }

369  void worker_loop() {
370  std::vector<LogEntry> batch;
371  batch.reserve(async_config_.batch_size);
372 
373  auto last_flush = std::chrono::steady_clock::now();
374 
375  while (true) {
376  std::unique_lock<std::mutex> lock(queue_mutex_);
377 
378  bool has_logs = queue_cv_.wait_for(lock, async_config_.flush_interval,
379  [this] { return !log_queue_.empty() || shutdown_requested_.load(); });
380 
381  if (has_logs && !log_queue_.empty()) {
382  size_t batch_size =
383  async_config_.enable_batch_processing ? std::min(async_config_.batch_size, log_queue_.size()) : 1;
384 
385  batch.clear();
386  batch.reserve(batch_size);
387 
388  for (size_t i = 0; i < batch_size && !log_queue_.empty(); ++i) {
389  batch.push_back(std::move(log_queue_.front()));
390  log_queue_.pop();
391  }
392 
393  lock.unlock();
394 
395  if (!batch.empty()) {
396  process_batch(batch);
397  update_stats_on_batch(batch.size());
398  }
399  } else {
400  lock.unlock();
401  }
402 
403  auto now = std::chrono::steady_clock::now();
404  if (now - last_flush >= async_config_.flush_interval) {
405  flush();
406  update_stats_on_flush();
407  last_flush = now;
408  }
409 
410  if (shutdown_requested_.load() && log_queue_.empty()) {
411  break;
412  }
413  }
414 
415  flush();
416  running_.store(false);
417  }

424  }

551 void Logger::log(LogLevel level, std::string_view component, std::string_view operation, std::string_view message) {
552  if (!get_impl()->enabled_.load() || level < get_impl()->current_level_.load()) {
553  return;
554  }
555 
556  if (get_impl()->async_enabled_.load()) {
557  LogEntry entry(level, component, operation, message);
558  impl_->update_stats_on_enqueue();
559 
560  if (get_impl()->async_config_.enable_backpressure && impl_->should_drop_log()) {
561  impl_->update_stats_on_drop();
562  return;
563  }
564 
565  {
566  std::lock_guard<std::mutex> lock(impl_->queue_mutex_);
567  impl_->log_queue_.push(entry);
568  }
569 
570  impl_->queue_cv_.notify_one();
571  return;
572  }
573 
574  std::string formatted_message =
575  impl_->format_message(std::chrono::system_clock::now(), level, component, operation, message);
576  if (get_impl()->outputs_.load(std::memory_order_relaxed) != 0) {
577  impl_->write_to_sinks(level, formatted_message);
578  }
579 }
