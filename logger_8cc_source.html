45template <typename Mutex>
46class callback_sink : public spdlog::sinks::base_sink<Mutex> {
47 public:
48 explicit callback_sink(Logger::LogCallback callback) : callback_(std::move(callback)) {}
49
50 void set_callback(Logger::LogCallback callback) {
51 std::lock_guard<Mutex> lock(spdlog::sinks::base_sink<Mutex>::mutex_);
52 callback_ = std::move(callback);
53 }
54
55 protected:
56 void sink_it_(const spdlog::details::log_msg& msg) override {
57 if (!callback_) return;
58
59 spdlog::memory_buf_t formatted;
60 spdlog::sinks::base_sink<Mutex>::formatter_->format(msg, formatted);
61 try {
62 callback_(from_spdlog_level(msg.level), fmt::to_string(formatted));
63 } catch (...) {
64 }
65 }
66
67 void flush_() override {}
68
69 private:
70 static LogLevel from_spdlog_level(spdlog::level::level_enum level) {
71 switch (level) {
72 case spdlog::level::debug:
73 return LogLevel::DEBUG;
74 case spdlog::level::info:
75 return LogLevel::INFO;
76 case spdlog::level::warn:
77 return LogLevel::WARNING;
78 case spdlog::level::err:
79 return LogLevel::ERROR;
80 case spdlog::level::critical:
81 return LogLevel::CRITICAL;
82 default:
83 return LogLevel::INFO;
84 }
85 }
86
87 Logger::LogCallback callback_;
88};

92class level_range_sink final : public spdlog::sinks::sink {
93 public:
94 level_range_sink(std::shared_ptr<spdlog::sinks::sink> sink, spdlog::level::level_enum min_level,
95 spdlog::level::level_enum max_level)
96 : sink_(std::move(sink)), min_level_(min_level), max_level_(max_level) {}
97
98 void log(const spdlog::details::log_msg& msg) override {
99 if (msg.level >= min_level_ && msg.level <= max_level_) {
100 sink_->log(msg);
101 }
102 }

105
106 void set_pattern(const std::string& pattern) override { sink_->set_pattern(pattern); }

108 void set_formatter(std::unique_ptr<spdlog::formatter> sink_formatter) override {
109 sink_->set_formatter(std::move(sink_formatter));
110 }

112 private:

114 spdlog::level::level_enum min_level_;

210 static void replace_all(std::string& str, const std::string& from, const std::string& to) {
211 if (from.empty()) {
212 return;
213 }
214
215 size_t pos = 0;
216 while ((pos = str.find(from, pos)) != std::string::npos) {
217 str.replace(pos, from.length(), to);
218 pos += to.length();
219 }
220 }

239 static std::string normalize_env_value(std::string_view value) {
240 std::string normalized(value);
241 normalized.erase(normalized.begin(), std::find_if(normalized.begin(), normalized.end(),
242 [](unsigned char c) { return !std::isspace(c); }));
243 normalized.erase(
244 std::find_if(normalized.rbegin(), normalized.rend(), [](unsigned char c) { return !std::isspace(c); }).base(),
245 normalized.end());
246 std::transform(normalized.begin(), normalized.end(), normalized.begin(),
247 [](unsigned char c) { return static_cast<char>(std::toupper(c)); });
248 return normalized;
249 }

251 static bool parse_log_level(std::string_view value, LogLevel& level, bool& disable_logging) {
252 const auto normalized = normalize_env_value(value);
253 disable_logging = false;
254 if (normalized == "DEBUG" || normalized == "TRACE") {
255 level = LogLevel::DEBUG;
256 return true;
257 }
258 if (normalized == "INFO") {
259 level = LogLevel::INFO;
260 return true;
261 }
262 if (normalized == "WARNING" || normalized == "WARN") {
263 level = LogLevel::WARNING;
264 return true;
265 }
266 if (normalized == "ERROR" || normalized == "ERR") {
267 level = LogLevel::ERROR;
268 return true;
269 }
270 if (normalized == "CRITICAL" || normalized == "FATAL") {
271 level = LogLevel::CRITICAL;
272 return true;
273 }
274 if (normalized == "OFF" || normalized == "NONE" || normalized == "DISABLED") {
275 disable_logging = true;
276 return true;
277 }
278 return false;
279 }

281 void apply_environment_settings() {
282 const char* env_level = std::getenv("UNILINK_LOG_LEVEL");
283 if (!env_level || *env_level == '\0') {
284 return;
285 }
286
287 LogLevel parsed_level = current_level_.load();
288 bool disable_logging = false;
289 if (!parse_log_level(env_level, parsed_level, disable_logging)) {
290 set_error("Invalid UNILINK_LOG_LEVEL: " + std::string(env_level));
291 return;
292 }
293
294 enabled_.store(!disable_logging);
295 if (!disable_logging) {
296 current_level_.store(parsed_level);
297 if (spd_logger_) {
298 spd_logger_->set_level(to_spdlog_level(parsed_level));
299 }
300 }
301 clear_error();
302 }

304 static std::string timestamp_now() {
305 using namespace std::chrono;
306 const auto now = system_clock::now();
307 const auto tt = system_clock::to_time_t(now);
308 std::tm tm{};
309#if defined(_WIN32)
310 localtime_s(&tm, &tt);
311#else
312 localtime_r(&tt, &tm);
313#endif
314 const auto ms = duration_cast<milliseconds>(now.time_since_epoch()) % 1000;
315
316 std::ostringstream oss;
317 oss << std::put_time(&tm, "%F %T") << '.' << std::setw(3) << std::setfill('0') << ms.count();
318 return oss.str();
319 }

321 static std::string format_message(const std::string& format, LogLevel level, std::string_view component,
322 std::string_view operation, std::string_view message,
323 const std::source_location& loc) {
324 const auto file = std::string(loc.file_name());
325 const auto line = std::to_string(loc.line());
326 const auto function = std::string(loc.function_name());
327 const auto source = fmt::format("{}:{}:{}", file, line, function);
328
329 std::string formatted = format;
330 replace_all(formatted, "{timestamp}", timestamp_now());
331 replace_all(formatted, "{level}", level_name(level));
332 replace_all(formatted, "{component}", std::string(component));
333 replace_all(formatted, "{operation}", std::string(operation));
334 replace_all(formatted, "{source}", source);
335 replace_all(formatted, "{file}", file);
336 replace_all(formatted, "{line}", line);
337 replace_all(formatted, "{function}", function);
338 replace_all(formatted, "{message}", std::string(message));
339 return formatted;
340 }

349 void setup_async_logging(const AsyncLogConfig& config) {
350 async_config_ = config;
351
352 spdlog::drop("unilink"); // Drop existing logger
353
354 auto overflow_policy = config.enable_backpressure ? spdlog::async_overflow_policy::block
355 : spdlog::async_overflow_policy::overrun_oldest;
356
357 async_thread_pool_ = std::make_shared<spdlog::details::thread_pool>(std::max<size_t>(1, config.max_queue_size), 1);
358
359 spd_logger_ = std::make_shared<spdlog::async_logger>("unilink", dist_sink_, async_thread_pool_, overflow_policy);
360 spd_logger_->set_level(to_spdlog_level(current_level_.load()));
361 apply_spdlog_pattern();
362
363 spdlog::register_logger(spd_logger_);
364
365 if (config.flush_interval.count() > 0) {
366 // spdlog::flush_every takes std::chrono::seconds.
367 // Ensure at least 1 second if a positive interval is requested.
368 auto secs =
369 std::max(std::chrono::seconds(1), std::chrono::duration_cast<std::chrono::seconds>(config.flush_interval));
370 spdlog::flush_every(secs);
371 }
372
373 async_enabled_.store(true);
374 }
375
376 void teardown_async_logging() {
377 if (!async_enabled_.load()) return;
378
379 async_enabled_.store(false);
380
381 // If there's a previous drop still pending, wait for it now to avoid multiple drop tasks
382 if (drop_future_.valid()) {
383 drop_future_.wait();
384 }
385
386 // Use std::async to drop the logger with a timeout
387 drop_future_ = std::async(std::launch::async, []() { spdlog::drop("unilink"); });
388
389 bool drop_success = true;
390 if (drop_future_.wait_for(async_config_.shutdown_timeout) == std::future_status::timeout) {
391 drop_success = false;
392 }
393
394 spd_logger_.reset();
395 if (drop_success) {
396 async_thread_pool_.reset();
397 }
398 dist_sink_->flush();
399
400 // Create sync logger
401 spd_logger_ = std::make_shared<spdlog::logger>("unilink", dist_sink_);
402 spd_logger_->set_level(to_spdlog_level(current_level_.load()));
403 apply_spdlog_pattern();
404
405 // Only register globally if the previous drop finished, to avoid race conditions.
406 // If it didn't finish, we still have spd_logger_ internally so log() works.
407 if (drop_success) {
408 spdlog::register_logger(spd_logger_);
409 }
410 }

479bool Logger::try_set_file_output_with_rotation(const std::string& filename, const LogRotationConfig& config) {
480 std::lock_guard<std::mutex> lock(impl_->mutex_);
481
482 if (filename.empty()) {
483 if (impl_->file_sink_) {
484 impl_->dist_sink_->remove_sink(impl_->file_sink_);
485 impl_->file_sink_.reset();
486 }
487 impl_->current_log_file_.clear();
488 impl_->outputs_.fetch_and(~static_cast<int>(LogOutput::FILE));
489 impl_->clear_error();
490 return true;
491 } else {
492 try {
493 if (config.enable_compression) {
494 impl_->set_error("Log compression is not supported by the current rotating file sink");
495 return false;
496 }
497 if (config.file_pattern != LogRotationConfig{}.file_pattern) {
498 impl_->set_error("Custom log rotation file_pattern is not supported by the current rotating file sink");
499 return false;
500 }
501 if (impl_->file_sink_) {
502 impl_->dist_sink_->remove_sink(impl_->file_sink_);
503 }
504
505 impl_->rotation_config_ = config;
506 impl_->current_log_file_ = filename;
507
508 impl_->file_sink_ = std::make_shared<spdlog::sinks::rotating_file_sink_mt>(filename, config.max_file_size_bytes,
509 config.max_files);
510
511 impl_->dist_sink_->add_sink(impl_->file_sink_);
512 impl_->apply_spdlog_pattern();
513 impl_->outputs_.fetch_or(static_cast<int>(LogOutput::FILE));
514 impl_->clear_error();
515 return true;
516 } catch (const spdlog::spdlog_ex& e) {
517 impl_->set_error("Failed to open log file: " + filename + " (" + e.what() + ")");
518 std::cerr << impl_->last_error_ << std::endl;
519 impl_->file_sink_.reset();
520 impl_->current_log_file_.clear();
521 impl_->outputs_.fetch_and(~static_cast<int>(LogOutput::FILE));
522 return false;
523 }
524 }

572void Logger::set_outputs(int outputs) {
573 std::lock_guard<std::mutex> lock(impl_->mutex_);
574 int effective_outputs = 0;
575
576 // Reconcile sinks in dist_sink_ based on the new bitmask
577
578 // Console Sink
579 if (outputs & static_cast<int>(LogOutput::CONSOLE)) {
580 impl_->add_console_sinks();
581 effective_outputs |= static_cast<int>(LogOutput::CONSOLE);
582 } else {
583 impl_->remove_console_sinks();
584 }
585
586 // File Sink
587 if (outputs & static_cast<int>(LogOutput::FILE)) {
588 if (!impl_->file_sink_ && !impl_->current_log_file_.empty()) {
589 impl_->file_sink_ = std::make_shared<spdlog::sinks::rotating_file_sink_mt>(
590 impl_->current_log_file_, impl_->rotation_config_.max_file_size_bytes, impl_->rotation_config_.max_files);
591 }
592 if (impl_->file_sink_) {
593 impl_->dist_sink_->remove_sink(impl_->file_sink_); // Ensure no duplicate
594 impl_->dist_sink_->add_sink(impl_->file_sink_);
595 impl_->apply_spdlog_pattern();
596 effective_outputs |= static_cast<int>(LogOutput::FILE);
597 }
598 } else {
599 if (impl_->file_sink_) {
600 impl_->dist_sink_->remove_sink(impl_->file_sink_);
601 impl_->file_sink_.reset();
602 }
603 }
604
605 // Callback Sink
606 if (outputs & static_cast<int>(LogOutput::CALLBACK)) {
607 if (impl_->callback_sink_) {
608 impl_->dist_sink_->remove_sink(impl_->callback_sink_); // Ensure no duplicate
609 impl_->dist_sink_->add_sink(impl_->callback_sink_);
610 impl_->apply_spdlog_pattern();
611 effective_outputs |= static_cast<int>(LogOutput::CALLBACK);
612 }
613 } else {
614 if (impl_->callback_sink_) {
615 impl_->dist_sink_->remove_sink(impl_->callback_sink_);
616 }
617 }
618
619 impl_->outputs_.store(effective_outputs);
620 impl_->clear_error();
621}

652void Logger::log(LogLevel level, std::string_view component, std::string_view operation, std::string_view message,
653 const std::source_location& loc) {
654 if (!should_log(level)) {
655 return;
656 }
657
658 std::shared_ptr<spdlog::logger> logger;
659 std::string format;
660 {
661 std::lock_guard<std::mutex> lock(impl_->mutex_);
662 logger = impl_->spd_logger_;
663 format = impl_->format_;
664 }
665
666 if (!logger) {
667 return;
668 }
669
670 const auto payload = Impl::format_message(format, level, component, operation, message, loc);
671 logger->log(Impl::to_spdlog_level(level), payload);
672}
