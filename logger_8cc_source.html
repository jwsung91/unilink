37template <typename Mutex>
38class callback_sink : public spdlog::sinks::base_sink<Mutex> {
39 public:
40 explicit callback_sink(Logger::LogCallback callback) : callback_(std::move(callback)) {}
41
42 void set_callback(Logger::LogCallback callback) {
43 std::lock_guard<Mutex> lock(spdlog::sinks::base_sink<Mutex>::mutex_);
44 callback_ = std::move(callback);
45 }
46
47 protected:
48 void sink_it_(const spdlog::details::log_msg& msg) override {
49 if (!callback_) return;
50
51 spdlog::memory_buf_t formatted;
52 spdlog::sinks::base_sink<Mutex>::formatter_->format(msg, formatted);
53 callback_(from_spdlog_level(msg.level), fmt::to_string(formatted));
54 }
55
56 void flush_() override {}
57
58 private:
59 static LogLevel from_spdlog_level(spdlog::level::level_enum level) {
60 switch (level) {
61 case spdlog::level::debug:
62 return LogLevel::DEBUG;
63 case spdlog::level::info:
64 return LogLevel::INFO;
65 case spdlog::level::warn:
66 return LogLevel::WARNING;
67 case spdlog::level::err:
68 return LogLevel::ERROR;
69 case spdlog::level::critical:
70 return LogLevel::CRITICAL;
71 default:
72 return LogLevel::INFO;
73 }
74 }
75
76 Logger::LogCallback callback_;
77};

122 void parse_format(const std::string& format) {
123 std::string pattern = format;
124
125 // Map unilink placeholders to spdlog pattern
126 // {timestamp} -> %Y-%m-%d %H:%M:%S.%e
127 // {level} -> %l
128 // {message} -> %v (Note: macros wrap component/operation into message)
129
130 auto replace_all = [](std::string& str, const std::string& from, const std::string& to) {
131 size_t pos = 0;
132 while ((pos = str.find(from, pos)) != std::string::npos) {
133 str.replace(pos, from.length(), to);
134 pos += to.length();
135 }
136 };
137
138 replace_all(pattern, "{timestamp}", "%Y-%m-%d %H:%M:%S.%e");
139 replace_all(pattern, "{level}", "%l");
140 replace_all(pattern, "{message}", "%v");
141
142 // component and operation are currently embedded in the message by macros
143 // but if they appear in format, we strip them or handle as literal since they are in %v
144 replace_all(pattern, "[{component}] ", "");
145 replace_all(pattern, "[{operation}] ", "");
146
147 dist_sink_->set_pattern(pattern);
148 }

161 void setup_async_logging(const AsyncLogConfig& config) {
162 async_config_ = config;
163
164 spdlog::drop("unilink"); // Drop existing logger
165
166 auto overflow_policy = config.enable_backpressure ? spdlog::async_overflow_policy::block
167 : spdlog::async_overflow_policy::overrun_oldest;
168
169 if (!spdlog::thread_pool()) {
170 spdlog::init_thread_pool(config.max_queue_size, 1);
171 }
172
173 spd_logger_ = std::make_shared<spdlog::async_logger>("unilink", dist_sink_, spdlog::thread_pool(), overflow_policy);
174 spd_logger_->set_level(to_spdlog_level(current_level_.load()));
175
176 spdlog::register_logger(spd_logger_);
177
178 if (config.flush_interval.count() > 0) {
179 // spdlog::flush_every takes std::chrono::seconds.
180 // Ensure at least 1 second if a positive interval is requested.
181 auto secs =
182 std::max(std::chrono::seconds(1), std::chrono::duration_cast<std::chrono::seconds>(config.flush_interval));
183 spdlog::flush_every(secs);
184 }
185
186 async_enabled_.store(true);
187 }
188
189 void teardown_async_logging() {
190 if (!async_enabled_.load()) return;
191
192 async_enabled_.store(false);
193
194 // If there's a previous drop still pending, wait for it now to avoid multiple drop tasks
195 if (drop_future_.valid()) {
196 drop_future_.wait();
197 }
198
199 // Use std::async to drop the logger with a timeout
200 drop_future_ = std::async(std::launch::async, []() { spdlog::drop("unilink"); });
201
202 bool drop_success = true;
203 if (drop_future_.wait_for(async_config_.shutdown_timeout) == std::future_status::timeout) {
204 drop_success = false;
205 }
206
207 spd_logger_.reset();
208 dist_sink_->flush();
209
210 // Create sync logger
211 spd_logger_ = std::make_shared<spdlog::logger>("unilink", dist_sink_);
212 spd_logger_->set_level(to_spdlog_level(current_level_.load()));
213
214 // Only register globally if the previous drop finished, to avoid race conditions.
215 // If it didn't finish, we still have spd_logger_ internally so log() works.
216 if (drop_success) {
217 spdlog::register_logger(spd_logger_);
218 }
219 }

260void Logger::set_console_output(bool enable) {
261 std::lock_guard<std::mutex> lock(impl_->mutex_);
262 if (enable) {
263 if (!impl_->console_sink_) {
264 impl_->console_sink_ = std::make_shared<spdlog::sinks::stdout_color_sink_mt>();
265 }
266 impl_->dist_sink_->remove_sink(impl_->console_sink_); // Ensure no duplicate
267 impl_->dist_sink_->add_sink(impl_->console_sink_);
268 impl_->outputs_.fetch_or(static_cast<int>(LogOutput::CONSOLE));
269 } else {
270 if (impl_->console_sink_) {
271 impl_->dist_sink_->remove_sink(impl_->console_sink_);
272 impl_->console_sink_.reset();
273 }
274 impl_->outputs_.fetch_and(~static_cast<int>(LogOutput::CONSOLE));

280void Logger::set_file_output_with_rotation(const std::string& filename, const LogRotationConfig& config) {
281 std::lock_guard<std::mutex> lock(impl_->mutex_);
282
283 if (filename.empty()) {
284 if (impl_->file_sink_) {
285 impl_->dist_sink_->remove_sink(impl_->file_sink_);
286 impl_->file_sink_.reset();
287 }
288 impl_->current_log_file_.clear();
289 impl_->outputs_.fetch_and(~static_cast<int>(LogOutput::FILE));
290 } else {
291 try {
292 if (impl_->file_sink_) {
293 impl_->dist_sink_->remove_sink(impl_->file_sink_);
294 }
295
296 impl_->rotation_config_ = config;
297 impl_->current_log_file_ = filename;
298
299 impl_->file_sink_ = std::make_shared<spdlog::sinks::rotating_file_sink_mt>(filename, config.max_file_size_bytes,
300 config.max_files);
301
302 impl_->dist_sink_->add_sink(impl_->file_sink_);
303 impl_->outputs_.fetch_or(static_cast<int>(LogOutput::FILE));
304 } catch (const spdlog::spdlog_ex& e) {
305 std::cerr << "Failed to open log file: " << filename << " (" << e.what() << ")" << std::endl;
306 impl_->file_sink_.reset();
307 impl_->current_log_file_.clear();
308 impl_->outputs_.fetch_and(~static_cast<int>(LogOutput::FILE));
309 }
310 }

323void Logger::set_callback(LogCallback callback) {
324 std::lock_guard<std::mutex> lock(impl_->mutex_);
325
326 if (!callback) {
327 if (impl_->callback_sink_) {
328 impl_->dist_sink_->remove_sink(impl_->callback_sink_);
329 impl_->callback_sink_.reset();
330 }
331 impl_->outputs_.fetch_and(~static_cast<int>(LogOutput::CALLBACK));
332 return;
333 }
334
335 if (impl_->callback_sink_) {
336 impl_->callback_sink_->set_callback(std::move(callback));
337 // Ensure it's in the dist_sink if outputs bit is set
338 if (impl_->outputs_.load() & static_cast<int>(LogOutput::CALLBACK)) {
339 impl_->dist_sink_->remove_sink(impl_->callback_sink_); // Avoid duplicates
340 impl_->dist_sink_->add_sink(impl_->callback_sink_);
341 }
342 } else {
343 impl_->callback_sink_ = std::make_shared<callback_sink_mt>(std::move(callback));
344 impl_->dist_sink_->add_sink(impl_->callback_sink_);
345 }
346 impl_->outputs_.fetch_or(static_cast<int>(LogOutput::CALLBACK));
347}

349void Logger::set_outputs(int outputs) {
350 std::lock_guard<std::mutex> lock(impl_->mutex_);
351 impl_->outputs_.store(outputs);
352
353 // Reconcile sinks in dist_sink_ based on the new bitmask
354
355 // Console Sink
356 if (outputs & static_cast<int>(LogOutput::CONSOLE)) {
357 if (!impl_->console_sink_) {
358 impl_->console_sink_ = std::make_shared<spdlog::sinks::stdout_color_sink_mt>();
359 }
360 impl_->dist_sink_->remove_sink(impl_->console_sink_); // Ensure no duplicate
361 impl_->dist_sink_->add_sink(impl_->console_sink_);
362 } else {
363 if (impl_->console_sink_) {
364 impl_->dist_sink_->remove_sink(impl_->console_sink_);
365 impl_->console_sink_.reset();
366 }
367 }
368
369 // File Sink
370 if (outputs & static_cast<int>(LogOutput::FILE)) {
371 if (!impl_->file_sink_ && !impl_->current_log_file_.empty()) {
372 impl_->file_sink_ = std::make_shared<spdlog::sinks::rotating_file_sink_mt>(
373 impl_->current_log_file_, impl_->rotation_config_.max_file_size_bytes, impl_->rotation_config_.max_files);
374 }
375 if (impl_->file_sink_) {
376 impl_->dist_sink_->remove_sink(impl_->file_sink_); // Ensure no duplicate
377 impl_->dist_sink_->add_sink(impl_->file_sink_);
378 }
379 } else {
380 if (impl_->file_sink_) {
381 impl_->dist_sink_->remove_sink(impl_->file_sink_);
382 impl_->file_sink_.reset();
383 }
384 }
385
386 // Callback Sink
387 if (outputs & static_cast<int>(LogOutput::CALLBACK)) {
388 if (impl_->callback_sink_) {
389 impl_->dist_sink_->remove_sink(impl_->callback_sink_); // Ensure no duplicate
390 impl_->dist_sink_->add_sink(impl_->callback_sink_);
391 }
392 } else {
393 if (impl_->callback_sink_) {
394 impl_->dist_sink_->remove_sink(impl_->callback_sink_);
395 }
396 }
397}

407void Logger::log(LogLevel level, std::string_view component, std::string_view operation, std::string_view message,
408 const std::source_location& loc) {
409 if (!get_impl()->enabled_.load() || level < get_impl()->current_level_.load()) {
410 return;
411 }
412
413 // Incorporate source location information into the payload
414 std::string payload = fmt::format("[{}] [{}] [{}:{}:{}] {}", component, operation, loc.file_name(), loc.line(),
415 loc.function_name(), message);
416
417 impl_->spd_logger_->log(Impl::to_spdlog_level(level), payload);
418}
