38
39 bool is_valid() const {
40 if (backpressure_threshold < base::constants::MIN_BACKPRESSURE_THRESHOLD ||
41 backpressure_threshold > base::constants::MAX_BACKPRESSURE_THRESHOLD) {
42 return false;
43 }
44 if (remote_address.has_value() != remote_port.has_value()) return false;
45 if (remote_port && *remote_port == 0) return false;
46 return true;
47 }
48
49 void validate_and_clamp() {
50 if (backpressure_threshold < base::constants::MIN_BACKPRESSURE_THRESHOLD) {
51 backpressure_threshold = base::constants::MIN_BACKPRESSURE_THRESHOLD;
52 } else if (backpressure_threshold > base::constants::MAX_BACKPRESSURE_THRESHOLD) {
53 backpressure_threshold = base::constants::MAX_BACKPRESSURE_THRESHOLD;
54 }
55 }
56};
