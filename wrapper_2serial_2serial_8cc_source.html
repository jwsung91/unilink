81  std::future<bool> start() {
82  if (started) {
83  std::promise<bool> p;
84  p.set_value(true);
85  return p.get_future();
86  }
87 
88  start_promise_ = std::promise<bool>();
89  start_promise_fulfilled_ = false;
90 
91  if (!channel) {
92  channel = factory::ChannelFactory::create(build_config(), external_ioc);
93  setup_internal_handlers();
94  }
95 
96  channel->start();
97  if (use_external_context && manage_external_context && !external_thread.joinable()) {
98  external_thread = std::thread([ioc = external_ioc]() {
99  boost::asio::executor_work_guard<boost::asio::io_context::executor_type> guard(ioc->get_executor());
100  ioc->run();
101  });
102  }
103 
104  started = true;
105  return start_promise_.get_future();
106  }
107 
108  void stop() {
109  if (!started) return;
110 
111  if (channel) {
112  channel->on_bytes(nullptr);
113  channel->on_state(nullptr);
114  channel->stop();
115  }
116 
117  if (use_external_context && manage_external_context && external_thread.joinable()) {
118  if (external_ioc) external_ioc->stop();
119  external_thread.join();
120  }
121 
122  started = false;
123 
124  if (!start_promise_fulfilled_) {
125  try {
126  start_promise_.set_value(false);
127  } catch (...) {
128  }
129  start_promise_fulfilled_ = true;
130  }
131  }
132 
133  void setup_internal_handlers() {
134  if (!channel) return;
135 
136  channel->on_bytes([this](memory::ConstByteSpan data) {
137  if (data_handler) {
138  std::string str_data = common::safe_convert::uint8_to_string(data.data(), data.size());
139  data_handler(MessageContext(0, str_data));
140  }
141  });
142 
143  channel->on_state([this](base::LinkState state) {
144  switch (state) {
145  case base::LinkState::Connected:
146  if (!start_promise_fulfilled_) {
147  start_promise_.set_value(true);
148  start_promise_fulfilled_ = true;
149  }
150  if (connect_handler) connect_handler(ConnectionContext(0));
151  break;
152  case base::LinkState::Closed:
153  if (disconnect_handler) disconnect_handler(ConnectionContext(0));
154  break;
155  case base::LinkState::Error:
156  if (!start_promise_fulfilled_) {
157  start_promise_.set_value(false);
158  start_promise_fulfilled_ = true;
159  }
160  if (error_handler) error_handler(ErrorContext(ErrorCode::IoError, "Connection error"));
161  break;
162  default:
163  break;
164  }
165  });
166  }

168  config::SerialConfig build_config() const {
169  config::SerialConfig config;
170  config.device = device;
171  config.baud_rate = baud_rate;
172  config.char_size = static_cast<unsigned int>(data_bits);
173  config.stop_bits = static_cast<unsigned int>(stop_bits);
174  std::string p = to_lower(parity);
175  if (p == "even")
176  config.parity = config::SerialConfig::Parity::Even;
177  else if (p == "odd")
178  config.parity = config::SerialConfig::Parity::Odd;
179  else
180  config.parity = config::SerialConfig::Parity::None;
181 
182  std::string f = to_lower(flow_control);
183  if (f == "software")
184  config.flow = config::SerialConfig::Flow::Software;
185  else if (f == "hardware")
186  config.flow = config::SerialConfig::Flow::Hardware;
187  else
188  config.flow = config::SerialConfig::Flow::None;
189 
190  config.retry_interval_ms = static_cast<unsigned int>(retry_interval.count());
191  return config;
192  }
