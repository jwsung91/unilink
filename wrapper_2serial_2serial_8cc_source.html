87 void fulfill_all_locked(bool value) {
88 for (auto& promise : pending_promises_) {

97 std::future<bool> start() {
98 std::unique_lock<std::mutex> lock(mutex_);
99 if (channel && channel->is_connected()) {
100 std::promise<bool> p;
101 p.set_value(true);
102 return p.get_future();
103 }
104
105 std::promise<bool> p;
106 auto future = p.get_future();
107 pending_promises_.emplace_back(std::move(p));
108
109 if (started_.exchange(true)) {
110 return future;
111 }
112
113 if (!channel) {
114 channel = factory::ChannelFactory::create(build_config(), external_ioc);
115 setup_internal_handlers();
116 }
117
118 lock.unlock();
119 channel->start();
120 if (use_external_context && manage_external_context && !external_thread.joinable()) {
121 if (external_ioc && external_ioc->stopped()) {
122 external_ioc->restart();
123 }
124 work_guard_ = std::make_unique<boost::asio::executor_work_guard<boost::asio::io_context::executor_type>>(
125 boost::asio::make_work_guard(*external_ioc));
126 external_thread = std::thread([ioc = external_ioc]() {
127 try {
128 ioc->run();
129 } catch (...) {
130 }
131 });
132 }
133
134 return future;
135 }

137 void stop() {
138 std::unique_lock<std::mutex> lock(mutex_);
139 if (!started_.exchange(false)) {
140 fulfill_all_locked(false);
141 return;
142 }
143
144 if (channel) {
145 channel->on_bytes(nullptr);
146 channel->on_state(nullptr);
147 lock.unlock();
148 channel->stop();
149 lock.lock();
150 }
151
152 if (work_guard_) {
153 work_guard_.reset();
154 }
155
156 if (use_external_context && manage_external_context && external_thread.joinable()) {
157 if (external_ioc) external_ioc->stop();
158 if (std::this_thread::get_id() != external_thread.get_id()) {
159 lock.unlock();
160 external_thread.join();
161 lock.lock();
162 } else {
163 external_thread.detach();
164 }
165 }
166
167 fulfill_all_locked(false);
168
169 if (framer) framer->reset();
170 }
171
172 void setup_internal_handlers() {
173 if (!channel) return;
174
175 std::weak_ptr<bool> weak_alive = alive_marker_;
176
177 channel->on_bytes([this, weak_alive](memory::ConstByteSpan data) {
178 auto alive = weak_alive.lock();
179 if (!alive) return;
180
181 // 1. Raw data handler
182 MessageHandler handler;
183 {
184 std::lock_guard<std::mutex> lock(mutex_);
185 handler = data_handler;
186 }
187 if (handler) {
188 std::string str_data = base::safe_convert::uint8_to_string(data.data(), data.size());
189 handler(MessageContext(0, str_data));
190 }
191
192 // 2. Framer integration
193 std::lock_guard<std::mutex> lock(mutex_);
194 if (framer) {
195 framer->push_bytes(data);
196 }
197 });
198
199 channel->on_state([this, weak_alive](base::LinkState state) {
200 auto alive = weak_alive.lock();
201 if (!alive) return;
202
203 switch (state) {
204 case base::LinkState::Connected: {
205 ConnectionHandler handler;
206 {
207 std::lock_guard<std::mutex> lock(mutex_);
208 fulfill_all_locked(true);
209 handler = connect_handler;
210 }
211 if (handler) handler(ConnectionContext(0));
212 break;
213 }
214 case base::LinkState::Closed:
215 case base::LinkState::Error:
216 case base::LinkState::Idle: {
217 ConnectionHandler disconnect_handler_snapshot;
218 ErrorHandler error_handler_snapshot;
219 {
220 std::lock_guard<std::mutex> lock(mutex_);
221 fulfill_all_locked(false);
222 if (state == base::LinkState::Error) {
223 error_handler_snapshot = error_handler;
224 } else {
225 disconnect_handler_snapshot = disconnect_handler;
226 }
227 }
228 if (disconnect_handler_snapshot) {
229 disconnect_handler_snapshot(ConnectionContext(0));
230 }
231 if (error_handler_snapshot) {
232 error_handler_snapshot(ErrorContext(ErrorCode::IoError, "Connection error"));
233 }
234 break;
235 }
236 default:
237 break;
238 }
239 });
240 }
241
242 // Attach the stored message_handler to framer->on_message().
243 // Must be called with mutex_ already held.
244 void attach_framer_callback() {
245 if (!framer) return;
246 framer->on_message([this](memory::ConstByteSpan msg) {
247 MessageHandler handler;
248 {
249 std::lock_guard<std::mutex> lock(mutex_);
250 handler = message_handler;
251 }
252 if (handler) {
253 handler(MessageContext(0, base::safe_convert::uint8_to_string(msg.data(), msg.size())));
254 }
255 });
256 }
257
258 void set_framer(std::unique_ptr<framer::IFramer> f) {
259 std::lock_guard<std::mutex> lock(mutex_);
260 framer = std::move(f);
261 if (framer && message_handler) attach_framer_callback();
262 }
263
264 void on_message(MessageHandler handler) {
265 std::lock_guard<std::mutex> lock(mutex_);
266 message_handler = std::move(handler);
267 if (framer) attach_framer_callback();
268 }

270 config::SerialConfig build_config() const {
271 config::SerialConfig config;
272 config.device = device;
273 config.baud_rate = baud_rate;
274 config.char_size = static_cast<unsigned int>(data_bits);
275 config.stop_bits = static_cast<unsigned int>(stop_bits);
276 std::string p = to_lower(parity);
277 if (p == "even")
278 config.parity = config::SerialConfig::Parity::Even;
279 else if (p == "odd")
280 config.parity = config::SerialConfig::Parity::Odd;
281 else
282 config.parity = config::SerialConfig::Parity::None;
283
284 std::string f = to_lower(flow_control);
285 if (f == "software")
286 config.flow = config::SerialConfig::Flow::Software;
287 else if (f == "hardware")
288 config.flow = config::SerialConfig::Flow::Hardware;
289 else
290 config.flow = config::SerialConfig::Flow::None;
291
292 config.retry_interval_ms = static_cast<unsigned int>(retry_interval.count());
293 return config;
294 }
