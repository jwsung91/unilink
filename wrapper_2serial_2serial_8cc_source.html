105 void fulfill_all_locked(bool value) {
106 for (auto& promise : pending_promises_) {
107 try {
108 promise.set_value(value);
109 } catch (...) {
110 }
111 }
112 pending_promises_.clear();
113 }

115 void flush_batches() {
116 std::unique_lock<std::shared_mutex> lock(mutex_);
117 if (!data_batch_queue_.empty()) {
118 auto handler = data_batch_handler_;
119 auto batch = std::move(data_batch_queue_);
120 data_batch_queue_.clear();
121 if (handler) {
122 lock.unlock();
123 handler(batch);
124 lock.lock();
125 }
126 }
127 if (!message_batch_queue_.empty()) {
128 auto handler = message_batch_handler_;
129 auto batch = std::move(message_batch_queue_);
130 message_batch_queue_.clear();
131 if (handler) {
132 lock.unlock();
133 handler(batch);
134 lock.lock();
135 }
136 }
137 if (batch_timer_) {
138 batch_timer_->cancel();
139 }
140 }

142 void schedule_batch_timer() {
143 if (!batch_timer_) return;
144 batch_timer_->expires_after(max_batch_latency_);
145 batch_timer_->async_wait(
146 [this, weak_alive = std::weak_ptr<bool>(alive_marker_)](const boost::system::error_code& ec) {
147 if (ec) return;
148 auto alive = weak_alive.lock();
149 if (!alive) return;
150 flush_batches();
151 });
152 }

154 std::future<bool> start() {
155 std::unique_lock<std::shared_mutex> lock(mutex_);
156 if (channel && channel->is_connected()) {
157 std::promise<bool> p;
158 p.set_value(true);
159 return p.get_future();
160 }
161
162 std::promise<bool> p;
163 auto future = p.get_future();
164 pending_promises_.emplace_back(std::move(p));
165
166 if (started_.load()) {
167 return future;
168 }
169
170 if (!channel) {
171 channel = factory::ChannelFactory::create(build_config_locked(), external_ioc);
172 setup_internal_handlers();
173 }
174 started_.store(true);
175
176 lock.unlock();
177 channel->start();
178 if (use_external_context && manage_external_context && !external_thread.joinable()) {
179 if (external_ioc && external_ioc->stopped()) {
180 external_ioc->restart();
181 }
182 work_guard_ = std::make_unique<boost::asio::executor_work_guard<boost::asio::io_context::executor_type>>(
183 boost::asio::make_work_guard(*external_ioc));
184 external_thread = std::thread([ioc = external_ioc]() {
185 try {
186 ioc->run();
187 } catch (...) {
188 }
189 });
190 }
191
192 return future;
193 }

195 void stop() {
196 std::unique_lock<std::shared_mutex> lock(mutex_);
197 if (!started_.load()) {
198 fulfill_all_locked(false);
199 return;
200 }
201 started_.store(false);
202 bp_cv_.notify_all();
203
204 if (batch_timer_) {
205 batch_timer_->cancel();
206 batch_timer_.reset();
207 }
208
209 if (channel) {
210 channel->on_bytes(nullptr);
211 channel->on_state(nullptr);
212 lock.unlock();
213 channel->stop();
214 lock.lock();
215 }
216
217 if (work_guard_) {
218 work_guard_.reset();
219 }
220
221 if (use_external_context && manage_external_context && external_thread.joinable()) {
222 if (external_ioc) external_ioc->stop();
223 if (std::this_thread::get_id() != external_thread.get_id()) {
224 lock.unlock();
225 external_thread.join();
226 lock.lock();
227 } else {
228 external_thread.detach();
229 }
230 }
231
232 fulfill_all_locked(false);
233
234 if (framer) framer->reset();
235 }

258 bool send_blocking(std::string_view data) {
259 std::unique_lock<std::mutex> bp_lock(bp_mutex_);
260 while (true) {
261 {
262 std::shared_lock<std::shared_mutex> lock(mutex_);
263 if (!started_.load() || !channel || !channel->is_connected()) return false;
264 if (!channel->is_backpressure_active()) break;
265 }
266 bp_cv_.wait(bp_lock);
267 }
268 std::shared_lock<std::shared_mutex> lock(mutex_);
269 auto binary_view = base::safe_convert::string_to_bytes(data);
270 channel->async_write_copy(memory::ConstByteSpan(binary_view.first, binary_view.second));
271 return true;
272 }

276 void setup_internal_handlers() {
277 if (!channel) return;
278
279 batch_timer_ = std::make_unique<boost::asio::steady_timer>(channel->get_executor());
280
281 std::weak_ptr<bool> weak_alive = alive_marker_;
282
283 channel->on_bytes([this, weak_alive](memory::ConstByteSpan data) {
284 auto alive = weak_alive.lock();
285 if (!alive) return;
286
287 std::unique_lock<std::shared_mutex> lock(mutex_);
288 if (data_batch_handler_) {
289 data_batch_queue_.emplace_back(0, memory::SafeDataBuffer(data));
290 if (data_batch_queue_.size() >= max_batch_size_) {
291 auto handler = data_batch_handler_;
292 auto batch = std::move(data_batch_queue_);
293 data_batch_queue_.clear();
294 lock.unlock();
295 handler(batch);
296 } else if (data_batch_queue_.size() == 1) {
297 schedule_batch_timer();
298 }
299 return;
300 }
301
302 MessageHandler handler = data_handler;
303 if (handler) {
304 lock.unlock();
305 handler(MessageContext(0, memory::SafeDataBuffer(data)));
306 lock.lock();
307 }
308
309 if (framer) {
310 framer->push_bytes(data);
311 }
312 });
313
314 channel->on_backpressure([this, weak_alive](size_t queued) {
315 bp_cv_.notify_all();
316 auto alive = weak_alive.lock();
317 if (!alive) return;
318 std::shared_lock<std::shared_mutex> lock(mutex_);
319 if (bp_handler) bp_handler(queued);
320 });
321
322 channel->on_state([this, weak_alive](base::LinkState state) {
323 auto alive = weak_alive.lock();
324 if (!alive) return;
325
326 switch (state) {
327 case base::LinkState::Connected: {
328 ConnectionHandler handler;
329 {
330 std::unique_lock<std::shared_mutex> lock(mutex_);
331 fulfill_all_locked(true);
332 handler = connect_handler;
333 }
334 if (handler) handler(ConnectionContext(0));
335 break;
336 }
337 case base::LinkState::Closed:
338 case base::LinkState::Error:
339 case base::LinkState::Idle: {
340 ConnectionHandler disconnect_handler_snapshot;
341 ErrorHandler error_handler_snapshot;
342 {
343 std::unique_lock<std::shared_mutex> lock(mutex_);
344 fulfill_all_locked(false);
345 if (state == base::LinkState::Error) {
346 error_handler_snapshot = error_handler;
347 } else {
348 disconnect_handler_snapshot = disconnect_handler;
349 }
350 }
351 if (disconnect_handler_snapshot) {
352 disconnect_handler_snapshot(ConnectionContext(0));
353 }
354 if (error_handler_snapshot) {
355 error_handler_snapshot(ErrorContext(ErrorCode::IoError, "Connection error"));
356 }
357 break;
358 }
359 default:
360 break;
361 }
362 });
363 }

365 void attach_framer_callback() {
366 if (!framer) return;
367 framer->on_message([this](memory::ConstByteSpan msg) {
368 std::unique_lock<std::shared_mutex> lock(mutex_);
369 if (message_batch_handler_) {
370 message_batch_queue_.emplace_back(0, memory::SafeDataBuffer(msg));
371 if (message_batch_queue_.size() >= max_batch_size_) {
372 auto handler = message_batch_handler_;
373 auto batch = std::move(message_batch_queue_);
374 message_batch_queue_.clear();
375 lock.unlock();
376 handler(batch);
377 } else if (message_batch_queue_.size() == 1) {
378 schedule_batch_timer();
379 }
380 return;
381 }
382
383 MessageHandler handler = message_handler;
384 if (handler) {
385 lock.unlock();
386 handler(MessageContext(0, memory::SafeDataBuffer(msg)));
387 }
388 });
389 }

409 config::SerialConfig build_config_locked() const {
410 config::SerialConfig config;
411 config.device = device;
412 config.baud_rate = baud_rate;
413 config.char_size = static_cast<unsigned int>(data_bits);
414 config.stop_bits = static_cast<unsigned int>(stop_bits);
415 std::string p = to_lower(parity);
416 if (p == "even")
417 config.parity = config::SerialConfig::Parity::Even;
418 else if (p == "odd")
419 config.parity = config::SerialConfig::Parity::Odd;
420 else
421 config.parity = config::SerialConfig::Parity::None;
422
423 std::string f = to_lower(flow_control);
424 if (f == "software")
425 config.flow = config::SerialConfig::Flow::Software;
426 else if (f == "hardware")
427 config.flow = config::SerialConfig::Flow::Hardware;
428 else
429 config.flow = config::SerialConfig::Flow::None;
430
431 config.retry_interval_ms = static_cast<unsigned int>(retry_interval.count());
432 config.backpressure_threshold = backpressure_threshold;
433 config.backpressure_strategy = backpressure_strategy;
434 return config;
435 }
