60 void Serial::start() {
61  if (started_) return;
62 
63  if (use_external_context_) {
64  if (!external_ioc_) {
65  throw std::runtime_error("External io_context is not set");
66  }
67  if (manage_external_context_) {
68  if (external_ioc_->stopped()) {
69  external_ioc_->restart();
70  }
71  }
72  }
73 
74  if (!channel_) {
75  channel_ = factory::ChannelFactory::create(build_config(), external_ioc_);
76  setup_internal_handlers();
77  }
78 
79  channel_->start();
80  if (use_external_context_ && manage_external_context_ && !external_thread_.joinable()) {
81  external_thread_ = std::thread([ioc = external_ioc_]() {
82  boost::asio::executor_work_guard<boost::asio::io_context::executor_type> guard(ioc->get_executor());
83  ioc->run();
84  });
85  }
86  started_ = true;
87 }

89 void Serial::stop() {

91 
92  channel_->stop();
93 
94  if (use_external_context_ && manage_external_context_) {
95  if (external_ioc_) {
96  external_ioc_->stop();
97  }
98  if (external_thread_.joinable()) {
99  external_thread_.join();
100  }
101  }
102 
103  started_ = false;
104 }

172 config::SerialConfig Serial::build_config() const {
173  config::SerialConfig config;
174  config.device = device_;
175  config.baud_rate = baud_rate_;
176  config.char_size = static_cast<unsigned int>(data_bits_);
177  config.stop_bits = static_cast<unsigned int>(stop_bits_);
178 
179  std::string parity_lower = to_lower(parity_);
180  if (parity_lower == "none")
181  config.parity = config::SerialConfig::Parity::None;
182  else if (parity_lower == "even")
183  config.parity = config::SerialConfig::Parity::Even;
184  else if (parity_lower == "odd")
185  config.parity = config::SerialConfig::Parity::Odd;
186 
187  std::string flow_lower = to_lower(flow_control_);
188  if (flow_lower == "none")
189  config.flow = config::SerialConfig::Flow::None;
190  else if (flow_lower == "software")
191  config.flow = config::SerialConfig::Flow::Software;
192  else if (flow_lower == "hardware")
193  config.flow = config::SerialConfig::Flow::Hardware;
194 
195  config.retry_interval_ms = static_cast<unsigned int>(retry_interval_.count());
196  return config;
197 }

201 void Serial::setup_internal_handlers() {
202  if (!channel_) return;
203 
204  channel_->on_bytes([this](memory::ConstByteSpan data) {
205  if (bytes_handler_) {
206  bytes_handler_(data);
207  }
208  if (data_handler_) {
209  std::string str_data = common::safe_convert::uint8_to_string(data.data(), data.size());
210  data_handler_(str_data);
211  }
212  });
213 
214  channel_->on_state([this](base::LinkState state) { notify_state_change(state); });
215 }

217 void Serial::notify_state_change(base::LinkState state) {
218  switch (state) {
219  case base::LinkState::Connected:
220  if (connect_handler_) connect_handler_();
221  break;
222  case base::LinkState::Closed:
223  if (disconnect_handler_) disconnect_handler_();
224  break;
225  case base::LinkState::Error:
226  if (error_handler_) error_handler_("Connection error");
227  break;
228  default:
229  break;
230  }
231 }
