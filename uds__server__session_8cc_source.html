104 void UdsServerSession::do_write() {
105  if (tx_.empty() || writing_) return;
106  writing_ = true;
107  current_write_buffer_ = std::move(tx_.front());
108  tx_.pop_front();
109 
110  net::const_buffer buffer;
111  std::visit(
112  [&buffer](auto&& arg) {
113  using T = std::decay_t<decltype(arg)>;
114  if constexpr (std::is_same_v<T, std::vector<uint8_t>>)
115  buffer = net::buffer(arg);
116  else if constexpr (std::is_same_v<T, std::shared_ptr<const std::vector<uint8_t>>>)
117  buffer = net::buffer(*arg);
118  else if constexpr (std::is_same_v<T, memory::PooledBuffer>)
119  buffer = net::buffer(arg.data(), arg.size());
120  },
121  *current_write_buffer_);
122 
123  size_t bytes_to_write = buffer.size();
124  socket_->async_write(buffer, net::bind_executor(strand_, [this, self = shared_from_this(), bytes_to_write](
125  const boost::system::error_code& ec, size_t) {
126  if (closing_ || !alive_) return;
127  writing_ = false;
128  current_write_buffer_ = std::nullopt;
129  queue_bytes_ = (queue_bytes_ >= bytes_to_write) ? (queue_bytes_ - bytes_to_write) : 0;
130  report_backpressure(queue_bytes_);
131 
132  if (ec) {
133  do_close();
134  return;
135  }
136  if (!tx_.empty()) do_write();
137  }));
138 }

140 void UdsServerSession::do_close() {
141  if (!closing_.exchange(true) && !alive_) return;
142  alive_ = false;
143  auto close_cb = std::move(on_close_);
144  on_bytes_ = nullptr;
145  on_bp_ = nullptr;
146  on_close_ = nullptr;
147  tx_.clear();
148  current_write_buffer_ = std::nullopt;
149  queue_bytes_ = 0;
150  writing_ = false;
151  boost::system::error_code ec;
152  socket_->close(ec);
153  if (close_cb) {
154  try {
155  close_cb();
156  } catch (...) {
157  }
158  }
159 }
