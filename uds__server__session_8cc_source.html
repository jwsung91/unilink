70void UdsServerSession::async_write_move(std::vector<uint8_t>&& data) {
71 net::post(strand_, [this, self = shared_from_this(), data = std::move(data)]() mutable {
72 if (!alive_) return;
73 size_t added = data.size();
74 if (queue_bytes_ + added > bp_limit_) return;
75
76 queue_bytes_ += added;
77 tx_.emplace_back(std::move(data));
78 report_backpressure(queue_bytes_);
79 if (!writing_) do_write();

100void UdsServerSession::start_read() {
101 socket_->async_read_some(
102 net::buffer(rx_),
103 net::bind_executor(strand_, [this, self = shared_from_this()](const boost::system::error_code& ec, size_t bytes) {
104 if (closing_ || !alive_) return;
105 if (ec) {
106 do_close();
107 return;
108 }
109 if (on_bytes_) on_bytes_(memory::ConstByteSpan(rx_.data(), bytes));
110 reset_idle_timer();
111 start_read();
112 }));
113}

115void UdsServerSession::do_write() {
116 if (tx_.empty() || writing_) return;
117 writing_ = true;
118 current_write_buffer_ = std::move(tx_.front());
119 tx_.pop_front();
120
121 net::const_buffer buffer;
122 std::visit(
123 [&buffer](auto&& arg) {
124 using T = std::decay_t<decltype(arg)>;
125 if constexpr (std::is_same_v<T, std::vector<uint8_t>>)
126 buffer = net::buffer(arg);
127 else if constexpr (std::is_same_v<T, std::shared_ptr<const std::vector<uint8_t>>>)
128 buffer = net::buffer(*arg);
129 else if constexpr (std::is_same_v<T, memory::PooledBuffer>)
130 buffer = net::buffer(arg.data(), arg.size());
131 },
132 *current_write_buffer_);
133
134 size_t bytes_to_write = buffer.size();
135 socket_->async_write(buffer, net::bind_executor(strand_, [this, self = shared_from_this(), bytes_to_write](
136 const boost::system::error_code& ec, size_t) {
137 if (closing_ || !alive_) return;
138 writing_ = false;
139 current_write_buffer_ = std::nullopt;
140 queue_bytes_ = (queue_bytes_ >= bytes_to_write) ? (queue_bytes_ - bytes_to_write) : 0;
141 report_backpressure(queue_bytes_);
142
143 if (ec) {
144 do_close();
145 return;
146 }
147 if (!tx_.empty()) do_write();
148 }));
149}

151void UdsServerSession::do_close() {
152 if (!closing_.exchange(true) && !alive_) return;
153 alive_ = false;
154 auto close_cb = std::move(on_close_);
155 on_bytes_ = nullptr;
156 on_bp_ = nullptr;
157 on_close_ = nullptr;
158 tx_.clear();
159 current_write_buffer_ = std::nullopt;
160 queue_bytes_ = 0;
161 writing_ = false;
162 boost::system::error_code ec;
163 socket_->close(ec);
164 if (close_cb) {
165 try {
166 close_cb();
167 } catch (...) {
168 }
169 }
170}

177void UdsServerSession::reset_idle_timer() {
178 if (idle_timeout_ms_ <= 0) return;
179
180 idle_timer_.cancel();
181 idle_timer_.expires_after(std::chrono::milliseconds(idle_timeout_ms_));
182
183 auto self = shared_from_this();
184 idle_timer_.async_wait(net::bind_executor(strand_, [self](const boost::system::error_code& ec) {
185 if (ec == boost::asio::error::operation_aborted) return;
186 if (!self->alive_ || self->closing_) return;
187 UNILINK_LOG_WARNING("uds_server_session", "timeout", "Connection idle timeout expired, closing session");
188 self->do_close();
189 }));
190}
