64 
65  if (stop_bits != 1 && stop_bits != 2) stop_bits = 1;
66

69  } else if (retry_interval_ms > common::constants::MAX_RETRY_INTERVAL_MS) {
70  retry_interval_ms = common::constants::MAX_RETRY_INTERVAL_MS;
71  }
72 
73  if (backpressure_threshold < common::constants::MIN_BACKPRESSURE_THRESHOLD) {
74  backpressure_threshold = common::constants::MIN_BACKPRESSURE_THRESHOLD;
75  } else if (backpressure_threshold > common::constants::MAX_BACKPRESSURE_THRESHOLD) {

77  }
78 
79  if (max_retries != -1 && max_retries > common::constants::MAX_RETRIES_LIMIT) {
80  max_retries = common::constants::MAX_RETRIES_LIMIT;
81  }
82  }
