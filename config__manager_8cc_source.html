50 ValidationResult ConfigManager::set(const std::string& key, const std::any& value) {
51  std::lock_guard<std::mutex> lock(mutex_);
52 
53  // Validate the value
54  auto validation_result = validate_value(key, value);
55  if (!validation_result.is_valid) {
56  return validation_result;
57  }
58 
59  // Store old value for change notification
60  std::any old_value;
61  bool had_key = false;
62  auto it = config_items_.find(key);
63  if (it != config_items_.end()) {
64  old_value = it->second.value;
65  had_key = true;
66  it->second.value = value;
67  } else {
68  // Create new config item
69  ConfigItem item(key, value, ConfigType::String, false);
70  config_items_[key] = item;
71  }
72 
73  // Notify change
74  if (had_key) {
75  notify_change(key, old_value, value);
76  }
77 
78  return ValidationResult::success();
79 }

167 bool ConfigManager::load_from_file(const std::string& filepath) {
168  std::lock_guard<std::mutex> lock(mutex_);
169 
170  try {
171  std::ifstream file(filepath);
172  if (!file.is_open()) {
173  return false;
174  }
175 
176  std::string line;
177  while (std::getline(file, line)) {
178  // Skip comments and empty lines
179  if (line.empty() || line[0] == '#') {
180  continue;
181  }
182 
183  // Parse key=value pairs
184  size_t pos = line.find('=');
185  if (pos != std::string::npos) {
186  std::string key = line.substr(0, pos);
187  std::string value_str = line.substr(pos + 1);
188 
189  // Trim whitespace
190  key.erase(0, key.find_first_not_of(" \t"));
191  key.erase(key.find_last_not_of(" \t") + 1);
192  value_str.erase(0, value_str.find_first_not_of(" \t"));
193  value_str.erase(value_str.find_last_not_of(" \t") + 1);
194 
195  // Try to determine type and deserialize
196  ConfigType type = ConfigType::String;
197  if (value_str == "true" || value_str == "false") {
198  type = ConfigType::Boolean;
199  } else if (std::all_of(value_str.begin(), value_str.end(),
200  [](char c) { return std::isdigit(c) || c == '-'; })) {
201  type = ConfigType::Integer;
202  } else if (std::count(value_str.begin(), value_str.end(), '.') == 1 &&
203  std::all_of(value_str.begin(), value_str.end(),
204  [](char c) { return std::isdigit(c) || c == '.' || c == '-'; })) {
205  type = ConfigType::Double;
206  }
207 
208  std::any value = deserialize_value(value_str, type);
209  ConfigItem item(key, value, type, false);
210  config_items_[key] = item;
211  }
212  }
213 
214  return true;
215  } catch (const std::exception& e) {
216  UNILINK_LOG_ERROR("config_manager", "load", "Error loading configuration: " + std::string(e.what()));
217  return false;
218  }

260 ValidationResult ConfigManager::validate_value(const std::string& key, const std::any& value) const {
261  auto it = config_items_.find(key);
262  if (it != config_items_.end()) {
263  // Check if there's a custom validator
264  if (it->second.validator) {
265  return it->second.validator(value);
266  }
267 
268  // Basic type validation
269  ConfigType expected_type = it->second.type;
270  ConfigType actual_type = ConfigType::String;
271 
272  if (value.type() == typeid(std::string)) {
273  actual_type = ConfigType::String;
274  } else if (value.type() == typeid(int)) {
275  actual_type = ConfigType::Integer;
276  } else if (value.type() == typeid(bool)) {
277  actual_type = ConfigType::Boolean;
278  } else if (value.type() == typeid(double)) {
279  actual_type = ConfigType::Double;
280  }
281 
282  if (expected_type != actual_type) {
283  return ValidationResult::error("Type mismatch for key '" + key + "'");
284  }
285  }
286 
287  return ValidationResult::success();
288 }

302 std::string ConfigManager::serialize_value(const std::any& value, ConfigType type) const {
303  try {
304  switch (type) {
305  case ConfigType::String:
306  return std::any_cast<std::string>(value);
307  case ConfigType::Integer:
308  return std::to_string(std::any_cast<int>(value));
309  case ConfigType::Boolean:
310  return std::any_cast<bool>(value) ? "true" : "false";
311  case ConfigType::Double:
312  return std::to_string(std::any_cast<double>(value));
313  default:
314  return "unknown";
315  }
316  } catch (const std::bad_any_cast&) {
317  // Fallback to string representation
318  return "unknown";
319  }
320 }

322 std::any ConfigManager::deserialize_value(const std::string& value_str, ConfigType type) const {
323  try {
324  switch (type) {

327  case ConfigType::Integer:
328  return std::any(std::stoi(value_str));
329  case ConfigType::Boolean:
330  return std::any(value_str == "true");
331  case ConfigType::Double:
332  return std::any(std::stod(value_str));
333  default:
334  return std::any(value_str);
335  }
336  } catch (const std::exception&) {
337  // Fallback to string if parsing fails
338  return std::any(value_str);
