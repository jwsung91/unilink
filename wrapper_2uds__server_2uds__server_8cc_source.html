77  std::future<bool> start() {
78  std::lock_guard<std::mutex> lock(mutex_);
79  if (started_) {
80  std::promise<bool> p;
81  p.set_value(true);
82  return p.get_future();
83  }
84 
85  start_promise_ = std::promise<bool>();
86  auto f = start_promise_.get_future();
87 
88  if (!server_) {
89  config::UdsServerConfig cfg;
90  cfg.socket_path = socket_path_;
91  cfg.max_connections = static_cast<int>(max_clients_);
92 
93  if (use_external_context_) {
94  server_ = std::dynamic_pointer_cast<transport::UdsServer>(factory::ChannelFactory::create(cfg, external_ioc_));
95  if (manage_external_context_ && !external_thread_.joinable()) {
96  work_guard_ = std::make_unique<boost::asio::executor_work_guard<boost::asio::io_context::executor_type>>(
97  boost::asio::make_work_guard(*external_ioc_));
98  external_thread_ = std::thread([this]() {
99  try {
100  external_ioc_->run();
101  } catch (...) {
102  }
103  });
104  }
105  } else {
106  server_ = std::dynamic_pointer_cast<transport::UdsServer>(factory::ChannelFactory::create(cfg));
107  }
108  setup_internal_handlers();
109  }
110 
111  server_->start();
112  started_ = true;
113  start_promise_.set_value(true);
114  return f;
115  }

117  void stop() {
118  std::unique_lock<std::mutex> lock(mutex_);
119  if (!started_.exchange(false)) {
120  return;
121  }
122 
123  if (server_) {
124  server_->on_multi_connect(nullptr);
125  server_->on_multi_data(nullptr);
126  server_->on_multi_disconnect(nullptr);
127  server_->on_bytes(nullptr);
128  server_->on_state(nullptr);
129  server_->on_backpressure(nullptr);
130  lock.unlock();
131  server_->stop();
132  lock.lock();
133  }
134 
135  if (work_guard_) {
136  work_guard_.reset();
137  }
138 
139  if (external_thread_.joinable()) {
140  if (std::this_thread::get_id() != external_thread_.get_id()) {
141  lock.unlock(); // RELEASE LOCK BEFORE JOINING
142  external_thread_.join();
143  lock.lock(); // RE-ACQUIRE
144  } else {
145  external_thread_.detach();
146  }
147  }
148 
149  server_.reset();
150  }

152  void setup_internal_handlers() {
153  if (!server_) return;
154 
155  std::weak_ptr<bool> weak_alive = alive_marker_;
156 
157  server_->on_multi_connect([weak_alive, this](size_t client_id, const std::string& info) {
158  auto alive = weak_alive.lock();
159  if (!alive) return;
160 
161  ConnectionHandler handler;
162  {
163  std::lock_guard<std::mutex> lock(mutex_);
164  handler = client_connect_handler_;
165  }
166  if (handler) {
167  handler(ConnectionContext(client_id, info));
168  }
169  });
170 
171  server_->on_multi_disconnect([weak_alive, this](size_t client_id) {
172  auto alive = weak_alive.lock();
173  if (!alive) return;
174 
175  ConnectionHandler handler;
176  {
177  std::lock_guard<std::mutex> lock(mutex_);
178  handler = client_disconnect_handler_;
179  }
180  if (handler) {
181  handler(ConnectionContext(client_id, "disconnected"));
182  }
183  });
184 
185  server_->on_multi_data([weak_alive, this](size_t client_id, const std::vector<uint8_t>& data) {
186  auto alive = weak_alive.lock();
187  if (!alive) return;
188 
189  MessageHandler handler;
190  {
191  std::lock_guard<std::mutex> lock(mutex_);
192  handler = data_handler_;
193  }
194  if (handler) {
195  handler(MessageContext(client_id, std::string_view(reinterpret_cast<const char*>(data.data()), data.size())));
196  }
197  });
198  }
