275 void setup_internal_handlers() {
276 if (!server_) return;
277
278 batch_timer_ = std::make_unique<boost::asio::steady_timer>(server_->get_executor());

282 auto transport_server = std::dynamic_pointer_cast<transport::UdsServer>(server_);

299 transport_server->on_multi_data([this, weak_alive](ClientId id, memory::ConstByteSpan data_span) {
300 auto alive = weak_alive.lock();
301 if (!alive) return;

304 std::unique_lock<std::shared_mutex> lock(mutex_);

307 if (data_batch_queue_.size() >= max_batch_size_) {
308 auto handler = data_batch_handler_;
309 auto batch = std::move(data_batch_queue_);
310 data_batch_queue_.clear();

313 lock.lock();
314 } else if (data_batch_queue_.size() == 1) {
315 schedule_batch_timer();

317 } else {

327 if (it != framers_.end()) {

329 }
330 lock.unlock();

332 });

344 });

360 auto alive = weak_alive.lock();

385 void attach_framer_callback(ClientId id) {
386 auto it = framers_.find(id);
387 if (it == framers_.end()) return;
388
389 it->second->on_message([this, id](memory::ConstByteSpan msg) {
390 std::unique_lock<std::shared_mutex> lock(mutex_);
391 if (on_message_batch_) {
392 message_batch_queue_.emplace_back(id, memory::SafeDataBuffer(msg));
393 if (message_batch_queue_.size() >= max_batch_size_) {
394 auto handler = on_message_batch_;
395 auto batch = std::move(message_batch_queue_);
396 message_batch_queue_.clear();
397 lock.unlock();
398 handler(batch);
399 } else if (message_batch_queue_.size() == 1) {
400 schedule_batch_timer();
401 }
402 return;
403 }
404
405 MessageHandler handler = on_message_;
406 if (handler) {
407 lock.unlock();
408 handler(MessageContext(id, memory::SafeDataBuffer(msg)));
409 }
410 });
