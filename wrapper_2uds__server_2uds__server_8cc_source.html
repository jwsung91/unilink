85 void fulfill_all_locked(bool value) {
86 for (auto& p : pending_promises_) {

88 p.set_value(value);

95 std::future<bool> start() {
96 std::unique_lock<std::shared_mutex> lock(mutex_);
97 if (is_listening_.load()) {
98 std::promise<bool> p;
99 p.set_value(true);
100 return p.get_future();
101 }
102
103 std::promise<bool> p;
104 auto f = p.get_future();
105 pending_promises_.emplace_back(std::move(p));
106
107 if (started_.exchange(true)) {
108 return f;
109 }
110
111 if (!server_) {
112 config::UdsServerConfig cfg;
113 cfg.socket_path = socket_path_;
114 cfg.max_connections = static_cast<int>(max_clients_);
115 cfg.idle_timeout_ms = idle_timeout_ms_;
116
117 if (use_external_context_) {
118 server_ = std::dynamic_pointer_cast<transport::UdsServer>(factory::ChannelFactory::create(cfg, external_ioc_));
119 if (manage_external_context_ && !external_thread_.joinable()) {
120 if (external_ioc_ && external_ioc_->stopped()) {
121 external_ioc_->restart();
122 }
123 work_guard_ = std::make_unique<boost::asio::executor_work_guard<boost::asio::io_context::executor_type>>(
124 boost::asio::make_work_guard(*external_ioc_));
125 external_thread_ = std::thread([this]() {
126 try {
127 external_ioc_->run();
128 } catch (...) {
129 }
130 });
131 }
132 } else {
133 server_ = std::dynamic_pointer_cast<transport::UdsServer>(factory::ChannelFactory::create(cfg));
134 }
135 setup_internal_handlers();
136 }
137
138 lock.unlock();
139 server_->start();
140 return f;
141 }
142
143 void stop() {
144 std::unique_lock<std::shared_mutex> lock(mutex_);
145 if (!started_.exchange(false)) {
146 fulfill_all_locked(false);
147 return;
148 }
149
150 if (server_) {
151 server_->on_multi_connect(nullptr);
152 server_->on_multi_data(nullptr);
153 server_->on_multi_disconnect(nullptr);
154 server_->on_bytes(nullptr);
155 server_->on_state(nullptr);
156 server_->on_backpressure(nullptr);
157 lock.unlock();
158 server_->stop();
159 lock.lock();
160 }
161
162 if (work_guard_) {
163 work_guard_.reset();
164 }
165
166 if (use_external_context_ && manage_external_context_ && external_ioc_) {
167 external_ioc_->stop();
168 }
169
170 if (external_thread_.joinable()) {
171 if (std::this_thread::get_id() != external_thread_.get_id()) {
172 lock.unlock(); // RELEASE LOCK BEFORE JOINING
173 external_thread_.join();
174 lock.lock(); // RE-ACQUIRE
175 } else {
176 external_thread_.detach();
177 }
178 }
179
180 is_listening_ = false;
181 fulfill_all_locked(false);
182 server_.reset();
183 }
184
185 void setup_internal_handlers() {
186 if (!server_) return;
187
188 std::weak_ptr<bool> weak_alive = alive_marker_;
189
190 server_->on_state([weak_alive, this](base::LinkState state) {
191 auto alive = weak_alive.lock();
192 if (!alive) return;
193
194 ErrorHandler error_handler;
195 {
196 std::lock_guard<std::shared_mutex> lock(mutex_);
197 if (state == base::LinkState::Listening || state == base::LinkState::Connected) {
198 is_listening_ = true;
199 fulfill_all_locked(true);
200 } else if (state == base::LinkState::Error || state == base::LinkState::Closed ||
201 state == base::LinkState::Idle) {
202 is_listening_ = false;
203 fulfill_all_locked(false);
204 if (state == base::LinkState::Error) {
205 error_handler = error_handler_;
206 }
207 }
208 }
209
210 if (error_handler) {
211 error_handler(ErrorContext(ErrorCode::IoError, "Server error"));
212 }
213 });
214
215 server_->on_multi_connect([weak_alive, this](ClientId client_id, const std::string& info) {
216 auto alive = weak_alive.lock();
217 if (!alive) return;
218
219 {
220 std::lock_guard<std::shared_mutex> lock(mutex_);
221 if (framer_factory_) {
222 auto framer = framer_factory_();
223 if (framer) {
224 framer->on_message([this, client_id](memory::ConstByteSpan msg) {
225 MessageHandler on_message_handler;
226 {
227 std::lock_guard<std::shared_mutex> lock(mutex_);
228 on_message_handler = on_message_;
229 }
230 if (on_message_handler) {
231 std::string str_msg = base::safe_convert::uint8_to_string(msg.data(), msg.size());
232 on_message_handler(MessageContext(client_id, str_msg));
233 }
234 });
235 framers_[client_id] = std::move(framer);
236 }
237 }
238 }
239
240 ConnectionHandler handler;
241 {
242 std::lock_guard<std::shared_mutex> lock(mutex_);
243 handler = client_connect_handler_;
244 }
245 if (handler) {
246 handler(ConnectionContext(client_id, info));
247 }
248 });
249
250 server_->on_multi_disconnect([weak_alive, this](ClientId client_id) {
251 auto alive = weak_alive.lock();
252 if (!alive) return;
253
254 {
255 std::lock_guard<std::shared_mutex> lock(mutex_);
256 framers_.erase(client_id);
257 }
258
259 ConnectionHandler handler;
260 {
261 std::shared_lock<std::shared_mutex> lock(mutex_);
262 handler = client_disconnect_handler_;
263 }
264 if (handler) {
265 handler(ConnectionContext(client_id, "disconnected"));
266 }
267 });
268
269 server_->on_multi_data([weak_alive, this](ClientId client_id, memory::ConstByteSpan data_span) {
270 auto alive = weak_alive.lock();
271 if (!alive) return;
272
273 // 1. Raw data handler
274 MessageHandler handler;
275 {
276 std::shared_lock<std::shared_mutex> lock(mutex_);
277 handler = data_handler_;
278 }
279 if (handler) {
280 handler(MessageContext(client_id,
281 std::string_view(reinterpret_cast<const char*>(data_span.data()), data_span.size())));
282 }
283
284 // 2. Framer integration
285 {
286 std::shared_lock<std::shared_mutex> lock(mutex_);
287 auto it = framers_.find(client_id);
288 if (it != framers_.end()) {
289 it->second->push_bytes(data_span);
290 }
291 }
292 });
293 }
