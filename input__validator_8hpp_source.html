164 inline void InputValidator::validate_memory_alignment(const void* ptr, size_t alignment) {
165  if (ptr == nullptr) {
166  throw ValidationException("memory pointer cannot be null", "ptr", "non-null pointer");
167  }
168 
169  uintptr_t address = reinterpret_cast<uintptr_t>(ptr);
170  if (address % alignment != 0) {
171  throw ValidationException("memory pointer not properly aligned", "ptr",
172  "aligned to " + std::to_string(alignment) + " bytes");
173  }
