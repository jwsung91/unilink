37 // std::hash<boost::asio::ip::udp::endpoint> is not available before Boost 1.74.
38 // Provide a portable hash by combining the raw address bytes and port.
39 struct UdpEndpointHash {
40  std::size_t operator()(const boost::asio::ip::udp::endpoint& ep) const noexcept {
41  std::size_t seed = 0;
42  auto combine = [&](std::size_t v) { seed ^= v + 0x9e3779b9u + (seed << 6) + (seed >> 2); };
43  if (ep.address().is_v4()) {
44  for (auto byte : ep.address().to_v4().to_bytes()) {
45  combine(std::hash<unsigned char>{}(byte));
46  }
47  } else {
48  for (auto byte : ep.address().to_v6().to_bytes()) {
49  combine(std::hash<unsigned char>{}(byte));
50  }
51  }
52  combine(std::hash<unsigned short>{}(ep.port()));
53  return seed;
54  }
55 };

120  void schedule_reaper() {
121  if (!started.load() || !reaper_timer) return;
122 
123  // Run reaper at interval proportional to timeout (min 100ms, max 5s)
124  auto interval =
125  std::max(std::chrono::milliseconds(100), std::min(std::chrono::milliseconds(5000), session_timeout / 2));
126 
127  reaper_timer->expires_after(interval);
128  reaper_timer->async_wait([this, alive = std::weak_ptr<bool>(is_alive)](const boost::system::error_code& ec) {
129  auto lock = alive.lock();
130  if (!lock || !(*lock)) return;
131 
132  if (!ec) {
133  run_reaper();
134  schedule_reaper();
135  }
136  });
137  }

139  void run_reaper() {
140  std::vector<std::pair<size_t, std::string>> to_remove_with_info;
141  auto now = std::chrono::steady_clock::now();
142 
143  {
144  std::lock_guard<std::shared_mutex> lock(mutex);
145  for (auto& [id, entry] : sessions) {
146  if (now - entry.last_seen > session_timeout) {
147  std::string info = entry.endpoint.address().to_string() + ":" + std::to_string(entry.endpoint.port());
148  endpoint_to_id.erase(entry.endpoint);
149  to_remove_with_info.push_back({id, info});
150  }
151  }
152  for (auto const& [id, info] : to_remove_with_info) {
153  sessions.erase(id);
154  }
155  }
156 
157  // Call disconnect handlers outside the lock
158  if (on_disconnect) {
159  for (auto const& [id, info] : to_remove_with_info) {
160  on_disconnect(ConnectionContext(id, info));
161  }
162  }
163  }

165  void setup_internal_handlers() {
166  if (!channel) return;
167 
168  channel->on_bytes_from([this](memory::ConstByteSpan data, const boost::asio::ip::udp::endpoint& ep) {
169  size_t client_id = 0;
170  bool is_new = false;
171  ConnectionHandler connect_handler_copy{nullptr};
172  MessageHandler data_handler_copy{nullptr};
173 
174  {
175  std::lock_guard<std::shared_mutex> lock(mutex);
176  auto it = endpoint_to_id.find(ep);
177  if (it == endpoint_to_id.end()) {
178  client_id = next_client_id++;
179  endpoint_to_id[ep] = client_id;
180  SessionEntry entry;
181  entry.endpoint = ep;
182  entry.last_seen = std::chrono::steady_clock::now();
183  is_new = true;
184 
185  // Create framer for new session
186  if (framer_factory) {
187  auto framer = framer_factory();
188  if (framer) {
189  framer->set_on_message([this, client_id](memory::ConstByteSpan msg) {
190  MessageHandler on_message_handler;
191  {
192  std::shared_lock<std::shared_mutex> lock(mutex);
193  on_message_handler = on_message;
194  }
195  if (on_message_handler) {
196  on_message_handler(
197  MessageContext(client_id, common::safe_convert::uint8_to_string(msg.data(), msg.size())));
198  }
199  });
200  entry.framer = std::move(framer);
201  }
202  }
203  sessions[client_id] = std::move(entry);
204  } else {
205  client_id = it->second;
206  sessions[client_id].last_seen = std::chrono::steady_clock::now();
207  }
208  connect_handler_copy = on_connect;
209  data_handler_copy = on_data;
210  }
211 
212  if (is_new && connect_handler_copy) {
213  connect_handler_copy(ConnectionContext(client_id, ep.address().to_string() + ":" + std::to_string(ep.port())));
214  }
215 
216  if (data_handler_copy) {
217  std::string str_data = common::safe_convert::uint8_to_string(data.data(), data.size());
218  data_handler_copy(MessageContext(client_id, str_data));
219  }
220 
221  // Push to framer
222  std::shared_ptr<framer::IFramer> target_framer;
223  {
224  std::shared_lock<std::shared_mutex> lock(mutex);
225  auto it = sessions.find(client_id);
226  if (it != sessions.end()) {
227  target_framer = it->second.framer;
228  }
229  }
230  if (target_framer) {
231  target_framer->push_bytes(data);
232  }
233  });
234 
235  channel->on_state([this](base::LinkState state) {
236  ErrorHandler error_handler_copy{nullptr};
237  if (state == base::LinkState::Listening || state == base::LinkState::Connected) {
238  std::lock_guard<std::shared_mutex> lock(mutex);
239  is_listening = true;
240  fulfill_all_locked(true);
241  } else if (state == base::LinkState::Error || state == base::LinkState::Closed ||
242  state == base::LinkState::Idle) {
243  std::lock_guard<std::shared_mutex> lock(mutex);
244  is_listening = false;
245  fulfill_all_locked(false);
246  if (state == base::LinkState::Error) {
247  error_handler_copy = on_error;
248  }
249  }
250 
251  if (error_handler_copy) {
252  error_handler_copy(ErrorContext(ErrorCode::IoError, "Server error"));
253  }
254  });
255  }

257  std::future<bool> start() {
258  std::unique_lock<std::shared_mutex> lock(mutex);
259  if (is_listening.load()) {
260  std::promise<bool> p;
261  p.set_value(true);
262  return p.get_future();
263  }
264 
265  std::promise<bool> p;
266  auto fut = p.get_future();
267  pending_promises.emplace_back(std::move(p));
268 
269  if (started.exchange(true)) {
270  return fut;
271  }
272 
273  if (!channel) {
274  channel = std::dynamic_pointer_cast<transport::UdpChannel>(factory::ChannelFactory::create(cfg, external_ioc));
275  setup_internal_handlers();
276  }
277 
278  lock.unlock();
279  channel->start();
280 
281  lock.lock();
282  if (use_external_context && manage_external_context && !external_thread.joinable()) {
283  if (external_ioc->stopped()) external_ioc->restart();
284  work_guard = std::make_unique<boost::asio::executor_work_guard<boost::asio::io_context::executor_type>>(
285  external_ioc->get_executor());
286  external_thread = std::thread([ioc = external_ioc]() {
287  try {
288  ioc->run();
289  } catch (...) {
290  }
291  });
292  }
293 
294  if (channel) {
295  reaper_timer = std::make_unique<boost::asio::steady_timer>(channel->get_executor());
296  schedule_reaper();
297  }
298 
299  return fut;
300  }
301 
302  void stop() {
303  bool should_join = false;
304  {
305  std::unique_lock<std::shared_mutex> lock(mutex);
306  if (!started.exchange(false)) {
307  is_listening = false;
308  fulfill_all_locked(false);
309  return;
310  }
311 
312  if (reaper_timer) {
313  reaper_timer->cancel();
314  reaper_timer.reset();
315  }
316 
317  if (channel) {
318  channel->on_bytes_from(nullptr);
319  channel->on_state(nullptr);
320  lock.unlock();
321  channel->stop();
322  lock.lock();
323  }
324 
325  if (use_external_context && manage_external_context && external_thread.joinable()) {
326  if (external_ioc) external_ioc->stop();
327  should_join = true;
328  }
329 
330  is_listening = false;
331  endpoint_to_id.clear();
332  sessions.clear();
333  next_client_id = 1;
334  fulfill_all_locked(false);
335  }
336 
337  if (should_join) {
338  try {
339  if (std::this_thread::get_id() != external_thread.get_id()) {
340  external_thread.join();
341  } else {
342  external_thread.detach();
343  }
344  } catch (...) {
345  }
346  }
347  }

372 bool UdpServer::broadcast(std::string_view data) {
373  std::lock_guard<std::shared_mutex> lock(impl_->mutex);
374  if (!impl_->channel) return false;
375 
376  auto bytes = common::safe_convert::string_to_bytes(data);
377  for (const auto& [id, entry] : impl_->sessions) {
378  impl_->channel->async_write_to(memory::ConstByteSpan(bytes.first, bytes.second), entry.endpoint);
379  }
380  return true;
381 }
