37// std::hash<boost::asio::ip::udp::endpoint> is not available before Boost 1.74.
38// Provide a portable hash by combining the raw address bytes and port.
39struct UdpEndpointHash {
40 std::size_t operator()(const boost::asio::ip::udp::endpoint& ep) const noexcept {
41 std::size_t seed = 0;
42 auto combine = [&](std::size_t v) { seed ^= v + 0x9e3779b9u + (seed << 6) + (seed >> 2); };
43 if (ep.address().is_v4()) {
44 for (auto byte : ep.address().to_v4().to_bytes()) {
45 combine(std::hash<unsigned char>{}(byte));
46 }
47 } else {
48 for (auto byte : ep.address().to_v6().to_bytes()) {
49 combine(std::hash<unsigned char>{}(byte));
50 }
51 }
52 combine(std::hash<unsigned short>{}(ep.port()));
53 return seed;
54 }
55};

129 void flush_batches() {
130 std::unique_lock<std::shared_mutex> lock(mutex);
131 if (!data_batch_queue_.empty()) {
132 auto handler = on_data_batch_;
133 auto batch = std::move(data_batch_queue_);
134 data_batch_queue_.clear();
135 if (handler) {
136 lock.unlock();
137 handler(batch);
138 lock.lock();
139 }
140 }
141 if (!message_batch_queue_.empty()) {
142 auto handler = on_message_batch_;
143 auto batch = std::move(message_batch_queue_);
144 message_batch_queue_.clear();
145 if (handler) {
146 lock.unlock();
147 handler(batch);
148 lock.lock();
149 }
150 }
151 if (batch_timer_) {
152 batch_timer_->cancel();
153 }
154 }

156 void schedule_batch_timer() {
157 if (!batch_timer_) return;
158 batch_timer_->expires_after(max_batch_latency_);
159 batch_timer_->async_wait([this, alive = std::weak_ptr<bool>(is_alive)](const boost::system::error_code& ec) {
160 auto lock = alive.lock();
161 if (!lock || !(*lock)) return;
162
163 if (!ec) {
164 flush_batches();
165 }
166 });

169 void schedule_reaper() {
170 if (!started.load() || !reaper_timer) return;
171
172 // Run reaper at interval proportional to timeout (min 100ms, max 5s)
173 auto interval =
174 std::max(std::chrono::milliseconds(100), std::min(std::chrono::milliseconds(5000), session_timeout / 2));
175
176 reaper_timer->expires_after(interval);
177 reaper_timer->async_wait([this, alive = std::weak_ptr<bool>(is_alive)](const boost::system::error_code& ec) {
178 auto lock = alive.lock();
179 if (!lock || !(*lock)) return;
180
181 if (!ec) {
182 run_reaper();
183 schedule_reaper();
184 }
185 });
186 }

188 void run_reaper() {
189 std::vector<std::pair<ClientId, std::string>> to_remove_with_info;
190 auto now = std::chrono::steady_clock::now();
191
192 ConnectionHandler disconnect_handler;
193 {
194 std::unique_lock<std::shared_mutex> lock(mutex);
195 for (auto it = sessions.begin(); it != sessions.end();) {
196 if (now - it->second.last_seen > session_timeout) {
197 std::string info =
198 it->second.endpoint.address().to_string() + ":" + std::to_string(it->second.endpoint.port());
199 endpoint_to_id.erase(it->second.endpoint);
200 to_remove_with_info.push_back({it->first, info});
201 it = sessions.erase(it);
202 } else {
203 ++it;
204 }
205 }
206 disconnect_handler = on_disconnect;
207 }
208
209 // Call disconnect handlers outside the lock
210 if (disconnect_handler) {
211 for (auto const& [id, info] : to_remove_with_info) {
212 disconnect_handler(ConnectionContext(id, info));
213 }
214 }
215 }

217 void setup_internal_handlers() {
218 if (!channel) return;
219
220 batch_timer_ = std::make_unique<boost::asio::steady_timer>(channel->get_executor());
221
222 channel->on_bytes_from([this](memory::ConstByteSpan data, const boost::asio::ip::udp::endpoint& ep) {
223 ClientId client_id = 0;
224 bool is_new = false;
225 ConnectionHandler connect_handler_copy{nullptr};
226
227 {
228 std::unique_lock<std::shared_mutex> lock(mutex);
229 auto it = endpoint_to_id.find(ep);
230 if (it == endpoint_to_id.end()) {
231 client_id = next_client_id++;
232 endpoint_to_id[ep] = client_id;
233 SessionEntry entry;
234 entry.endpoint = ep;
235 entry.last_seen = std::chrono::steady_clock::now();
236 is_new = true;
237
238 // Create framer for new session
239 if (framer_factory) {
240 auto framer = framer_factory();
241 if (framer) {
242 framer->on_message([this, client_id](memory::ConstByteSpan msg) {
243 std::unique_lock<std::shared_mutex> lock(mutex);
244 if (on_message_batch_) {
245 message_batch_queue_.emplace_back(client_id, memory::SafeDataBuffer(msg));
246 if (message_batch_queue_.size() >= max_batch_size_) {
247 auto handler = on_message_batch_;
248 auto batch = std::move(message_batch_queue_);
249 message_batch_queue_.clear();
250 lock.unlock();
251 handler(batch);
252 } else if (message_batch_queue_.size() == 1) {
253 schedule_batch_timer();
254 }
255 return;
256 }
257
258 MessageHandler on_message_handler = on_message;
259 if (on_message_handler) {
260 lock.unlock();
261 on_message_handler(MessageContext(client_id, memory::SafeDataBuffer(msg)));
262 }
263 });
264 entry.framer = std::move(framer);
265 }
266 }
267 sessions[client_id] = std::move(entry);
268 } else {
269 client_id = it->second;
270 sessions[client_id].last_seen = std::chrono::steady_clock::now();
271 }
272 connect_handler_copy = on_connect;
273 }
274
275 if (is_new && connect_handler_copy) {
276 connect_handler_copy(ConnectionContext(client_id, ep.address().to_string() + ":" + std::to_string(ep.port())));
277 }
278
279 {
280 std::unique_lock<std::shared_mutex> lock(mutex);
281 if (on_data_batch_) {
282 data_batch_queue_.emplace_back(client_id, memory::SafeDataBuffer(data));
283 if (data_batch_queue_.size() >= max_batch_size_) {
284 auto handler = on_data_batch_;
285 auto batch = std::move(data_batch_queue_);
286 data_batch_queue_.clear();
287 lock.unlock();
288 handler(batch);
289 } else if (data_batch_queue_.size() == 1) {
290 schedule_batch_timer();
291 }
292 } else {
293 MessageHandler data_handler_copy = on_data;
294 if (data_handler_copy) {
295 lock.unlock();
296 data_handler_copy(MessageContext(client_id, memory::SafeDataBuffer(data)));
297 lock.lock();
298 }
299 }
300 }
301
302 // Push to framer
303 std::shared_ptr<framer::IFramer> target_framer;
304 {
305 std::shared_lock<std::shared_mutex> lock(mutex);
306 auto it = sessions.find(client_id);
307 if (it != sessions.end()) {
308 target_framer = it->second.framer;
309 }
310 }
311 if (target_framer) {
312 target_framer->push_bytes(data);
313 }
314 });
315
316 channel->on_state([this](base::LinkState state) {
317 ErrorHandler error_handler_copy{nullptr};
318 if (state == base::LinkState::Listening || state == base::LinkState::Connected) {
319 is_listening.store(true);
320 std::unique_lock<std::shared_mutex> lock(mutex);
321 fulfill_all_locked(true);
322 } else if (state == base::LinkState::Error || state == base::LinkState::Closed ||
323 state == base::LinkState::Idle) {
324 is_listening.store(false);
325 std::unique_lock<std::shared_mutex> lock(mutex);
326 fulfill_all_locked(false);
327 if (state == base::LinkState::Error) {
328 error_handler_copy = on_error;
329 }
330 }
331
332 if (error_handler_copy) {
333 error_handler_copy(ErrorContext(ErrorCode::IoError, "Server error"));
334 }
335 });
336 }

338 std::future<bool> start() {
339 std::unique_lock<std::shared_mutex> lock(mutex);
340 if (is_listening.load()) {
341 std::promise<bool> p;
342 p.set_value(true);
343 return p.get_future();
344 }
345
346 std::promise<bool> p;
347 auto fut = p.get_future();
348 pending_promises.emplace_back(std::move(p));
349
350 if (started.exchange(true)) {
351 return fut;
352 }
353
354 if (!channel) {
355 channel = std::dynamic_pointer_cast<transport::UdpChannel>(factory::ChannelFactory::create(cfg, external_ioc));
356 setup_internal_handlers();
357 }
358
359 lock.unlock();
360 channel->start();
361
362 lock.lock();
363 if (use_external_context.load() && manage_external_context.load() && !external_thread.joinable()) {
364 if (external_ioc->stopped()) external_ioc->restart();
365 work_guard = std::make_unique<boost::asio::executor_work_guard<boost::asio::io_context::executor_type>>(
366 external_ioc->get_executor());
367 external_thread = std::thread([ioc = external_ioc]() {
368 try {
369 ioc->run();
370 } catch (...) {
371 }
372 });
373 }
374
375 if (channel) {
376 reaper_timer = std::make_unique<boost::asio::steady_timer>(channel->get_executor());
377 schedule_reaper();
378 }
379
380 return fut;
381 }
382
383 void stop() {
384 bool should_join = false;
385 {
386 std::unique_lock<std::shared_mutex> lock(mutex);
387 if (!started.exchange(false)) {
388 is_listening.store(false);
389 fulfill_all_locked(false);
390 return;
391 }
392
393 if (reaper_timer) {
394 reaper_timer->cancel();
395 reaper_timer.reset();
396 }
397
398 if (batch_timer_) {
399 batch_timer_->cancel();
400 batch_timer_.reset();
401 }
402
403 if (channel) {
404 channel->on_bytes_from(nullptr);
405 channel->on_state(nullptr);
406 lock.unlock();
407 channel->stop();
408 lock.lock();
409 }
410
411 if (use_external_context.load() && manage_external_context.load()) {
412 if (external_ioc) external_ioc->stop();
413 should_join = true;
414 }
415
416 is_listening.store(false);
417 endpoint_to_id.clear();
418 sessions.clear();
419 next_client_id = 1;
420 fulfill_all_locked(false);
421 }
422
423 if (should_join && external_thread.joinable()) {
424 try {
425 if (std::this_thread::get_id() != external_thread.get_id()) {
426 external_thread.join();
427 } else {
428 external_thread.detach();
429 }
430 } catch (...) {
431 }
432 }
433 std::unique_lock<std::shared_mutex> lock(mutex);
434 channel.reset();
435 }

460bool UdpServer::broadcast(std::string_view data) {
461 std::shared_lock<std::shared_mutex> lock(impl_->mutex);
462 if (!impl_->channel) return false;
463
464 auto bytes = base::safe_convert::string_to_bytes(data);
465 for (const auto& [id, entry] : impl_->sessions) {
466 impl_->channel->async_write_to(memory::ConstByteSpan(bytes.first, bytes.second), entry.endpoint);
467 }
468 return true;
469}

471bool UdpServer::send_to(ClientId client_id, std::string_view data) {
472 std::shared_lock<std::shared_mutex> lock(impl_->mutex);
473 if (!impl_->channel) return false;
474
475 auto it = impl_->sessions.find(client_id);
476 if (it == impl_->sessions.end()) return false;
477
478 auto bytes = base::safe_convert::string_to_bytes(data);
479 impl_->channel->async_write_to(memory::ConstByteSpan(bytes.first, bytes.second), it->second.endpoint);
480 return true;
481}
