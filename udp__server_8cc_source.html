77  void schedule_reaper() {
78  if (!started || !reaper_timer) return;
79 
80  // Run reaper at interval proportional to timeout (min 100ms, max 5s)
81  auto interval =
82  std::max(std::chrono::milliseconds(100), std::min(std::chrono::milliseconds(5000), session_timeout / 2));
83 
84  reaper_timer->expires_after(interval);
85  reaper_timer->async_wait([this, alive = std::weak_ptr<bool>(is_alive)](const boost::system::error_code& ec) {
86  auto lock = alive.lock();
87  if (!lock || !(*lock)) return;
88 
89  if (!ec) {
90  run_reaper();
91  schedule_reaper();
92  }
93  });
94  }
95 
96  void run_reaper() {
97  std::vector<std::pair<size_t, std::string>> to_remove_with_info;
98  auto now = std::chrono::steady_clock::now();
99 
100  {
101  std::lock_guard<std::mutex> lock(mutex);
102  for (auto const& [id, last_seen] : session_activity) {
103  if (now - last_seen > session_timeout) {
104  auto it_ep = id_to_endpoint.find(id);
105  std::string info = "timeout";
106  if (it_ep != id_to_endpoint.end()) {
107  info = it_ep->second.address().to_string() + ":" + std::to_string(it_ep->second.port());
108  endpoint_to_id.erase(it_ep->second);
109  id_to_endpoint.erase(it_ep);
110  }
111  client_framers.erase(id);
112  to_remove_with_info.push_back({id, info});
113  }
114  }
115 
116  for (auto const& [id, info] : to_remove_with_info) {
117  session_activity.erase(id);
118  }
119  }
120 
121  // Call disconnect handlers outside the lock
122  if (on_disconnect) {
123  for (auto const& [id, info] : to_remove_with_info) {
124  on_disconnect(ConnectionContext(id, info));
125  }
126  }
127  }

129  void setup_internal_handlers() {
130  if (!channel) return;
131 
132  channel->on_bytes_from([this](memory::ConstByteSpan data, const boost::asio::ip::udp::endpoint& ep) {
133  size_t client_id = 0;
134  bool is_new = false;
135 
136  {
137  std::lock_guard<std::mutex> lock(mutex);
138  auto it = endpoint_to_id.find(ep);
139  if (it == endpoint_to_id.end()) {
140  client_id = next_client_id++;
141  endpoint_to_id[ep] = client_id;
142  id_to_endpoint[client_id] = ep;
143  is_new = true;
144 
145  // Create framer for new session
146  if (framer_factory) {
147  auto framer = framer_factory();
148  if (framer) {
149  framer->set_on_message([this, client_id](memory::ConstByteSpan msg) {
150  if (on_message) {
151  std::string str_msg = common::safe_convert::uint8_to_string(msg.data(), msg.size());
152  on_message(MessageContext(client_id, str_msg));
153  }
154  });
155  client_framers[client_id] = std::shared_ptr<framer::IFramer>(std::move(framer));
156  }
157  }
158  } else {
159  client_id = it->second;
160  }
161  session_activity[client_id] = std::chrono::steady_clock::now();
162  }
163 
164  if (is_new && on_connect) {
165  on_connect(ConnectionContext(client_id, ep.address().to_string() + ":" + std::to_string(ep.port())));
166  }
167 
168  if (on_data) {
169  std::string str_data = common::safe_convert::uint8_to_string(data.data(), data.size());
170  on_data(MessageContext(client_id, str_data));
171  }
172 
173  // Push to framer
174  std::shared_ptr<framer::IFramer> target_framer;
175  {
176  std::lock_guard<std::mutex> lock(mutex);
177  auto it = client_framers.find(client_id);
178  if (it != client_framers.end()) {
179  target_framer = it->second;
180  }
181  }
182  if (target_framer) {
183  target_framer->push_bytes(data);
184  }
185  });
186 
187  channel->on_state([this](base::LinkState state) {
188  ErrorHandler error_handler_copy{nullptr};
189  if (state == base::LinkState::Listening || state == base::LinkState::Connected) {
190  std::lock_guard<std::mutex> lock(mutex);
191  if (start_promise) {
192  try {
193  start_promise->set_value(true);
194  } catch (...) {
195  }
196  start_promise.reset();
197  }
198  } else if (state == base::LinkState::Error || state == base::LinkState::Closed) {
199  std::lock_guard<std::mutex> lock(mutex);
200  if (start_promise) {
201  try {
202  start_promise->set_value(false);
203  } catch (...) {
204  }
205  start_promise.reset();
206  }
207  if (state == base::LinkState::Error) {
208  error_handler_copy = on_error;
209  }
210  }
211 
212  if (error_handler_copy) {
213  error_handler_copy(ErrorContext(ErrorCode::IoError, "UDP Transport Error"));
214  }
215  });
216  }

218  std::future<bool> start() {
219  std::future<bool> fut;
220  {
221  std::lock_guard<std::mutex> lock(mutex);
222  if (started) {
223  std::promise<bool> p;
224  p.set_value(true);
225  return p.get_future();
226  }
227 
228  start_promise = std::make_shared<std::promise<bool>>();
229  fut = start_promise->get_future();
230 
231  if (!channel) {
232  channel = std::dynamic_pointer_cast<transport::UdpChannel>(factory::ChannelFactory::create(cfg, external_ioc));
233  setup_internal_handlers();
234  }
235  }
236 
237  channel->start();
238 
239  {
240  std::lock_guard<std::mutex> lock(mutex);
241  if (use_external_context && manage_external_context && !external_thread.joinable()) {
242  if (external_ioc->stopped()) external_ioc->restart();
243  work_guard = std::make_unique<boost::asio::executor_work_guard<boost::asio::io_context::executor_type>>(
244  external_ioc->get_executor());
245  external_thread = std::thread([ioc = external_ioc]() {
246  try {
247  ioc->run();
248  } catch (...) {
249  }
250  });
251  }
252 
253  started = true;
254 
255  // Start Reaper Timer
256  if (channel) {
257  reaper_timer = std::make_unique<boost::asio::steady_timer>(channel->get_executor());
258  schedule_reaper();
259  }
260  }
261 
262  return fut;
263  }

265  void stop() {
266  bool should_join = false;
267  {
268  std::lock_guard<std::mutex> lock(mutex);
269  if (!started) return;
270 
271  if (channel) {
272  channel->on_bytes_from(nullptr);
273  channel->on_state(nullptr);
274  channel->stop();
275  }
276 
277  if (reaper_timer) {
278  reaper_timer->cancel();
279  reaper_timer.reset();
280  }
281 
282  if (use_external_context && manage_external_context && external_thread.joinable()) {
283  if (external_ioc) external_ioc->stop();
284  should_join = true;
285  }
286 
287  started = false;
288  endpoint_to_id.clear();
289  id_to_endpoint.clear();
290  client_framers.clear();
291  session_activity.clear();
292  next_client_id = 1;
293  if (start_promise) {
294  try {
295  start_promise->set_value(false);
296  } catch (...) {
297  }
298  start_promise.reset();
299  }
300  }
301 
302  if (should_join) {
303  try {
304  external_thread.join();
305  } catch (...) {
306  }
307  }
308  }

333 bool UdpServer::broadcast(std::string_view data) {
334  std::lock_guard<std::mutex> lock(impl_->mutex);
335  if (!impl_->channel) return false;
336 
337  auto bytes = common::safe_convert::string_to_bytes(data);
338  for (const auto& pair : impl_->id_to_endpoint) {
339  impl_->channel->async_write_to(memory::ConstByteSpan(bytes.first, bytes.second), pair.second);
340  }
341  return true;
