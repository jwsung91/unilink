160 void TcpClient::start() {
161  auto current_state = impl_->state_.get_state();
162  if (current_state == LinkState::Connecting || current_state == LinkState::Connected) {
163  UNILINK_LOG_DEBUG("tcp_client", "start", "Start called while already active, ignoring");
164  return;
165  }
166 
167  if (!impl_->ioc_) {
168  UNILINK_LOG_ERROR("tcp_client", "start", "io_context is null");
169  }
170 
171  impl_->recalculate_backpressure_bounds();
172 
173  if (impl_->ioc_ && impl_->ioc_->stopped()) {
174  UNILINK_LOG_DEBUG("tcp_client", "start", "io_context stopped; restarting before start");
175  impl_->ioc_->restart();
176  }
177 
178  if (impl_->ioc_thread_.joinable()) {
179  impl_->join_ioc_thread(false);
180  }
181 
182  const auto seq = impl_->lifecycle_seq_.fetch_add(1) + 1;
183  impl_->current_seq_.store(seq);
184 
185  if (impl_->owns_ioc_ && impl_->ioc_) {
186  impl_->work_guard_ =
187  std::make_unique<net::executor_work_guard<net::io_context::executor_type>>(impl_->ioc_->get_executor());
188  impl_->ioc_thread_ = std::thread([this]() {
189  try {
190  impl_->ioc_->run();
191  } catch (const std::exception& e) {
192  UNILINK_LOG_ERROR("tcp_client", "io_context", "IO context error: " + std::string(e.what()));
193  diagnostics::error_reporting::report_system_error("tcp_client", "io_context",
194  "Exception in IO context: " + std::string(e.what()));
195  }
196  });
197  }
198 
199  auto weak_self = weak_from_this();
200  if (impl_->ioc_) {
201  net::dispatch(impl_->strand_, [weak_self, seq] {
202  if (auto self = weak_self.lock()) {
203  if (seq <= self->impl_->stop_seq_.load()) {
204  return;
205  }
206  self->impl_->reset_start_state();
207  self->impl_->connected_.store(false);
208  self->impl_->reset_io_objects();
209  self->impl_->transition_to(LinkState::Connecting);
210  self->impl_->do_resolve_connect(self, seq);
211  }
212  });
213  } else {
214  UNILINK_LOG_ERROR("tcp_client", "start", "io_context is null");
215  }
216 }

218 void TcpClient::stop() {
219  if (impl_->stop_requested_.exchange(true)) {
220  return;
221  }
222 
223  impl_->stopping_.store(true);
224  impl_->stop_seq_.store(impl_->current_seq_.load());
225  auto weak_self = weak_from_this();
226  if (!impl_->ioc_) {
227  return;
228  }
229 
230  if (auto self = weak_self.lock()) {
231  net::post(impl_->strand_, [self]() { self->impl_->perform_stop_cleanup(); });
232  }
233 
234  impl_->join_ioc_thread(false);
235 }

239 void TcpClient::async_write_copy(memory::ConstByteSpan data) {
240  if (impl_->stop_requested_.load() || impl_->state_.is_state(LinkState::Closed) ||
241  impl_->state_.is_state(LinkState::Error) || !impl_->ioc_) {
242  return;
243  }
244 
245  size_t size = data.size();
246  if (size == 0) {
247  UNILINK_LOG_WARNING("tcp_client", "async_write_copy", "Ignoring zero-length write");
248  return;
249  }
250 
251  if (size > common::constants::MAX_BUFFER_SIZE) {
252  UNILINK_LOG_ERROR("tcp_client", "async_write_copy",
253  "Write size exceeds maximum allowed (" + std::to_string(size) + " bytes)");
254  return;
255  }
256 
257  if (size <= 65536) {
258  try {
259  memory::PooledBuffer pooled_buffer(size);
260  if (pooled_buffer.valid()) {
261  common::safe_memory::safe_memcpy(pooled_buffer.data(), data.data(), size);
262  const auto added = pooled_buffer.size();
263  net::dispatch(impl_->strand_, [self = shared_from_this(), buf = std::move(pooled_buffer), added]() mutable {
264  if (self->impl_->stop_requested_.load() || self->impl_->state_.is_state(LinkState::Closed) ||
265  self->impl_->state_.is_state(LinkState::Error)) {
266  return;
267  }
268 
269  if (self->impl_->queue_bytes_ + added > self->impl_->bp_limit_) {
270  UNILINK_LOG_ERROR("tcp_client", "async_write_copy",
271  "Queue limit exceeded (" + std::to_string(self->impl_->queue_bytes_ + added) + " bytes)");
272  self->impl_->connected_.store(false);
273  self->impl_->close_socket();
274  self->impl_->tx_.clear();
275  self->impl_->queue_bytes_ = 0;
276  self->impl_->writing_ = false;
277  self->impl_->backpressure_active_ = false;
278  self->impl_->transition_to(LinkState::Error);
279  return;
280  }
281 
282  self->impl_->queue_bytes_ += added;
283  self->impl_->tx_.emplace_back(std::move(buf));
284  self->impl_->report_backpressure(self->impl_->queue_bytes_);
285  if (!self->impl_->writing_) self->impl_->do_write(self, self->impl_->current_seq_.load());
286  });
287  return;
288  }
289  } catch (const std::exception& e) {
290  UNILINK_LOG_ERROR("tcp_client", "async_write_copy", "Failed to acquire pooled buffer: " + std::string(e.what()));
291  }
292  }
293 
294  std::vector<uint8_t> fallback(data.begin(), data.end());
295  const auto added = fallback.size();
296 
297  net::dispatch(impl_->strand_, [self = shared_from_this(), buf = std::move(fallback), added]() mutable {
298  if (self->impl_->stop_requested_.load() || self->impl_->state_.is_state(LinkState::Closed) ||
299  self->impl_->state_.is_state(LinkState::Error)) {
300  return;
301  }
302 
303  if (self->impl_->queue_bytes_ + added > self->impl_->bp_limit_) {
304  UNILINK_LOG_ERROR("tcp_client", "async_write_copy",
305  "Queue limit exceeded (" + std::to_string(self->impl_->queue_bytes_ + added) + " bytes)");
306  self->impl_->connected_.store(false);
307  self->impl_->close_socket();
308  self->impl_->tx_.clear();
309  self->impl_->queue_bytes_ = 0;
310  self->impl_->writing_ = false;
311  self->impl_->backpressure_active_ = false;
312  self->impl_->transition_to(LinkState::Error);
313  return;
314  }
315 
316  self->impl_->queue_bytes_ += added;
317  self->impl_->tx_.emplace_back(std::move(buf));
318  self->impl_->report_backpressure(self->impl_->queue_bytes_);
319  if (!self->impl_->writing_) self->impl_->do_write(self, self->impl_->current_seq_.load());
320  });
321 }

416 void TcpClient::Impl::do_resolve_connect(std::shared_ptr<TcpClient> self, uint64_t seq) {
417  resolver_.async_resolve(
418  cfg_.host, std::to_string(cfg_.port), [self, seq](auto ec, tcp::resolver::results_type results) {
419  if (ec == net::error::operation_aborted || seq != self->impl_->current_seq_.load()) {
420  return;
421  }
422  if (self->impl_->stop_requested_.load() || self->impl_->stopping_.load()) {
423  return;
424  }
425  if (ec) {
426  self->impl_->schedule_retry(self, seq);
427  return;
428  }
429  self->impl_->connect_timer_.expires_after(std::chrono::milliseconds(self->impl_->cfg_.connection_timeout_ms));
430  self->impl_->connect_timer_.async_wait([self, seq](const boost::system::error_code& timer_ec) {
431  if (timer_ec == net::error::operation_aborted || seq != self->impl_->current_seq_.load()) {
432  return;
433  }
434  if (!timer_ec && !self->impl_->stop_requested_.load() && !self->impl_->stopping_.load()) {
435  UNILINK_LOG_ERROR(
436  "tcp_client", "connect_timeout",
437  "Connection timed out after " + std::to_string(self->impl_->cfg_.connection_timeout_ms) + "ms");
438  self->impl_->handle_close(self, seq, boost::asio::error::timed_out);
439  }
440  });
441 
442  net::async_connect(self->impl_->socket_, results, [self, seq](auto ec2, const auto&) {
443  if (ec2 == net::error::operation_aborted || seq != self->impl_->current_seq_.load()) {
444  return;
445  }
446  if (self->impl_->stop_requested_.load() || self->impl_->stopping_.load()) {
447  self->impl_->close_socket();
448  self->impl_->connect_timer_.cancel();
449  return;
450  }
451  if (ec2) {
452  self->impl_->connect_timer_.cancel();
453  self->impl_->schedule_retry(self, seq);
454  return;
455  }
456  self->impl_->connect_timer_.cancel();
457  self->impl_->retry_attempts_ = 0;
458  self->impl_->connected_.store(true);
459 
460 #if defined(__APPLE__) || defined(__FreeBSD__) || defined(__OpenBSD__) || defined(__NetBSD__)
461  int yes = 1;
462  (void)::setsockopt(self->impl_->socket_.native_handle(), SOL_SOCKET, SO_NOSIGPIPE, &yes, sizeof(yes));
463 #endif
464 
465  self->impl_->transition_to(LinkState::Connected);
466  boost::system::error_code ep_ec;
467  auto rep = self->impl_->socket_.remote_endpoint(ep_ec);
468  if (!ep_ec) {
469  UNILINK_LOG_INFO("tcp_client", "connect",
470  "Connected to " + rep.address().to_string() + ":" + std::to_string(rep.port()));
471  }
472  self->impl_->start_read(self, seq);
473  self->impl_->do_write(self, seq);
474  });
475  });
476 }

478 void TcpClient::Impl::schedule_retry(std::shared_ptr<TcpClient> self, uint64_t seq) {
479  connected_.store(false);
480  if (stop_requested_.load() || stopping_.load()) {
481  return;
482  }
483  if (cfg_.max_retries != -1 && retry_attempts_ >= cfg_.max_retries) {
484  transition_to(LinkState::Error);
485  return;
486  }
487  ++retry_attempts_;
488  transition_to(LinkState::Connecting);
489 
490  UNILINK_LOG_INFO(
491  "tcp_client", "retry",
492  "Scheduling retry in " +
493  std::to_string((retry_attempts_ == 1 ? first_retry_interval_ms_ : cfg_.retry_interval_ms) / 1000.0) + "s");
494 
495  const unsigned interval = retry_attempts_ == 1 ? first_retry_interval_ms_ : cfg_.retry_interval_ms;
496  retry_timer_.expires_after(std::chrono::milliseconds(interval));
497  retry_timer_.async_wait([self, seq](const boost::system::error_code& ec) {
498  if (ec == net::error::operation_aborted || seq != self->impl_->current_seq_.load()) {
499  return;
500  }
501  if (!ec && !self->impl_->stop_requested_.load() && !self->impl_->stopping_.load())
502  self->impl_->do_resolve_connect(self, seq);
503  });
504 }

506 void TcpClient::Impl::start_read(std::shared_ptr<TcpClient> self, uint64_t seq) {
507  socket_.async_read_some(net::buffer(rx_.data(), rx_.size()), [self, seq](auto ec, std::size_t n) {
508  if (ec == net::error::operation_aborted || seq != self->impl_->current_seq_.load()) {
509  return;
510  }
511  if (self->impl_->stop_requested_.load()) {
512  return;
513  }
514  if (ec) {
515  self->impl_->handle_close(self, seq, ec);
516  return;
517  }
518  if (self->impl_->on_bytes_) {
519  try {
520  self->impl_->on_bytes_(memory::ConstByteSpan(self->impl_->rx_.data(), n));
521  } catch (const std::exception& e) {
522  UNILINK_LOG_ERROR("tcp_client", "on_bytes", "Exception in on_bytes callback: " + std::string(e.what()));
523  self->impl_->handle_close(self, seq, make_error_code(boost::asio::error::connection_aborted));
524  return;
525  } catch (...) {
526  UNILINK_LOG_ERROR("tcp_client", "on_bytes", "Unknown exception in on_bytes callback");
527  self->impl_->handle_close(self, seq, make_error_code(boost::asio::error::connection_aborted));
528  return;
529  }
530  }
531  self->impl_->start_read(self, seq);
532  });
533 }

535 void TcpClient::Impl::do_write(std::shared_ptr<TcpClient> self, uint64_t seq) {
536  if (stop_requested_.load()) {
537  tx_.clear();
538  queue_bytes_ = 0;
539  writing_ = false;
540  report_backpressure(queue_bytes_);
541  return;
542  }
543 
544  if (!connected_.load()) {
545  writing_ = false;
546  return;
547  }
548 
549  if (tx_.empty() || state_.is_state(LinkState::Closed) || state_.is_state(LinkState::Error)) {
550  writing_ = false;
551  return;
552  }
553  writing_ = true;
554 
555  current_write_buffer_ = std::move(tx_.front());
556  tx_.pop_front();
557 
558  auto& current = *current_write_buffer_;
559 
560  auto queued_bytes = std::visit(
561  [](auto&& buf) -> size_t {
562  using Buffer = std::decay_t<decltype(buf)>;
563  if constexpr (std::is_same_v<Buffer, std::shared_ptr<const std::vector<uint8_t>>>) {
564  return buf ? buf->size() : 0;
565  } else {
566  return buf.size();
567  }
568  },
569  current);
570 
571  auto on_write = [self, queued_bytes, seq](auto ec, std::size_t) {
572  if (ec == net::error::operation_aborted || seq != self->impl_->current_seq_.load()) {
573  self->impl_->current_write_buffer_.reset();
574  self->impl_->queue_bytes_ =
575  (self->impl_->queue_bytes_ > queued_bytes) ? (self->impl_->queue_bytes_ - queued_bytes) : 0;
576  self->impl_->report_backpressure(self->impl_->queue_bytes_);
577  self->impl_->writing_ = false;
578  return;
579  }
580 
581  if (ec) {

583  self->impl_->tx_.push_front(std::move(*self->impl_->current_write_buffer_));
584  }
585  self->impl_->current_write_buffer_.reset();
586 
587  UNILINK_LOG_ERROR("tcp_client", "do_write", "Write failed: " + ec.message());
588  self->impl_->writing_ = false;
589  self->impl_->handle_close(self, seq, ec);
590  return;
591  }
592 
593  self->impl_->current_write_buffer_.reset();
594  self->impl_->queue_bytes_ =
595  (self->impl_->queue_bytes_ > queued_bytes) ? (self->impl_->queue_bytes_ - queued_bytes) : 0;
596  self->impl_->report_backpressure(self->impl_->queue_bytes_);
597 
598  if (self->impl_->stop_requested_.load() || self->impl_->state_.is_state(LinkState::Closed) ||
599  self->impl_->state_.is_state(LinkState::Error)) {
600  self->impl_->writing_ = false;
601  return;
602  }
603 
604  self->impl_->do_write(self, seq);
605  };

612  } else {

683  if (ec == net::error::operation_aborted) {
684  return;
685  }
686 
687  const auto current = state_.get_state();
688  const bool retrying_same_state = (next == LinkState::Connecting && current == LinkState::Connecting);
689  if ((current == LinkState::Closed || current == LinkState::Error) &&
690  (next == LinkState::Closed || next == LinkState::Error)) {
691  return;
692  }
693 
694  if (next == LinkState::Closed || next == LinkState::Error) {
695  if (terminal_state_notified_.exchange(true)) {
696  return;
697  }
698  } else if (current == next && !retrying_same_state) {
699  return;
700  }
701 
702  state_.set_state(next);
703  notify_state();
