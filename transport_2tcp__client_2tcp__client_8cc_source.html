179void TcpClient::start() {
180 auto current_state = impl_->state_.state();
181 if (current_state == LinkState::Connecting || current_state == LinkState::Connected) {
182 UNILINK_LOG_DEBUG("tcp_client", "start", "Start called while already active, ignoring");
183 return;
184 }
185
186 if (!impl_->ioc_) {
187 UNILINK_LOG_ERROR("tcp_client", "start", "io_context is null");
188 }
189
190 impl_->recalculate_backpressure_bounds();
191
192 if (impl_->ioc_ && impl_->ioc_->stopped()) {
193 UNILINK_LOG_DEBUG("tcp_client", "start", "io_context stopped; restarting before start");
194 impl_->ioc_->restart();
195 }
196
197 if (impl_->ioc_thread_.joinable()) {
198 impl_->join_ioc_thread(false);
199 }
200
201 const auto seq = impl_->lifecycle_seq_.fetch_add(1) + 1;
202 impl_->current_seq_.store(seq);
203
204 if (impl_->owns_ioc_ && impl_->ioc_) {
205 impl_->work_guard_ =
206 std::make_unique<net::executor_work_guard<net::io_context::executor_type>>(impl_->ioc_->get_executor());
207 impl_->ioc_thread_ = std::thread([this]() {
208 try {
209 impl_->ioc_->run();
210 } catch (const std::exception& e) {
211 UNILINK_LOG_ERROR("tcp_client", "io_context", "IO context error: " + std::string(e.what()));
212 diagnostics::error_reporting::report_system_error("tcp_client", "io_context",
213 "Exception in IO context: " + std::string(e.what()));
214 }
215 });
216 }
217
218 auto weak_self = weak_from_this();
219 if (impl_->ioc_) {
220 net::dispatch(impl_->strand_, [weak_self, seq] {
221 if (auto self = weak_self.lock()) {
222 if (seq <= self->impl_->stop_seq_.load()) {
223 return;
224 }
225 self->impl_->reset_start_state();
226 self->impl_->connected_.store(false);
227 self->impl_->reset_io_objects();
228 self->impl_->transition_to(LinkState::Connecting);
229 self->impl_->do_resolve_connect(self, seq);
230 }
231 });
232 } else {
233 UNILINK_LOG_ERROR("tcp_client", "start", "io_context is null");
234 }
235}

237void TcpClient::stop() {
238 if (impl_->stop_requested_.exchange(true)) {
239 return;
240 }
241
242 impl_->stopping_.store(true);
243 impl_->stop_seq_.store(impl_->current_seq_.load());
244 auto weak_self = weak_from_this();
245 if (!impl_->ioc_) {
246 return;
247 }
248
249 if (auto self = weak_self.lock()) {
250 net::post(impl_->strand_, [self]() { self->impl_->perform_stop_cleanup(); });
251 }
252
253 impl_->join_ioc_thread(false);
254}

258void TcpClient::async_write_copy(memory::ConstByteSpan data) {
259 if (impl_->stop_requested_.load() || impl_->state_.is_state(LinkState::Closed) ||
260 impl_->state_.is_state(LinkState::Error) || !impl_->ioc_) {
261 return;
262 }
263
264 size_t size = data.size();
265 if (size == 0) {
266 UNILINK_LOG_WARNING("tcp_client", "async_write_copy", "Ignoring zero-length write");
267 return;
268 }
269
270 if (size > base::constants::MAX_BUFFER_SIZE) {
271 UNILINK_LOG_ERROR("tcp_client", "async_write_copy",
272 "Write size exceeds maximum allowed (" + std::to_string(size) + " bytes)");
273 return;
274 }
275
276 if (size <= 65536) {
277 try {
278 memory::PooledBuffer pooled_buffer(size);
279 if (pooled_buffer.valid()) {
280 base::safe_memory::safe_memcpy(pooled_buffer.data(), data.data(), size);
281 const auto added = pooled_buffer.size();
282 net::dispatch(impl_->strand_, [self = shared_from_this(), buf = std::move(pooled_buffer), added]() mutable {
283 if (self->impl_->stop_requested_.load() || self->impl_->state_.is_state(LinkState::Closed) ||
284 self->impl_->state_.is_state(LinkState::Error)) {
285 return;
286 }
287
288 if (self->impl_->queue_bytes_ + added > self->impl_->bp_limit_) {
289 UNILINK_LOG_ERROR("tcp_client", "async_write_copy",
290 "Queue limit exceeded (" + std::to_string(self->impl_->queue_bytes_ + added) + " bytes)");
291 self->impl_->record_error(diagnostics::ErrorLevel::ERROR, diagnostics::ErrorCategory::COMMUNICATION,
292 "async_write_copy", boost::asio::error::no_buffer_space, "Queue limit exceeded",
293 false, 0);
294 self->impl_->connected_.store(false);
295 self->impl_->close_socket();
296 self->impl_->tx_.clear();
297 self->impl_->queue_bytes_ = 0;
298 self->impl_->writing_ = false;
299 self->impl_->backpressure_active_ = false;
300 self->impl_->transition_to(LinkState::Error);
301 return;
302 }
303
304 self->impl_->queue_bytes_ += added;
305 self->impl_->tx_.emplace_back(std::move(buf));
306 self->impl_->report_backpressure(self->impl_->queue_bytes_);
307 if (!self->impl_->writing_) self->impl_->do_write(self, self->impl_->current_seq_.load());
308 });
309 return;
310 }
311 } catch (const std::exception& e) {
312 UNILINK_LOG_ERROR("tcp_client", "async_write_copy", "Failed to acquire pooled buffer: " + std::string(e.what()));
313 }
314 }
315
316 std::vector<uint8_t> fallback(data.begin(), data.end());
317 const auto added = fallback.size();
318
319 net::dispatch(impl_->strand_, [self = shared_from_this(), buf = std::move(fallback), added]() mutable {
320 if (self->impl_->stop_requested_.load() || self->impl_->state_.is_state(LinkState::Closed) ||
321 self->impl_->state_.is_state(LinkState::Error)) {
322 return;
323 }
324
325 if (self->impl_->queue_bytes_ + added > self->impl_->bp_limit_) {
326 UNILINK_LOG_ERROR("tcp_client", "async_write_copy",
327 "Queue limit exceeded (" + std::to_string(self->impl_->queue_bytes_ + added) + " bytes)");
328 self->impl_->record_error(diagnostics::ErrorLevel::ERROR, diagnostics::ErrorCategory::COMMUNICATION,
329 "async_write_copy", boost::asio::error::no_buffer_space, "Queue limit exceeded", false,
330 0);
331 self->impl_->connected_.store(false);
332 self->impl_->close_socket();
333 self->impl_->tx_.clear();
334 self->impl_->queue_bytes_ = 0;
335 self->impl_->writing_ = false;
336 self->impl_->backpressure_active_ = false;
337 self->impl_->transition_to(LinkState::Error);
338 return;
339 }
340
341 self->impl_->queue_bytes_ += added;
342 self->impl_->tx_.emplace_back(std::move(buf));
343 self->impl_->report_backpressure(self->impl_->queue_bytes_);
344 if (!self->impl_->writing_) self->impl_->do_write(self, self->impl_->current_seq_.load());
345 });
346}

463void TcpClient::Impl::do_resolve_connect(std::shared_ptr<TcpClient> self, uint64_t seq) {
464 resolver_.async_resolve(
465 cfg_.host, std::to_string(cfg_.port), [self, seq](auto ec, tcp::resolver::results_type results) {
466 if (ec == net::error::operation_aborted || seq != self->impl_->current_seq_.load()) {
467 return;
468 }
469 if (self->impl_->stop_requested_.load() || self->impl_->stopping_.load()) {
470 return;
471 }
472 if (ec) {
473 uint32_t current_attempts = self->impl_->reconnect_policy_
474 ? self->impl_->reconnect_attempt_count_
475 : static_cast<uint32_t>(self->impl_->retry_attempts_);
476 self->impl_->record_error(diagnostics::ErrorLevel::ERROR, diagnostics::ErrorCategory::CONNECTION, "resolve",
477 ec, "Resolution failed: " + ec.message(),
478 diagnostics::is_retryable_tcp_connect_error(ec), current_attempts);
479 self->impl_->schedule_retry(self, seq);
480 return;
481 }
482 self->impl_->connect_timer_.expires_after(std::chrono::milliseconds(self->impl_->cfg_.connection_timeout_ms));
483 self->impl_->connect_timer_.async_wait([self, seq](const boost::system::error_code& timer_ec) {
484 if (timer_ec == net::error::operation_aborted || seq != self->impl_->current_seq_.load()) {
485 return;
486 }
487 if (!timer_ec && !self->impl_->stop_requested_.load() && !self->impl_->stopping_.load()) {
488 UNILINK_LOG_ERROR(
489 "tcp_client", "connect_timeout",
490 "Connection timed out after " + std::to_string(self->impl_->cfg_.connection_timeout_ms) + "ms");
491 uint32_t current_attempts = self->impl_->reconnect_policy_
492 ? self->impl_->reconnect_attempt_count_
493 : static_cast<uint32_t>(self->impl_->retry_attempts_);
494 self->impl_->record_error(diagnostics::ErrorLevel::ERROR, diagnostics::ErrorCategory::CONNECTION, "connect",
495 boost::asio::error::timed_out, "Connection timed out",
496 diagnostics::is_retryable_tcp_connect_error(boost::asio::error::timed_out),
497 current_attempts);
498 self->impl_->handle_close(self, seq, boost::asio::error::timed_out);
499 }
500 });
501
502 net::async_connect(self->impl_->socket_, results, [self, seq](auto ec2, const auto&) {
503 if (ec2 == net::error::operation_aborted || seq != self->impl_->current_seq_.load()) {
504 return;
505 }
506 if (self->impl_->stop_requested_.load() || self->impl_->stopping_.load()) {
507 self->impl_->close_socket();
508 self->impl_->connect_timer_.cancel();
509 return;
510 }
511 if (ec2) {
512 self->impl_->connect_timer_.cancel();
513 uint32_t current_attempts = self->impl_->reconnect_policy_
514 ? self->impl_->reconnect_attempt_count_
515 : static_cast<uint32_t>(self->impl_->retry_attempts_);
516 self->impl_->record_error(diagnostics::ErrorLevel::ERROR, diagnostics::ErrorCategory::CONNECTION, "connect",
517 ec2, "Connection failed: " + ec2.message(),
518 diagnostics::is_retryable_tcp_connect_error(ec2), current_attempts);
519 self->impl_->schedule_retry(self, seq);
520 return;
521 }
522 self->impl_->connect_timer_.cancel();
523 self->impl_->retry_attempts_ = 0;
524 self->impl_->reconnect_attempt_count_ = 0;
525 self->impl_->connected_.store(true);
526
527#if defined(__APPLE__) || defined(__FreeBSD__) || defined(__OpenBSD__) || defined(__NetBSD__)
528 int yes = 1;
529 (void)::setsockopt(self->impl_->socket_.native_handle(), SOL_SOCKET, SO_NOSIGPIPE, &yes, sizeof(yes));
530#endif
531
532 self->impl_->transition_to(LinkState::Connected);
533 boost::system::error_code ep_ec;
534 auto rep = self->impl_->socket_.remote_endpoint(ep_ec);
535 if (!ep_ec) {
536 UNILINK_LOG_INFO("tcp_client", "connect",
537 "Connected to " + rep.address().to_string() + ":" + std::to_string(rep.port()));
538 }
539 self->impl_->start_read(self, seq);
540 self->impl_->do_write(self, seq);
541 });
542 });
543}

545void TcpClient::Impl::schedule_retry(std::shared_ptr<TcpClient> self, uint64_t seq) {
546 connected_.store(false);
547 if (stop_requested_.load() || stopping_.load()) {
548 return;
549 }
550
551 // Prevent double scheduling of reconnect
552 if (reconnect_pending_.exchange(true)) {
553 return;
554 }
555
556 std::optional<diagnostics::ErrorInfo> last_err;
557 {
558 std::lock_guard<std::mutex> lock(last_err_mtx_);
559 last_err = last_error_info_;
560 }
561
562 if (!last_err) {
563 last_err = diagnostics::ErrorInfo(diagnostics::ErrorLevel::ERROR, diagnostics::ErrorCategory::CONNECTION,
564 "tcp_client", "schedule_retry", "Unknown error",
565 make_error_code(boost::asio::error::not_connected), true);
566 }
567
568 // Determine current attempt count based on active mode
569 uint32_t current_attempts = reconnect_policy_ ? reconnect_attempt_count_ : static_cast<uint32_t>(retry_attempts_);
570
571 auto decision = detail::decide_reconnect(cfg_, *last_err, current_attempts, reconnect_policy_);
572
573 if (!decision.should_retry) {
574 UNILINK_LOG_INFO("tcp_client", "retry", "Reconnect stopped by policy/config");
575 transition_to(LinkState::Error);
576 reconnect_pending_.store(false);
577 return;
578 }
579
580 // The decider returns a base delay for both policy and fallback paths.
581 std::chrono::milliseconds delay = decision.delay.value_or(std::chrono::milliseconds(cfg_.retry_interval_ms));
582 if (reconnect_policy_) {
583 reconnect_attempt_count_++;
584 } else {
585 // Preserve existing "fast first retry" behavior for non-policy mode.
586 ++retry_attempts_;
587 if (retry_attempts_ == 1) {
588 delay = std::chrono::milliseconds(first_retry_interval_ms_);
589 }
590 }
591
592 transition_to(LinkState::Connecting);
593
594 UNILINK_LOG_INFO("tcp_client", "retry",
595 "Scheduling retry in " + std::to_string(static_cast<double>(delay.count()) / 1000.0) + "s");
596
597 retry_timer_.expires_after(delay);
598 retry_timer_.async_wait([self, seq](const boost::system::error_code& ec) {
599 // Clear pending flag regardless of result (fired or aborted)
600 self->impl_->reconnect_pending_.store(false);
601
602 if (ec == net::error::operation_aborted || seq != self->impl_->current_seq_.load()) {
603 return;
604 }
605 if (!ec && !self->impl_->stop_requested_.load() && !self->impl_->stopping_.load())
606 self->impl_->do_resolve_connect(self, seq);
607 });
608}

648void TcpClient::Impl::do_write(std::shared_ptr<TcpClient> self, uint64_t seq) {
649 if (stop_requested_.load()) {
650 tx_.clear();
651 queue_bytes_ = 0;
652 writing_ = false;
653 report_backpressure(queue_bytes_);
654 return;
655 }
656
657 if (!connected_.load()) {
658 writing_ = false;
659 return;
660 }
661
662 if (tx_.empty() || state_.is_state(LinkState::Closed) || state_.is_state(LinkState::Error)) {
663 writing_ = false;
664 return;
665 }
666 writing_ = true;
667
668 current_write_buffer_ = std::move(tx_.front());
669 tx_.pop_front();
670
671 auto& current = *current_write_buffer_;
672
673 auto queued_bytes = std::visit(
674 [](auto&& buf) -> size_t {
675 using Buffer = std::decay_t<decltype(buf)>;
676 if constexpr (std::is_same_v<Buffer, std::shared_ptr<const std::vector<uint8_t>>>) {
677 return buf ? buf->size() : 0;
678 } else {
679 return buf.size();
680 }
681 },
682 current);
683
684 auto on_write = [self, queued_bytes, seq](auto ec, std::size_t) {
685 if (ec == net::error::operation_aborted || seq != self->impl_->current_seq_.load()) {
686 self->impl_->current_write_buffer_.reset();
687 self->impl_->queue_bytes_ =
688 (self->impl_->queue_bytes_ > queued_bytes) ? (self->impl_->queue_bytes_ - queued_bytes) : 0;
689 self->impl_->report_backpressure(self->impl_->queue_bytes_);
690 self->impl_->writing_ = false;
691 return;
692 }
693
694 if (ec) {
695 if (self->impl_->current_write_buffer_) {
696 self->impl_->tx_.push_front(std::move(*self->impl_->current_write_buffer_));
697 }
698 self->impl_->current_write_buffer_.reset();
699
700 UNILINK_LOG_ERROR("tcp_client", "do_write", "Write failed: " + ec.message());
701 self->impl_->record_error(diagnostics::ErrorLevel::ERROR, diagnostics::ErrorCategory::COMMUNICATION, "write", ec,
702 "Write failed: " + ec.message(), false, 0);
703 self->impl_->writing_ = false;
704 self->impl_->handle_close(self, seq, ec);
705 return;
706 }
707
708 self->impl_->current_write_buffer_.reset();
709 self->impl_->queue_bytes_ =
710 (self->impl_->queue_bytes_ > queued_bytes) ? (self->impl_->queue_bytes_ - queued_bytes) : 0;
711 self->impl_->report_backpressure(self->impl_->queue_bytes_);
712
713 if (self->impl_->stop_requested_.load() || self->impl_->state_.is_state(LinkState::Closed) ||
714 self->impl_->state_.is_state(LinkState::Error)) {
715 self->impl_->writing_ = false;
716 return;
717 }
718
719 self->impl_->do_write(self, seq);
720 };
721
722 std::visit(
723 [&](const auto& buf) {
724 using T = std::decay_t<decltype(buf)>;
725 if constexpr (std::is_same_v<T, std::shared_ptr<const std::vector<uint8_t>>>) {
726 net::async_write(socket_, net::buffer(buf->data(), buf->size()), on_write);
727 } else {
728 net::async_write(socket_, net::buffer(buf.data(), buf.size()), on_write);
729 }
730 },
731 current);
732}

734void TcpClient::Impl::handle_close(std::shared_ptr<TcpClient> self, uint64_t seq, const boost::system::error_code& ec) {
735 if (ec == net::error::operation_aborted || seq != current_seq_.load()) {
736 return;
737 }
738 UNILINK_LOG_INFO("tcp_client", "handle_close", "Closing connection. Error: " + ec.message());
739 if (ec) {
740 const bool retryable = diagnostics::is_retryable_tcp_connect_error(ec);
741 const uint32_t current_attempts =
742 reconnect_policy_ ? reconnect_attempt_count_ : static_cast<uint32_t>(retry_attempts_);
743
744 record_error(diagnostics::ErrorLevel::ERROR, diagnostics::ErrorCategory::CONNECTION, "handle_close", ec,
745 "Connection closed with error: " + ec.message(), retryable, current_attempts);
746 }
747 connected_.store(false);
748 writing_ = false;
749 connect_timer_.cancel();
750 close_socket();
751 if (stop_requested_.load() || stopping_.load() || state_.is_state(LinkState::Closed)) {
752 transition_to(LinkState::Closed, ec);
753 return;
754 }
755 transition_to(LinkState::Connecting, ec);
756 schedule_retry(self, seq);
757}

779void TcpClient::Impl::report_backpressure(size_t queued_bytes) {
780 if (stop_requested_.load() || stopping_.load()) return;
781
782 OnBackpressure on_bp;
783 {
784 std::lock_guard<std::mutex> lock(callback_mtx_);
785 on_bp = on_bp_;
786 }
787 if (!on_bp) return;
788
789 if (!backpressure_active_ && queued_bytes >= bp_high_) {
790 backpressure_active_ = true;
791 try {
792 on_bp(queued_bytes);
793 } catch (const std::exception& e) {
794 UNILINK_LOG_ERROR("tcp_client", "on_backpressure",
795 "Exception in backpressure callback: " + std::string(e.what()));
796 } catch (...) {
797 UNILINK_LOG_ERROR("tcp_client", "on_backpressure", "Unknown exception in backpressure callback");
798 }
799 } else if (backpressure_active_ && queued_bytes <= bp_low_) {
800 backpressure_active_ = false;
801 try {
802 on_bp(queued_bytes);
803 } catch (const std::exception& e) {
804 UNILINK_LOG_ERROR("tcp_client", "on_backpressure",
805 "Exception in backpressure callback: " + std::string(e.what()));
806 } catch (...) {
807 UNILINK_LOG_ERROR("tcp_client", "on_backpressure", "Unknown exception in backpressure callback");
808 }

812void TcpClient::Impl::transition_to(LinkState next, const boost::system::error_code& ec) {
813 if (ec == net::error::operation_aborted) {
814 return;
815 }
816
817 const auto current = state_.state();
818 const bool retrying_same_state = (next == LinkState::Connecting && current == LinkState::Connecting);
819 if ((current == LinkState::Closed || current == LinkState::Error) &&
820 (next == LinkState::Closed || next == LinkState::Error)) {
821 return;
822 }
823
824 if (next == LinkState::Closed || next == LinkState::Error) {
825 if (terminal_state_notified_.exchange(true)) {
826 return;
827 }
828 } else if (current == next && !retrying_same_state) {
829 return;
830 }
831
832 state_.set_state(next);
833 notify_state();
834}
