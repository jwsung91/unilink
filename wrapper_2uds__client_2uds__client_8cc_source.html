85 void fulfill_all_locked(bool value) {
86 for (auto& p : pending_promises_) {
87 try {
88 p.set_value(value);
89 } catch (...) {
90 }
91 }
92 pending_promises_.clear();
93 }

95 std::future<bool> start() {
96 std::unique_lock<std::mutex> lock(mutex_);
97 if (channel_ && channel_->is_connected()) {
98 std::promise<bool> p;
99 p.set_value(true);
100 return p.get_future();
101 }
102
103 std::promise<bool> p;
104 auto f = p.get_future();
105 pending_promises_.emplace_back(std::move(p));
106
107 if (started_.exchange(true)) {
108 return f;
109 }
110
111 if (!channel_) {
112 config::UdsClientConfig cfg;
113 cfg.socket_path = socket_path_;
114 cfg.retry_interval_ms = static_cast<unsigned>(retry_interval_.count());
115 cfg.max_retries = max_retries_;
116 cfg.connection_timeout_ms = static_cast<unsigned>(connection_timeout_.count());
117
118 if (use_external_context_) {
119 channel_ = factory::ChannelFactory::create(cfg, external_ioc_);
120 if (manage_external_context_ && !external_thread_.joinable()) {
121 if (external_ioc_ && external_ioc_->stopped()) {
122 external_ioc_->restart();
123 }
124 work_guard_ = std::make_unique<boost::asio::executor_work_guard<boost::asio::io_context::executor_type>>(
125 boost::asio::make_work_guard(*external_ioc_));
126 external_thread_ = std::thread([this]() {
127 try {
128 external_ioc_->run();
129 } catch (...) {
130 }
131 });
132 }
133 } else {
134 channel_ = factory::ChannelFactory::create(cfg);
135 }
136 setup_internal_handlers();
137 }
138
139 lock.unlock(); // UNLOCK BEFORE START
140 channel_->start();
141 lock.lock();
142
143 return f;
144 }
145
146 void stop() {
147 std::unique_lock<std::mutex> lock(mutex_);
148 if (!started_.exchange(false)) {
149 return;
150 }
151
152 // RELEASE LOCK before calling channel_->stop() because it might trigger
153 // callbacks that try to acquire this same lock (e.g., on_state -> fulfill_all)
154 if (channel_) {
155 channel_->on_bytes(nullptr);
156 channel_->on_state(nullptr);
157 lock.unlock();
158 channel_->stop();
159 lock.lock();
160 }
161
162 if (work_guard_) {
163 work_guard_.reset();
164 }
165
166 if (use_external_context_ && manage_external_context_ && external_ioc_) {
167 external_ioc_->stop();
168 }
169
170 if (external_thread_.joinable()) {
171 if (std::this_thread::get_id() != external_thread_.get_id()) {
172 lock.unlock(); // RELEASE LOCK BEFORE JOINING
173 external_thread_.join();
174 lock.lock(); // RE-ACQUIRE
175 } else {
176 external_thread_.detach();
177 }
178 }
179
180 fulfill_all_locked(false);
181 channel_.reset();
182 if (framer_) {
183 framer_->reset();
184 }
185 }
186
187 void setup_internal_handlers() {
188 if (!channel_) return;
189
190 std::weak_ptr<bool> weak_alive = alive_marker_;
191
192 channel_->on_state([this, weak_alive](base::LinkState state) {
193 auto alive = weak_alive.lock();
194 if (!alive) return;
195
196 if (state == base::LinkState::Connected) {
197 ConnectionHandler handler;
198 {
199 std::lock_guard<std::mutex> lock(mutex_);
200 fulfill_all_locked(true);
201 handler = connect_handler_;
202 }
203 if (handler) {
204 handler(ConnectionContext(0));
205 }
206 } else if (state == base::LinkState::Error) {
207 ErrorHandler handler;
208 std::shared_ptr<interface::Channel> channel_snapshot;
209 {
210 std::lock_guard<std::mutex> lock(mutex_);
211 fulfill_all_locked(false);
212 handler = error_handler_;
213 channel_snapshot = channel_;
214 }
215 if (handler) {
216 bool handled = false;
217 if (auto transport = std::dynamic_pointer_cast<transport::UdsClient>(channel_snapshot)) {
218 if (auto info = transport->last_error_info()) {
219 handler(diagnostics::to_error_context(*info));
220 handled = true;
221 }
222 }
223 if (!handled) {
224 handler(ErrorContext(ErrorCode::IoError, "Connection error"));
225 }
226 }
227 } else if (state == base::LinkState::Closed || state == base::LinkState::Idle) {
228 ConnectionHandler handler;
229 {
230 std::lock_guard<std::mutex> lock(mutex_);
231 fulfill_all_locked(false);
232 handler = disconnect_handler_;
233 }
234 if (handler) {
235 handler(ConnectionContext(0));
236 }
237 }
238 });
239
240 channel_->on_bytes([this, weak_alive](memory::ConstByteSpan data) {
241 auto alive = weak_alive.lock();
242 if (!alive) return;
243
244 // 1. Raw data handler
245 MessageHandler handler;
246 {
247 std::lock_guard<std::mutex> lock(mutex_);
248 handler = data_handler_;
249 }
250 if (handler) {
251 handler(MessageContext(0, base::safe_convert::uint8_to_string(data.data(), data.size())));
252 }
253
254 // 2. Framer integration
255 std::lock_guard<std::mutex> lock(mutex_);
256 if (framer_) {
257 framer_->push_bytes(data);
258 }
259 });
260 }
261
262 // Attach the stored message_handler_ to framer_->on_message().
263 // Must be called with mutex_ already held.
264 void attach_framer_callback() {
265 if (!framer_) return;
266 framer_->on_message([this](memory::ConstByteSpan msg) {
267 MessageHandler handler;
268 {
269 std::lock_guard<std::mutex> lock(mutex_);
270 handler = message_handler_;
271 }
272 if (handler) {
273 handler(MessageContext(0, base::safe_convert::uint8_to_string(msg.data(), msg.size())));
274 }
275 });
276 }
277
278 void set_framer(std::unique_ptr<framer::IFramer> framer) {
279 std::lock_guard<std::mutex> lock(mutex_);
280 framer_ = std::move(framer);
281 if (framer_ && message_handler_) attach_framer_callback();
282 }
283
284 void on_message(MessageHandler handler) {
285 std::lock_guard<std::mutex> lock(mutex_);
286 message_handler_ = std::move(handler);
287 if (framer_) attach_framer_callback();
288 }
