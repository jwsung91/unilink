103 void fulfill_all_locked(bool value) {
104 for (auto& p : pending_promises_) {
105 try {
106 p.set_value(value);
107 } catch (...) {
108 }
109 }
110 pending_promises_.clear();
111 }

113 void flush_batches() {
114 std::unique_lock<std::shared_mutex> lock(mutex_);
115 if (!data_batch_queue_.empty()) {
116 auto handler = data_batch_handler_;
117 auto batch = std::move(data_batch_queue_);
118 data_batch_queue_.clear();
119 if (handler) {
120 lock.unlock();
121 handler(batch);
122 lock.lock();
123 }
124 }
125 if (!message_batch_queue_.empty()) {
126 auto handler = message_batch_handler_;
127 auto batch = std::move(message_batch_queue_);
128 message_batch_queue_.clear();
129 if (handler) {
130 lock.unlock();
131 handler(batch);
132 lock.lock();
133 }
134 }
135 if (batch_timer_) {
136 batch_timer_->cancel();
137 }
138 }
139
140 void schedule_batch_timer() {
141 if (!batch_timer_) return;
142 batch_timer_->expires_after(max_batch_latency_);
143 batch_timer_->async_wait(
144 [this, weak_alive = std::weak_ptr<bool>(alive_marker_)](const boost::system::error_code& ec) {
145 if (ec) return;
146 auto alive = weak_alive.lock();
147 if (!alive) return;
148 flush_batches();
149 });
150 }

152 std::future<bool> start() {
153 std::unique_lock<std::shared_mutex> lock(mutex_);
154 if (channel_ && channel_->is_connected()) {
155 std::promise<bool> p;
156 p.set_value(true);
157 return p.get_future();
158 }
159
160 std::promise<bool> p;
161 auto f = p.get_future();
162 pending_promises_.emplace_back(std::move(p));
163
164 if (started_.load()) {
165 return f;
166 }
167
168 if (!channel_) {
169 config::UdsClientConfig cfg;
170 cfg.socket_path = socket_path_;
171 cfg.retry_interval_ms = static_cast<unsigned>(retry_interval_.count());
172 cfg.max_retries = max_retries_;
173 cfg.connection_timeout_ms = static_cast<unsigned>(connection_timeout_.count());
174 cfg.backpressure_threshold = backpressure_threshold_;
175 cfg.backpressure_strategy = backpressure_strategy_;
176
177 if (use_external_context_) {
178 channel_ = factory::ChannelFactory::create(cfg, external_ioc_);
179 if (manage_external_context_ && !external_thread_.joinable()) {
180 if (external_ioc_ && external_ioc_->stopped()) {
181 external_ioc_->restart();
182 }
183 work_guard_ = std::make_unique<boost::asio::executor_work_guard<boost::asio::io_context::executor_type>>(
184 boost::asio::make_work_guard(*external_ioc_));
185 external_thread_ = std::thread([this]() {
186 try {
187 external_ioc_->run();
188 } catch (...) {
189 }
190 });
191 }
192 } else {
193 channel_ = factory::ChannelFactory::create(cfg);
194 }
195 setup_internal_handlers();
196 }
197 started_.store(true);
198
199 lock.unlock(); // UNLOCK BEFORE START
200 channel_->start();
201 lock.lock();
202
203 return f;
204 }

206 void stop() {
207 std::unique_lock<std::shared_mutex> lock(mutex_);
208 if (!started_.load()) {
209 fulfill_all_locked(false);
210 return;
211 }
212 started_.store(false);
213 bp_cv_.notify_all();
214
215 if (batch_timer_) {
216 batch_timer_->cancel();
217 batch_timer_.reset();
218 }
219
220 // RELEASE LOCK before calling channel_->stop() because it might trigger
221 // callbacks that try to acquire this same lock (e.g., on_state -> fulfill_all)
222 if (channel_) {
223 channel_->on_bytes(nullptr);
224 channel_->on_state(nullptr);
225 lock.unlock();
226 channel_->stop();
227 lock.lock();
228 }
229
230 if (work_guard_) {
231 work_guard_.reset();
232 }
233
234 if (use_external_context_ && manage_external_context_ && external_ioc_) {
235 external_ioc_->stop();
236 }
237
238 if (external_thread_.joinable()) {
239 if (std::this_thread::get_id() != external_thread_.get_id()) {
240 lock.unlock(); // RELEASE LOCK BEFORE JOINING
241 external_thread_.join();
242 lock.lock(); // RE-ACQUIRE
243 } else {
244 external_thread_.detach();
245 }
246 }
247
248 fulfill_all_locked(false);
249 channel_.reset();
250 if (framer_) {
251 framer_->reset();
252 }
253 }

267 bool send_blocking(std::string_view data) {
268 std::unique_lock<std::mutex> bp_lock(bp_mutex_);
269 bp_cv_.wait(bp_lock, [this] {
270 std::shared_lock<std::shared_mutex> lock(mutex_);
271 return !started_.load() || !channel_ || !channel_->is_connected() || !channel_->is_backpressure_active();
272 });

277 std::shared_lock<std::shared_mutex> lock(mutex_);
278 if (channel_ && channel_->is_connected()) {
279 return channel_->async_write_copy(

282 return false;

287 void setup_internal_handlers() {
288 if (!channel_) return;
289
290 batch_timer_ = std::make_unique<boost::asio::steady_timer>(channel_->get_executor());
291
292 std::weak_ptr<bool> weak_alive = alive_marker_;
293
294 channel_->on_state([this, weak_alive](base::LinkState state) {
295 auto alive = weak_alive.lock();
296 if (!alive) return;
297
298 if (state == base::LinkState::Connected) {
299 ConnectionHandler handler;
300 {
301 std::unique_lock<std::shared_mutex> lock(mutex_);
302 fulfill_all_locked(true);
303 handler = connect_handler_;
304 }
305 if (handler) {
306 handler(ConnectionContext(0));
307 }
308 } else if (state == base::LinkState::Error) {
309 ErrorHandler handler;
310 std::shared_ptr<interface::Channel> channel_snapshot;
311 {
312 std::unique_lock<std::shared_mutex> lock(mutex_);
313 fulfill_all_locked(false);
314 handler = error_handler_;
315 channel_snapshot = channel_;
316 }
317 if (handler) {
318 bool handled = false;
319 if (auto transport = std::dynamic_pointer_cast<transport::UdsClient>(channel_snapshot)) {
320 if (auto info = transport->last_error_info()) {
321 handler(diagnostics::to_error_context(*info));
322 handled = true;
323 }
324 }
325 if (!handled) {
326 handler(ErrorContext(ErrorCode::IoError, "Connection error"));
327 }
328 }
329 } else if (state == base::LinkState::Closed || state == base::LinkState::Idle) {
330 ConnectionHandler handler;
331 {
332 std::unique_lock<std::shared_mutex> lock(mutex_);
333 fulfill_all_locked(false);
334 handler = disconnect_handler_;
335 }
336 if (handler) {
337 handler(ConnectionContext(0));
338 }
339 }
340 });
341
342 channel_->on_bytes([this, weak_alive](memory::ConstByteSpan data) {
343 auto alive = weak_alive.lock();
344 if (!alive) return;
345
346 std::unique_lock<std::shared_mutex> lock(mutex_);
347 if (data_batch_handler_) {
348 data_batch_queue_.emplace_back(0, memory::SafeDataBuffer(data));
349 if (data_batch_queue_.size() >= max_batch_size_) {
350 auto handler = data_batch_handler_;
351 auto batch = std::move(data_batch_queue_);
352 data_batch_queue_.clear();
353 lock.unlock();
354 handler(batch);
355 } else if (data_batch_queue_.size() == 1) {
356 schedule_batch_timer();
357 }
358 return;
359 }
360
361 MessageHandler handler = data_handler_;
362 if (handler) {
363 lock.unlock();
364 handler(MessageContext(0, memory::SafeDataBuffer(data)));
365 lock.lock();
366 }
367
368 if (framer_) {
369 framer_->push_bytes(data);
370 }
371 });
372
373 channel_->on_backpressure([this, weak_alive](size_t queued) {
374 bp_cv_.notify_all();
375 auto alive = weak_alive.lock();
376 if (!alive) return;
377 std::shared_lock<std::shared_mutex> lock(mutex_);
378 if (bp_handler_) bp_handler_(queued);
379 });
380 }
381
382 // Attach the stored message_handler_ or message_batch_handler_ to framer_->on_message().
383 // Must be called with mutex_ already held.
384 void attach_framer_callback() {
385 if (!framer_) return;
386 framer_->on_message([this](memory::ConstByteSpan msg) {
387 std::unique_lock<std::shared_mutex> lock(mutex_);
388 if (message_batch_handler_) {
389 message_batch_queue_.emplace_back(0, memory::SafeDataBuffer(msg));
390 if (message_batch_queue_.size() >= max_batch_size_) {
391 auto handler = message_batch_handler_;
392 auto batch = std::move(message_batch_queue_);
393 message_batch_queue_.clear();
394 lock.unlock();
395 handler(batch);
396 } else if (message_batch_queue_.size() == 1) {
397 schedule_batch_timer();
398 }
399 return;
400 }
401
402 MessageHandler handler = message_handler_;
403 if (handler) {
404 lock.unlock();
405 handler(MessageContext(0, memory::SafeDataBuffer(msg)));
406 }
407 });
408 }

410 void set_framer(std::unique_ptr<framer::IFramer> framer) {

412 framer_ = std::move(framer);
413 if (framer_ && (message_handler_ || message_batch_handler_)) attach_framer_callback();
414 }
