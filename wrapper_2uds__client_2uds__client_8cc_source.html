97 void fulfill_all_locked(bool value) {
98 for (auto& p : pending_promises_) {
99 try {
100 p.set_value(value);
101 } catch (...) {
102 }
103 }
104 pending_promises_.clear();
105 }
106
107 void flush_batches() {
108 std::unique_lock<std::shared_mutex> lock(mutex_);
109 if (!data_batch_queue_.empty()) {
110 auto handler = data_batch_handler_;
111 auto batch = std::move(data_batch_queue_);
112 data_batch_queue_.clear();
113 if (handler) {
114 lock.unlock();
115 handler(batch);
116 lock.lock();
117 }
118 }
119 if (!message_batch_queue_.empty()) {
120 auto handler = message_batch_handler_;
121 auto batch = std::move(message_batch_queue_);
122 message_batch_queue_.clear();
123 if (handler) {
124 lock.unlock();
125 handler(batch);
126 lock.lock();
127 }
128 }
129 if (batch_timer_) {
130 batch_timer_->cancel();
131 }
132 }
133
134 void schedule_batch_timer() {
135 if (!batch_timer_) return;
136 batch_timer_->expires_after(max_batch_latency_);
137 batch_timer_->async_wait(
138 [this, weak_alive = std::weak_ptr<bool>(alive_marker_)](const boost::system::error_code& ec) {
139 if (ec) return;
140 auto alive = weak_alive.lock();
141 if (!alive) return;
142 flush_batches();
143 });
144 }

146 std::future<bool> start() {
147 std::unique_lock<std::shared_mutex> lock(mutex_);
148 if (channel_ && channel_->is_connected()) {
149 std::promise<bool> p;
150 p.set_value(true);
151 return p.get_future();
152 }
153
154 std::promise<bool> p;
155 auto f = p.get_future();
156 pending_promises_.emplace_back(std::move(p));
157
158 if (started_.load()) {
159 return f;
160 }
161
162 if (!channel_) {
163 config::UdsClientConfig cfg;
164 cfg.socket_path = socket_path_;
165 cfg.retry_interval_ms = static_cast<unsigned>(retry_interval_.count());
166 cfg.max_retries = max_retries_;
167 cfg.connection_timeout_ms = static_cast<unsigned>(connection_timeout_.count());
168
169 if (use_external_context_) {
170 channel_ = factory::ChannelFactory::create(cfg, external_ioc_);
171 if (manage_external_context_ && !external_thread_.joinable()) {
172 if (external_ioc_ && external_ioc_->stopped()) {
173 external_ioc_->restart();
174 }
175 work_guard_ = std::make_unique<boost::asio::executor_work_guard<boost::asio::io_context::executor_type>>(
176 boost::asio::make_work_guard(*external_ioc_));
177 external_thread_ = std::thread([this]() {
178 try {
179 external_ioc_->run();
180 } catch (...) {
181 }
182 });
183 }
184 } else {
185 channel_ = factory::ChannelFactory::create(cfg);
186 }
187 setup_internal_handlers();
188 }
189 started_.store(true);
190
191 lock.unlock(); // UNLOCK BEFORE START
192 channel_->start();
193 lock.lock();
194
195 return f;
196 }
197
198 void stop() {
199 std::unique_lock<std::shared_mutex> lock(mutex_);
200 if (!started_.load()) {
201 fulfill_all_locked(false);
202 return;
203 }
204 started_.store(false);
205
206 if (batch_timer_) {
207 batch_timer_->cancel();
208 batch_timer_.reset();
209 }
210
211 // RELEASE LOCK before calling channel_->stop() because it might trigger
212 // callbacks that try to acquire this same lock (e.g., on_state -> fulfill_all)
213 if (channel_) {
214 channel_->on_bytes(nullptr);
215 channel_->on_state(nullptr);
216 lock.unlock();
217 channel_->stop();
218 lock.lock();
219 }
220
221 if (work_guard_) {
222 work_guard_.reset();
223 }
224
225 if (use_external_context_ && manage_external_context_ && external_ioc_) {
226 external_ioc_->stop();
227 }
228
229 if (external_thread_.joinable()) {
230 if (std::this_thread::get_id() != external_thread_.get_id()) {
231 lock.unlock(); // RELEASE LOCK BEFORE JOINING
232 external_thread_.join();
233 lock.lock(); // RE-ACQUIRE
234 } else {
235 external_thread_.detach();
236 }
237 }
238
239 fulfill_all_locked(false);
240 channel_.reset();
241 if (framer_) {
242 framer_->reset();
243 }
244 }

246 void setup_internal_handlers() {
247 if (!channel_) return;
248
249 batch_timer_ = std::make_unique<boost::asio::steady_timer>(channel_->get_executor());
250
251 std::weak_ptr<bool> weak_alive = alive_marker_;
252
253 channel_->on_state([this, weak_alive](base::LinkState state) {
254 auto alive = weak_alive.lock();
255 if (!alive) return;
256
257 if (state == base::LinkState::Connected) {
258 ConnectionHandler handler;
259 {
260 std::unique_lock<std::shared_mutex> lock(mutex_);
261 fulfill_all_locked(true);
262 handler = connect_handler_;
263 }
264 if (handler) {
265 handler(ConnectionContext(0));
266 }
267 } else if (state == base::LinkState::Error) {
268 ErrorHandler handler;
269 std::shared_ptr<interface::Channel> channel_snapshot;
270 {
271 std::unique_lock<std::shared_mutex> lock(mutex_);
272 fulfill_all_locked(false);
273 handler = error_handler_;
274 channel_snapshot = channel_;
275 }
276 if (handler) {
277 bool handled = false;
278 if (auto transport = std::dynamic_pointer_cast<transport::UdsClient>(channel_snapshot)) {
279 if (auto info = transport->last_error_info()) {
280 handler(diagnostics::to_error_context(*info));
281 handled = true;
282 }
283 }
284 if (!handled) {
285 handler(ErrorContext(ErrorCode::IoError, "Connection error"));
286 }
287 }
288 } else if (state == base::LinkState::Closed || state == base::LinkState::Idle) {
289 ConnectionHandler handler;
290 {
291 std::unique_lock<std::shared_mutex> lock(mutex_);
292 fulfill_all_locked(false);
293 handler = disconnect_handler_;
294 }
295 if (handler) {
296 handler(ConnectionContext(0));
297 }
298 }
299 });
300
301 channel_->on_bytes([this, weak_alive](memory::ConstByteSpan data) {
302 auto alive = weak_alive.lock();
303 if (!alive) return;
304
305 std::unique_lock<std::shared_mutex> lock(mutex_);
306 if (data_batch_handler_) {
307 data_batch_queue_.emplace_back(0, memory::SafeDataBuffer(data));
308 if (data_batch_queue_.size() >= max_batch_size_) {
309 auto handler = data_batch_handler_;
310 auto batch = std::move(data_batch_queue_);
311 data_batch_queue_.clear();
312 lock.unlock();
313 handler(batch);
314 } else if (data_batch_queue_.size() == 1) {
315 schedule_batch_timer();
316 }
317 return;
318 }
319
320 MessageHandler handler = data_handler_;
321 if (handler) {
322 lock.unlock();
323 handler(MessageContext(0, memory::SafeDataBuffer(data)));
324 lock.lock();
325 }
326
327 if (framer_) {
328 framer_->push_bytes(data);
329 }
330 });
331 }
332
333 // Attach the stored message_handler_ or message_batch_handler_ to framer_->on_message().
334 // Must be called with mutex_ already held.
335 void attach_framer_callback() {
336 if (!framer_) return;
337 framer_->on_message([this](memory::ConstByteSpan msg) {
338 std::unique_lock<std::shared_mutex> lock(mutex_);
339 if (message_batch_handler_) {
340 message_batch_queue_.emplace_back(0, memory::SafeDataBuffer(msg));
341 if (message_batch_queue_.size() >= max_batch_size_) {
342 auto handler = message_batch_handler_;
343 auto batch = std::move(message_batch_queue_);
344 message_batch_queue_.clear();
345 lock.unlock();
346 handler(batch);
347 } else if (message_batch_queue_.size() == 1) {
348 schedule_batch_timer();
349 }
350 return;
351 }
352
353 MessageHandler handler = message_handler_;
354 if (handler) {
355 lock.unlock();
356 handler(MessageContext(0, memory::SafeDataBuffer(msg)));
357 }
358 });
359 }
360
361 void set_framer(std::unique_ptr<framer::IFramer> framer) {
362 std::unique_lock<std::shared_mutex> lock(mutex_);
363 framer_ = std::move(framer);
364 if (framer_ && (message_handler_ || message_batch_handler_)) attach_framer_callback();
365 }

367 void on_message(MessageHandler handler) {
368 std::unique_lock<std::shared_mutex> lock(mutex_);

371 }
372
373 void on_message_batch(BatchMessageHandler handler) {
374 std::unique_lock<std::shared_mutex> lock(mutex_);
375 message_batch_handler_ = std::move(handler);
376 if (framer_) attach_framer_callback();
377 }

396bool UdsClient::send(std::string_view data) {
397 std::shared_lock<std::shared_mutex> lock(impl_->mutex_);
398 if (impl_->channel_ && impl_->channel_->is_connected()) {
399 impl_->channel_->async_write_copy(
400 memory::ConstByteSpan(reinterpret_cast<const uint8_t*>(data.data()), data.size()));
401 return true;
402 }
403 return false;
404}
