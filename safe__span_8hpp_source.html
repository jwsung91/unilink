99  // Subspan operations
100  constexpr SafeSpan<T> subspan(size_type offset, size_type count = SIZE_MAX) const {
101  if (offset > size_) {
102  throw std::out_of_range("SafeSpan subspan offset out of range");
103  }
104  size_type actual_count = (count == SIZE_MAX) ? (size_ - offset) : count;
105  if (offset + actual_count > size_) {
