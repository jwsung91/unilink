32template <typename T, std::size_t Extent = std::dynamic_extent>
33class SafeSpan : public std::span<T, Extent> {
34 public:
35 using Base = std::span<T, Extent>;
36 using Base::Base;
37
38 // Polyfill for C++17 pointer+size constructor (std::span requires contiguous_iterator)
39 constexpr SafeSpan(T* data, std::size_t size) noexcept : Base(data, size) {}
40
41 // Constructor from std::span
42 constexpr SafeSpan(Base s) noexcept : Base(s) {}
43
44 // Polyfill for at() which is missing in std::span
45 constexpr typename Base::reference at(std::size_t index) const {
46 if (index >= this->size()) {
47 throw std::out_of_range("SafeSpan index out of range");
48 }
49 return (*this)[index];
50 }
51
52 // Override subspan to return SafeSpan instead of std::span
53 constexpr SafeSpan<T, std::dynamic_extent> subspan(std::size_t offset,
54 std::size_t count = std::dynamic_extent) const {
55 return SafeSpan<T, std::dynamic_extent>(Base::subspan(offset, count));
56 }
57
58 constexpr SafeSpan<T, std::dynamic_extent> first(std::size_t count) const {
59 return SafeSpan<T, std::dynamic_extent>(Base::first(count));
60 }
61
62 constexpr SafeSpan<T, std::dynamic_extent> last(std::size_t count) const {
63 return SafeSpan<T, std::dynamic_extent>(Base::last(count));
64 }
65};
66

70using CharSpan = SafeSpan<char>;
