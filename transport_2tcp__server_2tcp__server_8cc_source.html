85 void TcpServer::start() {
86  // Prevent duplicate start calls when already running or in-progress
87  auto current = state_.get_state();
88  if (current == common::LinkState::Listening || current == common::LinkState::Connected ||
89  current == common::LinkState::Connecting) {
90  UNILINK_LOG_DEBUG("tcp_server", "start", "Start called while already active, ignoring");
91  return;
92  }
93  stopping_.store(false);
94 
95  // Ensure the shared io_context is running when we rely on the global manager.
96  if (uses_global_ioc_) {
97  auto& manager = common::IoContextManager::instance();
98  if (!manager.is_running()) {
99  manager.start();
100  }
101  }
102 
103  if (!acceptor_) {
104  UNILINK_LOG_ERROR("tcp_server", "start", "Acceptor is null");
105  common::error_reporting::report_system_error("tcp_server", "start", "Acceptor is null");
106  state_.set_state(common::LinkState::Error);
107  notify_state();
108  return;
109  }
110 
111  if (owns_ioc_) {
112  ioc_thread_ = std::thread([this] { ioc_.run(); });
113  }
114  auto self = shared_from_this();
115  if (ioc_.get_executor().running_in_this_thread()) {
116  // If start is invoked from the IO thread, bind immediately to avoid race with early clients
117  if (!self->stopping_.load()) {
118  self->attempt_port_binding(0);
119  }
120  } else {
121  // dispatch allows immediate execution if the IO thread is already running
122  net::dispatch(ioc_, [self] {
123  if (self->stopping_.load()) return;
124  self->attempt_port_binding(0);
125  });
126  }
127 }

129 void TcpServer::stop() {
130  if (stopping_.exchange(true)) {
131  return; // Already stopping/stopped
132  }
133 
134  auto cleanup = [this]() {
135  boost::system::error_code ec;
136  if (acceptor_ && acceptor_->is_open()) {
137  acceptor_->close(ec);
138  }
139  std::vector<std::shared_ptr<TcpServerSession>> sessions_copy;
140  {
141  std::lock_guard<std::mutex> lock(sessions_mutex_);
142  sessions_copy.reserve(sessions_.size());
143  for (auto& kv : sessions_) {
144  sessions_copy.push_back(kv.second);
145  }
146  sessions_.clear();
147  current_session_.reset();
148  }
149  for (auto& session : sessions_copy) {
150  if (session) {
151  session->stop();
152  }
153  }
154  // Clean up all sessions
155  state_.set_state(common::LinkState::Closed);
156  };
157 
158  const bool in_ioc_thread = ioc_.get_executor().running_in_this_thread();
159  const bool has_active_ioc_thread =
160  owns_ioc_ || (uses_global_ioc_ && common::IoContextManager::instance().is_running());
161 
162  if (in_ioc_thread) {
163  cleanup();
164  if (owns_ioc_) {
165  ioc_.stop();
166  }
167  return;
168  }
169 
170  if (has_active_ioc_thread) {
171  auto self = shared_from_this();
172  auto cleanup_promise = std::make_shared<std::promise<void>>();
173  auto cleanup_future = cleanup_promise->get_future();
174 
175  net::post(ioc_, [self, cleanup, cleanup_promise] {
176  cleanup();
177  cleanup_promise->set_value();
178  });
179 
180  cleanup_future.wait();
181  } else {
182  // io_context is not running (e.g., never started); perform best-effort cleanup synchronously
183  cleanup();
184  }
185 
186  if (owns_ioc_ && ioc_thread_.joinable()) {
187  ioc_.stop();
188  ioc_thread_.join();
189  // Reset io_context to clear any remaining work
190  ioc_.restart();
191  }
192  // Don't call notify_state() in stop() as it may cause issues with callbacks
193  // during destruction
194 }

267 void TcpServer::attempt_port_binding(int retry_count) {
268  if (stopping_.load()) return;
269  boost::system::error_code ec;
270 
271  // Log retry configuration for debugging
272  if (retry_count > 0) {
273  UNILINK_LOG_DEBUG("tcp_server", "bind",
274  "Attempting port binding - retry enabled: " + std::to_string(cfg_.enable_port_retry) +
275  ", max retries: " + std::to_string(cfg_.max_port_retries) +
276  ", retry count: " + std::to_string(retry_count));
277  }
278 
279  // Open acceptor (only if not already open)
280  if (!acceptor_->is_open()) {
281  acceptor_->open(tcp::v4(), ec);
282  if (ec) {
283  UNILINK_LOG_ERROR("tcp_server", "open", "Failed to open acceptor: " + ec.message());
284  common::error_reporting::report_connection_error("tcp_server", "open", ec, false);
285  state_.set_state(common::LinkState::Error);
286  notify_state();
287  return;
288  }
289  }
290 
291  // Bind to port
292  acceptor_->bind(tcp::endpoint(tcp::v4(), cfg_.port), ec);
293  if (ec) {
294  // Check if retry is enabled and we haven't exceeded max retries
295  if (cfg_.enable_port_retry && retry_count < cfg_.max_port_retries) {
296  UNILINK_LOG_WARNING("tcp_server", "bind",
297  "Failed to bind to port " + std::to_string(cfg_.port) + " (attempt " +
298  std::to_string(retry_count + 1) + "/" + std::to_string(cfg_.max_port_retries) + "): " +
299  ec.message() + ". Retrying in " + std::to_string(cfg_.port_retry_interval_ms) + "ms...");
300 
301  // Schedule retry
302  auto self = shared_from_this();
303  auto timer = std::make_shared<net::steady_timer>(ioc_);
304  timer->expires_after(std::chrono::milliseconds(cfg_.port_retry_interval_ms));
305  timer->async_wait([self, retry_count, timer](const boost::system::error_code& timer_ec) {
306  if (!timer_ec) {
307  self->attempt_port_binding(retry_count + 1);
308  }
309  });
310  return;
311  } else {
312  // No retry enabled or max retries exceeded
313  std::string error_msg = "Failed to bind to port: " + std::to_string(cfg_.port) + " - " + ec.message();
314  if (cfg_.enable_port_retry) {
315  error_msg += " (after " + std::to_string(retry_count) + " retries)";
316  }
317  UNILINK_LOG_ERROR("tcp_server", "bind", error_msg);
318  common::error_reporting::report_connection_error("tcp_server", "bind", ec, false);
319  state_.set_state(common::LinkState::Error);
320  notify_state();
321  return;
322  }
323  }
324 
325  // Listen for connections
326  acceptor_->listen(boost::asio::socket_base::max_listen_connections, ec);
327  if (ec) {
328  UNILINK_LOG_ERROR("tcp_server", "listen",
329  "Failed to listen on port: " + std::to_string(cfg_.port) + " - " + ec.message());
330  common::error_reporting::report_connection_error("tcp_server", "listen", ec, false);
331  state_.set_state(common::LinkState::Error);
332  notify_state();
333  return;
334  }
335 
336  // Success
337  if (retry_count > 0) {
338  UNILINK_LOG_INFO("tcp_server", "bind",
339  "Successfully bound to port " + std::to_string(cfg_.port) + " after " +
340  std::to_string(retry_count) + " retries");
341  } else {
342  UNILINK_LOG_INFO("tcp_server", "bind", "Successfully bound to port " + std::to_string(cfg_.port));
343  }
344 
345  state_.set_state(common::LinkState::Listening);
346  notify_state();
347  do_accept();
348 }

350 void TcpServer::do_accept() {
351  if (stopping_.load() || !acceptor_ || !acceptor_->is_open()) return;
352 
353  auto self = shared_from_this();
354  acceptor_->async_accept([self](auto ec, tcp::socket sock) {
355  if (self->stopping_.load()) {
356  return;
357  }
358  if (ec) {
359  // "Operation canceled"
360  if (ec == boost::asio::error::operation_aborted) {
361  UNILINK_LOG_DEBUG("tcp_server", "accept", "Accept canceled (server shutting down)");
362  } else {
363  UNILINK_LOG_ERROR("tcp_server", "accept", "Accept error: " + ec.message());
364  common::error_reporting::report_connection_error("tcp_server", "accept", ec, true);
365  self->state_.set_state(common::LinkState::Error);
366  self->notify_state();
367  }
368  // Continue accepting only if server is not shutting down
369  if (!self->state_.is_state(common::LinkState::Closed) && !self->stopping_.load()) {
370  self->do_accept();
371  }
372  return;
373  }
374 
375  boost::system::error_code ep_ec;
376  auto rep = sock.remote_endpoint(ep_ec);
377  std::string client_info = "unknown";
378  if (!ep_ec) {
379  client_info = rep.address().to_string() + ":" + std::to_string(rep.port());
380  }
381 
382  // Client limit check (after connection acceptance, before session creation)
383  if (self->client_limit_enabled_) {
384  std::lock_guard<std::mutex> lock(self->sessions_mutex_);
385  if (self->sessions_.size() >= self->max_clients_) {
386  UNILINK_LOG_WARNING("tcp_server", "accept",
387  "Client connection rejected - server at capacity (" +
388  std::to_string(self->sessions_.size()) + "/" + std::to_string(self->max_clients_) +
389  "): " + client_info);
390 
391  //
392  boost::system::error_code close_ec;
393  sock.close(close_ec);
394  if (close_ec) {
395  UNILINK_LOG_DEBUG("tcp_server", "accept", "Error closing rejected socket: " + close_ec.message());
396  }
397 
398  //
399  self->do_accept();
400  return;
401  }
402  }
403 
404  if (!ep_ec) {
405  UNILINK_LOG_INFO("tcp_server", "accept", "Client connected: " + client_info);
406  } else {
407  UNILINK_LOG_INFO("tcp_server", "accept", "Client connected (endpoint unknown)");
408  }
409 
410  // Create new session
411  auto new_session =
412  std::make_shared<TcpServerSession>(self->ioc_, std::move(sock), self->cfg_.backpressure_threshold);
413 
414  // Add session to list
415  size_t client_id;
416  {
417  std::lock_guard<std::mutex> lock(self->sessions_mutex_);
418 
419  client_id = self->next_client_id_.fetch_add(1);
420  self->sessions_.emplace(client_id, new_session);
421 
422  // Update current active session (existing API compatibility)
423  self->current_session_ = new_session;
424  }
425 
426  // Set session callbacks
427  if (self->on_bytes_) {
428  new_session->on_bytes([self, client_id](const uint8_t* data, size_t size) {
429  // Call existing callback (compatibility)
430  if (self->on_bytes_) {
431  self->on_bytes_(data, size);
432  }
433 
434  // Call multi-client callback
435  if (self->on_multi_data_) {
436  std::string str_data = common::safe_convert::uint8_to_string(data, size);
437  self->on_multi_data_(client_id, str_data);
438  }
439  });
440  }
441 
442  if (self->on_bp_) new_session->on_backpressure(self->on_bp_);
443 
444  // Handle session termination
445  new_session->on_close([self, client_id, new_session] {
446  if (self->stopping_.load()) {
447  return;
448  }
449  // Call multi-client callback
450  if (self->on_multi_disconnect_) {
451  self->on_multi_disconnect_(client_id);
452  }
453 
454  bool was_current = false;
455  // Remove from session list
456  {
457  std::lock_guard<std::mutex> lock(self->sessions_mutex_);
458  self->sessions_.erase(client_id);
459  was_current = (self->current_session_ == new_session);
460  if (was_current) {
461  if (!self->sessions_.empty()) {
462  self->current_session_ = self->sessions_.begin()->second;
463  } else {
464  self->current_session_.reset();
465  }
466  }
467  }
468 
469  // Clean up if current session is the terminated session
470  if (was_current) {
471  self->state_.set_state(common::LinkState::Listening);
472  self->notify_state();
473  }
474  });
475 
476  // Call multi-client connection callback
477  if (self->on_multi_connect_) {
478  self->on_multi_connect_(client_id, client_info);
479  }
480 
481  // Update state for existing API compatibility
482  self->state_.set_state(common::LinkState::Connected);
483  self->notify_state();
484 
485  new_session->start();
486 
487  // Immediately accept next connection (multi-client support)
488  self->do_accept();
489  });
490 }

524 bool TcpServer::send_to_client(size_t client_id, const std::string& message) {
525  std::lock_guard<std::mutex> lock(sessions_mutex_);
526 
527  auto it = sessions_.find(client_id);
528  if (it != sessions_.end() && it->second && it->second->alive()) {
529  auto data = common::safe_convert::string_to_uint8(message);
530  it->second->async_write_copy(data.data(), data.size());
531  return true;

534  "Send failed: client_id " + std::to_string(client_id) + " not found");
