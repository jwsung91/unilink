132 void attempt_port_binding(std::shared_ptr<TcpServer> self, int retry_count) {
133 if (stopping_.load()) return;
134 boost::system::error_code ec;
135
136 auto address = net::ip::make_address(cfg_.bind_address, ec);
137 if (ec) {
138 UNILINK_LOG_ERROR("tcp_server", "bind", "Invalid bind address: " + cfg_.bind_address + ", " + ec.message());
139 state_.set_state(base::LinkState::Error);
140 notify_state();
141 return;
142 }
143
144 if (!acceptor_->is_open()) {
145 acceptor_->open(address.is_v6() ? tcp::v6() : tcp::v4(), ec);
146 if (ec) {
147 UNILINK_LOG_ERROR("tcp_server", "open", "Failed to open acceptor: " + ec.message());
148 state_.set_state(base::LinkState::Error);
149 notify_state();
150 return;
151 }
152 }
153
154 acceptor_->bind(tcp::endpoint(address, cfg_.port), ec);
155 if (ec) {
156 if (cfg_.enable_port_retry && retry_count < cfg_.max_port_retries) {
157 auto timer = std::make_shared<net::steady_timer>(ioc_);
158 timer->expires_after(std::chrono::milliseconds(cfg_.port_retry_interval_ms));
159 timer->async_wait([self, retry_count, timer](const boost::system::error_code& timer_ec) {
160 if (!timer_ec) {
161 auto* timer_impl = self->get_impl();
162 if (!timer_impl->stopping_.load()) {
163 timer_impl->attempt_port_binding(self, retry_count + 1);
164 }
165 }
166 });
167 return;
168 } else {
169 UNILINK_LOG_ERROR("tcp_server", "bind",
170 "Failed to bind to port " + std::to_string(cfg_.port) + ": " + ec.message());
171 state_.set_state(base::LinkState::Error);
172 notify_state();
173 return;
174 }
175 }
176
177 acceptor_->listen(boost::asio::socket_base::max_listen_connections, ec);
178 if (ec) {
179 UNILINK_LOG_ERROR("tcp_server", "listen",
180 "Failed to listen on port " + std::to_string(cfg_.port) + ": " + ec.message());
181 state_.set_state(base::LinkState::Error);
182 notify_state();
183 return;
184 }
185
186 state_.set_state(base::LinkState::Listening);
187 notify_state();
188 do_accept(self);
189 }

191 void do_accept(std::shared_ptr<TcpServer> self) {
192 if (stopping_.load() || !acceptor_ || !acceptor_->is_open()) return;
193
194 acceptor_->async_accept([self](auto ec, tcp::socket sock) {
195 auto* accept_impl = self->get_impl();
196 if (accept_impl->stopping_.load()) {
197 return;
198 }
199 if (ec) {
200 if (ec != boost::asio::error::operation_aborted) {
201 accept_impl->state_.set_state(base::LinkState::Error);
202 accept_impl->notify_state();
203 }
204 if (!accept_impl->state_.is_state(base::LinkState::Closed) && !accept_impl->stopping_.load()) {
205 auto timer = std::make_shared<net::steady_timer>(accept_impl->ioc_);
206 timer->expires_after(std::chrono::milliseconds(100));
207 timer->async_wait([self, timer](const boost::system::error_code&) {
208 auto* retry_impl = self->get_impl();
209 if (!retry_impl->stopping_.load()) {
210 retry_impl->do_accept(self);
211 }
212 });
213 }
214 return;
215 }
216
217 boost::system::error_code ep_ec;
218 auto rep = sock.remote_endpoint(ep_ec);
219 std::string client_info = "unknown";
220 if (!ep_ec) {
221 client_info = rep.address().to_string() + ":" + std::to_string(rep.port());
222 }
223
224 if (accept_impl->client_limit_enabled_) {
225 std::lock_guard<std::mutex> lock(accept_impl->sessions_mutex_);
226 if (accept_impl->sessions_.size() >= accept_impl->max_clients_) {
227 boost::system::error_code close_ec;
228 sock.close(close_ec);
229 accept_impl->paused_accept_ = true;
230 return;
231 }
232 }
233
234 auto new_session = std::make_shared<TcpServerSession>(accept_impl->ioc_, std::move(sock),
235 accept_impl->cfg_.backpressure_threshold,
236 accept_impl->cfg_.idle_timeout_ms);
237
238 ClientId client_id;
239 {
240 std::lock_guard<std::mutex> lock(accept_impl->sessions_mutex_);
241 client_id = accept_impl->next_client_id_.fetch_add(1);
242 accept_impl->sessions_.emplace(client_id, new_session);
243 accept_impl->current_session_ = new_session;
244 }
245
246 std::weak_ptr<TcpServer> weak_self = self;
247
248 new_session->on_bytes([weak_self, client_id](memory::ConstByteSpan data) {
249 auto shared_self = weak_self.lock();
250 if (!shared_self) return;
251 auto* bytes_impl = shared_self->get_impl();
252
253 OnBytes cb;
254 MultiClientDataHandler multi_cb;
255 {
256 std::lock_guard<std::mutex> lock(bytes_impl->sessions_mutex_);
257 cb = bytes_impl->on_bytes_;
258 multi_cb = bytes_impl->on_multi_data_;
259 }
260 if (cb) cb(data);
261 if (multi_cb) {
262 multi_cb(client_id, data);
263 }
264 });
265
266 if (accept_impl->on_bp_) new_session->on_backpressure(accept_impl->on_bp_);
267
268 new_session->on_close([weak_self, client_id, new_session] {
269 auto shared_self = weak_self.lock();
270 if (!shared_self) return;
271 auto* close_impl = shared_self->get_impl();
272 if (close_impl->stopping_.load()) return;
273
274 MultiClientDisconnectHandler disconnect_cb;
275 {
276 std::lock_guard<std::mutex> lock(close_impl->sessions_mutex_);
277 disconnect_cb = close_impl->on_multi_disconnect_;
278 }
279 if (disconnect_cb) disconnect_cb(client_id);
280
281 bool was_current = false;
282 {
283 std::lock_guard<std::mutex> lock(close_impl->sessions_mutex_);
284 close_impl->sessions_.erase(client_id);
285 if (close_impl->paused_accept_ &&
286 (!close_impl->client_limit_enabled_ || close_impl->sessions_.size() < close_impl->max_clients_)) {
287 close_impl->paused_accept_ = false;
288 net::post(close_impl->ioc_, [shared_self] { shared_self->get_impl()->do_accept(shared_self); });
289 }
290 was_current = (close_impl->current_session_ == new_session);
291 if (was_current) {
292 if (!close_impl->sessions_.empty())
293 close_impl->current_session_ = close_impl->sessions_.begin()->second;
294 else
295 close_impl->current_session_.reset();
296 }
297 }
298 if (was_current) {
299 close_impl->state_.set_state(base::LinkState::Listening);
300 close_impl->notify_state();
301 }
302 });
303
304 MultiClientConnectHandler connect_cb;
305 {
306 std::lock_guard<std::mutex> lock(accept_impl->sessions_mutex_);
307 connect_cb = accept_impl->on_multi_connect_;
308 }
309 if (connect_cb) connect_cb(client_id, client_info);
310
311 accept_impl->state_.set_state(base::LinkState::Connected);
312 accept_impl->notify_state();
313 new_session->start();
314 accept_impl->do_accept(self);
315 });
316 }

318 void perform_cleanup() {
319 try {
320 boost::system::error_code ec;
321 if (acceptor_ && acceptor_->is_open()) {
322 acceptor_->close(ec);
323 }
324
325 std::vector<std::shared_ptr<TcpServerSession>> sessions_copy;
326 {
327 std::lock_guard<std::mutex> lock(sessions_mutex_);
328 sessions_copy.reserve(sessions_.size());
329 for (auto& kv : sessions_) {
330 sessions_copy.push_back(kv.second);
331 }
332 sessions_.clear();
333 current_session_.reset();
334 }
335
336 for (auto& session : sessions_copy) {
337 if (session) {
338 session->stop();
339 }
340 }
341
342 state_.set_state(base::LinkState::Closed);
343 notify_state();
344 } catch (...) {
345 }
346 }

348 void stop(std::shared_ptr<TcpServer> self) {
349 if (stopping_.exchange(true)) {
350 return;
351 }
352
353 {
354 std::lock_guard<std::mutex> lock(sessions_mutex_);
355 on_bytes_ = nullptr;
356 on_state_ = nullptr;
357 on_bp_ = nullptr;
358 on_multi_connect_ = nullptr;
359 on_multi_data_ = nullptr;
360 on_multi_disconnect_ = nullptr;
361 }
362
363 if (ioc_.get_executor().running_in_this_thread()) {
364 perform_cleanup();
365 if (owns_ioc_) ioc_.stop();
366 return;
367 }
368
369 bool has_active_ioc = owns_ioc_ || (uses_global_ioc_ && concurrency::IoContextManager::instance().is_running());
370
371 if (has_active_ioc && self) {
372 auto cleanup_promise = std::make_shared<std::promise<void>>();
373 auto cleanup_future = cleanup_promise->get_future();
374
375 std::weak_ptr<TcpServer> weak_self = self;
376 net::dispatch(ioc_, [weak_self, cleanup_promise]() {
377 if (auto shared_self = weak_self.lock()) {
378 auto* cleanup_impl = shared_self->get_impl();
379 cleanup_impl->perform_cleanup();
380 }
381 cleanup_promise->set_value();
382 });
383
384 if (cleanup_future.wait_for(std::chrono::seconds(2)) == std::future_status::timeout) {
385 perform_cleanup();
386 }
387 } else {
388 perform_cleanup();
389 }
390
391 if (owns_ioc_ && ioc_thread_.joinable()) {
392 ioc_thread_.join();
393 ioc_.restart();
394 }
395 }

424void TcpServer::start() {
425 auto impl = get_impl();
426 auto current = impl->state_.state();
427 if (current == base::LinkState::Listening || current == base::LinkState::Connected ||
428 current == base::LinkState::Connecting) {
429 return;
430 }
431 impl->stopping_.store(false);
432
433 if (impl->uses_global_ioc_) {
434 auto& manager = concurrency::IoContextManager::instance();
435 if (!manager.is_running()) {
436 manager.start();
437 }
438 }
439
440 if (!impl->acceptor_) {
441 impl->state_.set_state(base::LinkState::Error);
442 impl->notify_state();
443 return;
444 }
445
446 if (impl->owns_ioc_) {
447 impl->ioc_thread_ = std::thread([impl] { impl->ioc_.run(); });
448 }
449 auto self = shared_from_this();
450 if (impl->ioc_.get_executor().running_in_this_thread()) {
451 if (!impl->stopping_.load()) {
452 impl->attempt_port_binding(self, 0);
453 }
454 } else {
455 net::dispatch(impl->ioc_, [self] {
456 auto impl = self->get_impl();
457 if (impl->stopping_.load()) return;
458 impl->attempt_port_binding(self, 0);
459 });
460 }
461}

543bool TcpServer::broadcast(std::string_view message) {
544 auto shared_data = std::make_shared<const std::vector<uint8_t>>(message.begin(), message.end());
545 auto impl = get_impl();
546 std::lock_guard<std::mutex> lock(impl->sessions_mutex_);
547 bool sent = false;
548 for (auto& entry : impl->sessions_) {
549 auto& session = entry.second;
550 if (session && session->alive()) {
551 session->async_write_shared(shared_data);
552 sent = true;
553 }
554 }
555 return sent;
