239 bool TcpServer::broadcast(std::string_view message) {
240  // Safe copy: create shared_ptr<vector> *before* async calls.
241  // This ensures the data persists even if the string_view source dies.
242  auto shared_data = std::make_shared<const std::vector<uint8_t>>(message.begin(), message.end());
243 
244  std::shared_lock<std::shared_mutex> lock(impl_->sessions_mutex_);
245  bool sent = false;
246  for (auto& entry : impl_->sessions_) {
247  auto& session = entry.second;
248  if (session && session->alive()) {
249  session->async_write_shared(shared_data);
250  sent = true;
251  }
252  }
253 
254  return sent;
255 }

257 bool TcpServer::send_to_client(size_t client_id, std::string_view message) {
258  // Safe copy: Convert string_view to a vector *synchronously* here.
259  // We can use async_write_move to transfer ownership.
260  std::vector<uint8_t> data(message.begin(), message.end());
261 
262  std::shared_lock<std::shared_mutex> lock(impl_->sessions_mutex_);
263 
264  auto it = impl_->sessions_.find(client_id);
265  if (it != impl_->sessions_.end() && it->second && it->second->alive()) {
266  // async_write_move takes rvalue ref and will move it into its internal queue
267  it->second->async_write_move(std::move(data));
268  return true;
269  }
270  UNILINK_LOG_DEBUG("tcp_server", "send_to_client",
271  "Send failed: client_id " + std::to_string(client_id) + " not found");
272  return false;
273 }

344 void TcpServer::Impl::start(std::shared_ptr<TcpServer> self) {
345  auto current = state_.get_state();
346  if (current == base::LinkState::Listening || current == base::LinkState::Connected ||
347  current == base::LinkState::Connecting) {
348  UNILINK_LOG_DEBUG("tcp_server", "start", "Start called while already active, ignoring");
349  return;
350  }
351  stopping_.store(false);
352 
353  if (uses_global_ioc_) {
354  auto& manager = concurrency::IoContextManager::instance();
355  if (!manager.is_running()) {
356  manager.start();
357  }
358  }
359 
360  if (!acceptor_) {
361  UNILINK_LOG_ERROR("tcp_server", "start", "Acceptor is null");
362  diagnostics::error_reporting::report_system_error("tcp_server", "start", "Acceptor is null");
363  state_.set_state(base::LinkState::Error);
364  notify_state();
365  return;
366  }
367 
368  if (owns_ioc_) {
369  ioc_thread_ = std::thread([this] { ioc_.run(); });
370  }
371 
372  if (ioc_.get_executor().running_in_this_thread()) {
373  if (!stopping_.load()) {
374  attempt_port_binding(self, 0);
375  }
376  } else {
377  net::dispatch(ioc_, [self] {
378  if (self->impl_->stopping_.load()) return;
379  self->impl_->attempt_port_binding(self, 0);
380  });
381  }
382 }

384 void TcpServer::Impl::stop(std::shared_ptr<TcpServer> self) {
385  if (stopping_.exchange(true)) {
386  return;
387  }
388 
389  {
390  std::unique_lock<std::shared_mutex> lock(sessions_mutex_);
391  on_bytes_ = nullptr;
392  on_state_ = nullptr;
393  on_bp_ = nullptr;
394  on_multi_connect_ = nullptr;
395  on_multi_data_ = nullptr;
396  on_multi_disconnect_ = nullptr;
397  }
398 
399  auto cleanup = [this, self]() {
400  boost::system::error_code ec;
401  if (acceptor_ && acceptor_->is_open()) {
402  acceptor_->close(ec);
403  }
404 
405  std::vector<std::shared_ptr<TcpServerSession>> sessions_copy;
406  {
407  std::unique_lock<std::shared_mutex> lock(sessions_mutex_);
408  sessions_copy.reserve(sessions_.size());
409  for (auto& kv : sessions_) {
410  sessions_copy.push_back(kv.second);
411  }
412  sessions_.clear();
413  current_session_.reset();
414  }
415 
416  for (auto& session : sessions_copy) {
417  if (session) {
418  session->stop();
419  }
420  }
421 
422  state_.set_state(base::LinkState::Closed);
423  };
424 
425  const bool in_ioc_thread = ioc_.get_executor().running_in_this_thread();
426 
427  if (in_ioc_thread) {
428  cleanup();
429  if (owns_ioc_) {
430  ioc_.stop();
431  }
432  return;
433  }
434 
435  bool dispatched = false;
436  const bool has_active_ioc_thread =
437  owns_ioc_ || (uses_global_ioc_ && concurrency::IoContextManager::instance().is_running());
438 
439  if (has_active_ioc_thread && self) {
440  try {
441  auto cleanup_promise = std::make_shared<std::promise<void>>();
442  auto cleanup_future = cleanup_promise->get_future();
443 
444  net::dispatch(ioc_, [self, cleanup, cleanup_promise] {
445  cleanup();
446  cleanup_promise->set_value();
447  });
448 
449  cleanup_future.wait();
450  dispatched = true;
451  } catch (const std::bad_weak_ptr&) {
452  UNILINK_LOG_DEBUG("tcp_server", "stop", "shared_from_this() failed, performing synchronous cleanup");
453  } catch (const std::exception& e) {
454  UNILINK_LOG_ERROR("tcp_server", "stop", "Dispatch failed: " + std::string(e.what()));
455  }
456  }
457 
458  if (!dispatched) {
459  cleanup();
460  }
461 
462  if (owns_ioc_ && ioc_thread_.joinable()) {
463  // Avoid calling ioc_.stop() to allow handlers to complete normally (prevents
464  // socket destruction during io_context destruction)
465  ioc_thread_.join();
466  ioc_.restart();
467  }
468 }

475 void TcpServer::Impl::attempt_port_binding(std::shared_ptr<TcpServer> self, int retry_count) {
476  if (stopping_.load()) return;
477  boost::system::error_code ec;
478 
479  if (retry_count > 0) {
480  UNILINK_LOG_DEBUG("tcp_server", "bind",
481  "Attempting port binding - retry enabled: " + std::to_string(cfg_.enable_port_retry) +
482  ", max retries: " + std::to_string(cfg_.max_port_retries) +
483  ", retry count: " + std::to_string(retry_count));
484  }
485 
486  if (!acceptor_->is_open()) {
487  acceptor_->open(tcp::v4(), ec);
488  if (ec) {
489  UNILINK_LOG_ERROR("tcp_server", "open", "Failed to open acceptor: " + ec.message());
490  diagnostics::error_reporting::report_connection_error("tcp_server", "open", ec, false);
491  state_.set_state(base::LinkState::Error);
492  notify_state();
493  return;
494  }
495  }
496 
497  acceptor_->bind(tcp::endpoint(tcp::v4(), cfg_.port), ec);
498  if (ec) {
499  if (cfg_.enable_port_retry && retry_count < cfg_.max_port_retries) {
500  UNILINK_LOG_WARNING("tcp_server", "bind",
501  "Failed to bind to port " + std::to_string(cfg_.port) + " (attempt " +
502  std::to_string(retry_count + 1) + "/" + std::to_string(cfg_.max_port_retries) + "): " +
503  ec.message() + ". Retrying in " + std::to_string(cfg_.port_retry_interval_ms) + "ms...");
504 
505  auto timer = std::make_shared<net::steady_timer>(ioc_);
506  timer->expires_after(std::chrono::milliseconds(cfg_.port_retry_interval_ms));
507  timer->async_wait([self, retry_count, timer](const boost::system::error_code& timer_ec) {
508  if (!timer_ec) {
509  self->impl_->attempt_port_binding(self, retry_count + 1);
510  }
511  });
512  return;
513  } else {
514  std::string error_msg = "Failed to bind to port: " + std::to_string(cfg_.port) + " - " + ec.message();
515  if (cfg_.enable_port_retry) {
516  error_msg += " (after " + std::to_string(retry_count) + " retries)";
517  }
518  UNILINK_LOG_ERROR("tcp_server", "bind", error_msg);
519  diagnostics::error_reporting::report_connection_error("tcp_server", "bind", ec, false);
520  state_.set_state(base::LinkState::Error);
521  notify_state();
522  return;
523  }
524  }
525 
526  acceptor_->listen(boost::asio::socket_base::max_listen_connections, ec);
527  if (ec) {
528  UNILINK_LOG_ERROR("tcp_server", "listen",
529  "Failed to listen on port: " + std::to_string(cfg_.port) + " - " + ec.message());
530  diagnostics::error_reporting::report_connection_error("tcp_server", "listen", ec, false);
531  state_.set_state(base::LinkState::Error);
532  notify_state();
533  return;
534  }
535 
536  if (retry_count > 0) {
537  UNILINK_LOG_INFO("tcp_server", "bind",
538  "Successfully bound to port " + std::to_string(cfg_.port) + " after " +
539  std::to_string(retry_count) + " retries");
540  } else {
541  UNILINK_LOG_INFO("tcp_server", "bind", "Successfully bound to port " + std::to_string(cfg_.port));
542  }
543 
544  state_.set_state(base::LinkState::Listening);
545  notify_state();
546  do_accept(self);
547 }

549 void TcpServer::Impl::do_accept(std::shared_ptr<TcpServer> self) {
550  if (stopping_.load() || !acceptor_ || !acceptor_->is_open()) return;
551 
552  acceptor_->async_accept([self](auto ec, tcp::socket sock) {
553  if (self->impl_->stopping_.load()) {
554  return;
555  }
556  if (ec) {
557  if (ec == boost::asio::error::operation_aborted) {
558  UNILINK_LOG_DEBUG("tcp_server", "accept", "Accept canceled (server shutting down)");
559  } else {
560  UNILINK_LOG_ERROR("tcp_server", "accept", "Accept error: " + ec.message());
561  diagnostics::error_reporting::report_connection_error("tcp_server", "accept", ec, true);
562  self->impl_->state_.set_state(base::LinkState::Error);
563  self->impl_->notify_state();
564  }
565  if (!self->impl_->state_.is_state(base::LinkState::Closed) && !self->impl_->stopping_.load()) {
566  auto timer = std::make_shared<net::steady_timer>(self->impl_->ioc_);
567  timer->expires_after(std::chrono::milliseconds(100));
568  timer->async_wait([self, timer](const boost::system::error_code&) {
569  if (!self->impl_->stopping_.load()) {
570  self->impl_->do_accept(self);
571  }
572  });
573  }
574  return;
575  }
576 
577  boost::system::error_code ep_ec;
578  auto rep = sock.remote_endpoint(ep_ec);
579  std::string client_info = "unknown";
580  if (!ep_ec) {
581  client_info = rep.address().to_string() + ":" + std::to_string(rep.port());
582  }
583 
584  if (self->impl_->client_limit_enabled_) {
585  std::unique_lock<std::shared_mutex> lock(self->impl_->sessions_mutex_);
586  if (self->impl_->sessions_.size() >= self->impl_->max_clients_) {
587  UNILINK_LOG_WARNING("tcp_server", "accept",
588  "Client connection rejected - server at capacity (" +
589  std::to_string(self->impl_->sessions_.size()) + "/" +
590  std::to_string(self->impl_->max_clients_) + "): " + client_info);
591 
592  boost::system::error_code close_ec;
593  sock.close(close_ec);
594  if (close_ec) {
595  UNILINK_LOG_DEBUG("tcp_server", "accept", "Error closing rejected socket: " + close_ec.message());
596  }
597 
598  self->impl_->paused_accept_ = true;
599  return;
600  }
601  }
602 
603  if (!ep_ec) {
604  UNILINK_LOG_INFO("tcp_server", "accept", "Client connected: " + client_info);
605  } else {
606  UNILINK_LOG_INFO("tcp_server", "accept", "Client connected (endpoint unknown)");
607  }
608 
609  auto new_session = std::make_shared<TcpServerSession>(self->impl_->ioc_, std::move(sock),
610  self->impl_->cfg_.backpressure_threshold);
611 
612  size_t client_id;
613  {
614  std::unique_lock<std::shared_mutex> lock(self->impl_->sessions_mutex_);
615 
616  client_id = self->impl_->next_client_id_.fetch_add(1);
617  self->impl_->sessions_.emplace(client_id, new_session);
618  self->impl_->current_session_ = new_session;
619  }
620 
621  new_session->on_bytes([self, client_id](memory::ConstByteSpan data) {
622  OnBytes cb;
623  MultiClientDataHandler multi_cb;
624  {
625  std::shared_lock<std::shared_mutex> lock(self->impl_->sessions_mutex_);
